311 void KeyMappingManager::saveNode( QXmlStreamWriter& stream, const QDomNode& domNode ) {
312  if ( stream.hasError() ) { return; }
313 
314  if ( domNode.isElement() )
315  {
316  const QDomElement domElement = domNode.toElement();
317  if ( !domElement.isNull() )
318  {
319  auto tagName = domElement.tagName().toStdString();
320  stream.writeStartElement( domElement.tagName() );
321 
322  if ( tagName == "keymap" )
323  {
324 
325  auto saveAttrib = [&domElement, &stream]( const QString& attribName,
326  const QString& attribDefault,
327  bool optional = false ) {
328  QString attribValue = domElement.attribute( attribName, attribDefault );
329  if ( optional && attribValue == attribDefault ) { return false; }
330  stream.writeAttribute( attribName, attribValue );
331  return true;
332  };
333 
334  if ( !saveAttrib( "context", "" ) )
335  {
336  LOG( logERROR ) << "Error, missing context when saving keymap element";
337  return;
338  }
339  if ( !saveAttrib( "action", "" ) )
340  {
341  LOG( logERROR ) << "Error, missing action when saving keymap element";
342  return;
343  }
344  saveAttrib( "buttons", "" );
345  saveAttrib( "modifiers", "" );
346  saveAttrib( "key", "" );
347  saveAttrib( "wheel", "false", true );
348  }
349  else
350  {
351  if ( domElement.hasAttributes() )
352  {
353  QMap<QString, QString> attributes;
354  const QDomNamedNodeMap attributeMap = domElement.attributes();
355  for ( int i = 0; i < attributeMap.count(); ++i )
356  {
357  const QDomNode attribute = attributeMap.item( i );
358  attributes.insert( attribute.nodeName(), attribute.nodeValue() );
359  }
360 
361  QMap<QString, QString>::const_iterator i = attributes.constBegin();
362  while ( i != attributes.constEnd() )
363  {
364  stream.writeAttribute( i.key(), i.value() );
365  ++i;
366  }
367  }
368  }
369 
370  if ( domElement.hasChildNodes() )
371  {
372  QDomNode elementChild = domElement.firstChild();
373  while ( !elementChild.isNull() )
374  {
375  saveNode( stream, elementChild );
376  elementChild = elementChild.nextSibling();
377  }
378  }
379 
380  stream.writeEndElement();
381  }
382  }
383  else if ( domNode.isComment() )
384  { stream.writeComment( domNode.nodeValue() ); }
385  else if ( domNode.isText() )
386  { stream.writeCharacters( domNode.nodeValue() ); }
387 }
