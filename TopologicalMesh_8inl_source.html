637 template <typename T>
638 void TopologicalMesh::interpolatePropsOnFaces(
639  FaceHandle fh,
640  const std::vector<OpenMesh::HPropHandleT<T>>& hProps,
641  const std::vector<OpenMesh::FPropHandleT<T>>& fProps ) {
642  auto heh = halfedge_handle( fh );
643  const size_t v = valence( fh );
644 
645  // init sum to first
646  for ( size_t j = 0; j < fProps.size(); ++j )
647  {
648  auto hp = hProps[j];
649  auto fp = fProps[j];
650  property( fp, fh ) = property( hp, heh );
651  }
652  heh = next_halfedge_handle( heh );
653  // sum others
654  for ( size_t i = 1; i < v; ++i )
655  {
656  for ( size_t j = 0; j < fProps.size(); ++j )
657  {
658  auto hp = hProps[j];
659  auto fp = fProps[j];
660  property( fp, fh ) += property( hp, heh );
661  }
662  heh = next_halfedge_handle( heh );
663  }
664  // normalize
665  for ( size_t j = 0; j < fProps.size(); ++j )
666  {
667  auto fp = fProps[j];
668  property( fp, fh ) = property( fp, fh ) / v;
669  }
