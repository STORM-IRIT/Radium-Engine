48 void Mesh::loadGeometry( const Core::Vector3Array& vertices, const std::vector<uint>& indices ) {
49  // Do not remove this function to force everyone to use TriangleMesh.
50  // ... because we have some line meshes as well...
52  Core::Geometry::TriangleMesh mesh;
53 
54  auto nIdx = indices.size();
55 
56  if ( indices.empty() )
57  {
58  m_numElements = vertices.size();
59  setRenderMode( RM_POINTS );
60  }
61  else
62  m_numElements = nIdx;
63  mesh.setVertices( vertices );
64 
65  // Check that when loading a TriangleMesh we actually have triangles or lines.
66  CORE_ASSERT( m_renderMode != GL_TRIANGLES || nIdx % 3 == 0,
67  "There should be 3 indices per triangle " );
68  CORE_ASSERT( m_renderMode != GL_LINES || nIdx % 2 == 0, "There should be 2 indices per line" );
69  CORE_ASSERT( m_renderMode != GL_LINES_ADJACENCY || nIdx % 4 == 0,
70  "There should be 4 indices per line adjacency" );
71  Core::Geometry::TriangleMesh::IndexContainerType mindices;
72  for ( uint i = 0; i < indices.size(); i = i + 3 )
73  {
74  // We store all indices in order. This means that for lines we have
75  // (L00, L01, L10), (L11, L20, L21) etc. We fill the missing by wrapping around indices.
76  mindices.push_back( {indices[i], indices[( i + 1 ) % nIdx], indices[( i + 2 ) % nIdx]} );
77  }
78 
79  mesh.setIndices( std::move( mindices ) );
80  m_dataDirty.clear();
81  m_vbos.clear();
82 
84  loadGeometry( std::move( mesh ) );
85 }
