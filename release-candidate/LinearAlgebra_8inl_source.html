110 getOrthogonalVectors( const Vector3& fx, Eigen::Ref<Vector3> fy, Eigen::Ref<Vector3> fz ) {
111  // taken from [Duff et al. 17] Building An Orthonormal Basis, Revisited. JCGT. 2017
112  Scalar sign = std::copysign( Scalar( 1.0 ), fx( 2 ) );
113  const Scalar a = Scalar( -1.0 ) / ( sign + fx( 2 ) );

133 // Reference : Paolo Baerlocher, Inverse Kinematics techniques for interactive posture control
134 // of articulated figures (PhD Thesis), p.138. EPFL, 2001.
135 // http://www0.cs.ucl.ac.uk/research/equator/papers/Documents2002/Paolo%20Baerlocher_Thesis_2001.pdf
136 inline void getSwingTwist( const Quaternion& in, Quaternion& swingOut, Quaternion& twistOut ) {
137  // singular case.
138  if ( UNLIKELY( in.z() == 0 && in.w() == 0 ) ) {
139  twistOut = in;
140  swingOut.setIdentity();
141  }
142  else {
143  const Scalar gamma = std::atan2( in.z(), in.w() );
144  // beta = atan2 ( sqrt ( in.x^2 + in.y^2), sqrt ( in.z^2 + in.w^2))
145  const Scalar beta =
146  std::atan2( in.coeffs().head<2>().norm(), in.coeffs().tail<2>().norm() );
147 
148  // k = 2 / sinc(beta) = 2 / ( sin(beta)/ beta))
149  const Scalar k = 2.f * beta / std::sin( beta );
150 
151  // The parentheses here are important. because otherwise k * rotation2D would
152  // be evaluated first (which would yield the rotation of angle k * gamma).
153  const Vector2 sxy = k * ( Eigen::Rotation2D<Scalar>( gamma ) * in.coeffs().head<2>() );
154 
155  AngleAxis twist( 2 * gamma, Vector3::UnitZ() );
156  twistOut = twist;
157 
158  Vector3 swingAxis( Vector3::Zero() );
159  swingAxis.head<2>() = sxy.normalized();
160 
161  AngleAxis swing( sxy.norm(), swingAxis );
162  swingOut = swing;
163  }
164 }

167 inline Matrix4 lookAt( const Vector3& position, const Vector3& target, const Vector3& up ) {
168  Matrix3 R;
169  R.col( 2 ) = ( position - target ).normalized();
170  R.col( 0 ) = up.cross( R.col( 2 ) ).normalized();
171  R.col( 1 ) = R.col( 2 ).cross( R.col( 0 ) );
172 
173  Matrix4 result = Matrix4::Zero();
174  result.topLeftCorner<3, 3>() = R.transpose();
175  result.topRightCorner<3, 1>() = -R.transpose() * position;
176  result( 3, 3 ) = 1.0;
177 
178  return result;
179 }

181 inline Matrix4 perspective( Scalar fovy, Scalar aspect, Scalar znear, Scalar zfar ) {
182  Scalar theta = fovy * 0.5f;
183  Scalar range = zfar - znear;
184  Scalar invtan = 1.f / std::tan( theta );
185 
186  Matrix4 result = Matrix4::Zero();
187  result( 0, 0 ) = invtan / aspect;
188  result( 1, 1 ) = invtan;
189  result( 2, 2 ) = -( znear + zfar ) / range;
190  result( 3, 2 ) = -1.f;
191  result( 2, 3 ) = -2 * znear * zfar / range;
192  result( 3, 3 ) = 0.0;
193 
194  return result;
195 }
