92 void SkeletonComponent::handleAnimationLoading(
93  const std::vector<Core::Asset::AnimationData*>& data ) {
94  CORE_ASSERT( ( m_skel.size() != 0 ), "A Skeleton should be loaded first." );
95  m_animations.clear();
96  m_animations.reserve( data.size() );
97 
98  auto pose = m_skel.getPose( SpaceType::LOCAL );
99  for ( uint n = 0; n < data.size(); ++n ) {
100  m_animations.emplace_back();
101  m_animations.back().reserve( m_skel.size() );
102  auto handleAnim = data[n]->getHandleData();
103  for ( uint i = 0; i < m_skel.size(); ++i ) {
104  auto it =
105  std::find_if( handleAnim.cbegin(), handleAnim.cend(), [this, i]( const auto& ha ) {
106  return m_skel.getLabel( i ) == ha.m_name;
107  } );
108  if ( it == handleAnim.cend() ) {
109  m_animations.back().push_back( KeyFramedValue( 0_ra, pose[i] ) );
110  }
111  else {
112  m_animations.back().push_back( it->m_anim );
113  }
114  }
115  }
116  if ( m_animations.empty() ) {
117  m_animations.emplace_back();
118  for ( uint i = 0; i < m_skel.size(); ++i ) {
119  m_animations[0].push_back( KeyFramedValue( 0_ra, pose[i] ) );
120  }
121  }
122  m_animationID = 0;
123  m_animationTime = 0_ra;
124 }

158 void SkeletonComponent::update( Scalar t ) {
159  m_wasReset = Core::Math::areApproxEqual( t, 0_ra );
160  if ( m_wasReset ) {
161  m_animationTime = t;
162  m_skel.setPose( m_refPose, SpaceType::LOCAL );
163 
164  updateDisplay();
165  return;
166  }
167 
168  m_animationTime = m_speed * t;
169  Scalar lastTime = 0;
170  if ( !m_animations.empty() ) {
171  // m_animationID is always < m_animation.size() unless m_animations.empty()
172  for ( const auto& boneAnim : m_animations[m_animationID] ) {
173  lastTime = std::max( lastTime, *boneAnim.getTimes().rbegin() );
174  }
175  }
176  if ( m_autoRepeat ) {
177  if ( !m_pingPong ) { m_animationTime = std::fmod( m_animationTime, lastTime ); }
178  else {
179  m_animationTime = std::fmod( m_animationTime, 2 * lastTime );
180  if ( m_animationTime > lastTime ) { m_animationTime = 2 * lastTime - m_animationTime; }
181  }
182  }
183  else if ( m_pingPong ) {
184  if ( m_animationTime > 2 * lastTime ) { m_animationTime = 0_ra; }
185  else if ( m_animationTime > lastTime ) {
186  m_animationTime = 2 * lastTime - m_animationTime;
187  }
188  }
189 
190  // get the current pose from the animation
191  Core::Animation::Pose pose = m_skel.getPose( SpaceType::LOCAL );
192  if ( !m_animations.empty() ) {
193 #pragma omp parallel for
194  for ( int i = 0; i < int( m_animations[m_animationID].size() ); ++i ) {
195  pose[uint( i )] = m_animations[m_animationID][uint( i )].at(
196  m_animationTime, Core::Animation::linearInterpolate<Core::Transform> );
197  }
198  }
199  else {
200  pose = m_refPose;
201  }
202  m_skel.setPose( pose, SpaceType::LOCAL );
203 
204  updateDisplay();
205 }

301 void SkeletonComponent::setupSkeletonDisplay() {
302  m_renderObjects.clear();
303  m_boneDrawables.clear();
304  if ( !s_boneMesh ) {
305  s_boneMesh = std::make_shared<Data::Mesh>( "Bone Mesh" );
306  s_boneMesh->loadGeometry( makeBoneShape() );
307  s_boneMaterial.reset( new Data::BlinnPhongMaterial( "Bone Material" ) );
308  s_boneMaterial->m_kd = Color::Grey( 0.4_ra );
309  s_boneMaterial->m_ks = Color::Black();
310  s_boneRenderTechnique.reset( new Rendering::RenderTechnique );
311  s_boneRenderTechnique->setConfiguration(
312  *Data::ShaderConfigurationFactory::getConfiguration( "BlinnPhong" ) );
313  s_boneRenderTechnique->setParametersProvider( s_boneMaterial );
314  }
315  for ( uint i = 0; i < m_skel.size(); ++i ) {
316  if ( !m_skel.m_graph.isLeaf( i ) && !m_skel.m_graph.isRoot( i ) &&
317  m_skel.getLabel( i ).find( "_$AssimpFbx$_" ) == std::string::npos ) {
318  std::string name = m_skel.getLabel( i ) + "_" + std::to_string( i );
319  auto ro = new Engine::Rendering::RenderObject(
320  name, this, Rendering::RenderObjectType::Geometry );
321  ro->setRenderTechnique( s_boneRenderTechnique );
322  ro->setMesh( s_boneMesh );
323  ro->setMaterial( s_boneMaterial );
324  ro->setXRay( false );
325  addRenderObject( ro );
326  m_boneMap[m_renderObjects.back()] = i;
327  m_boneDrawables.push_back( ro );
328  }
329  else {
330  LOG( logDEBUG ) << "Bone " << m_skel.getLabel( i ) << " not displayed.";
331  }
332  }
333  updateDisplay();
334 }

357 void SkeletonComponent::updateDisplay() {
358  for ( auto& bone : m_boneDrawables ) {
359  uint boneIdx = m_boneMap.at( bone->getIndex() );
360  Core::Vector3 start;
361  Core::Vector3 end;
362  m_skel.getBonePoints( boneIdx, start, end );
363 
364  Core::Transform scale = Core::Transform::Identity();
365  scale.scale( ( end - start ).norm() );
366 
367  Core::Quaternion rot =
368  Core::Quaternion::FromTwoVectors( Core::Vector3::UnitZ(), end - start );
369 
370  Core::Transform boneTransform = m_skel.getTransform( boneIdx, SpaceType::MODEL );
371  Core::Matrix3 rotation = rot.toRotationMatrix();
372  Core::Transform drawTransform;
373  drawTransform.linear() = rotation;
374  drawTransform.translation() = boneTransform.translation();
375 
376  bone->setLocalTransform( drawTransform * scale );
377  }
378 }
