126 void TaskQueue::detectCycles() {
127 #if defined( CORE_DEBUG )
128  // Do a depth-first search of the nodes.
129  std::vector<bool> visited( m_tasks.size(), false );
130  std::stack<TaskId> pending;
131 
132  for ( uint id = 0; id < m_tasks.size(); ++id ) {
133  if ( m_dependencies[id].size() == 0 ) { pending.push( TaskId( id ) ); }
134  }
135 
136  // If you hit this assert, there are tasks in the list but
137  // all tasks have dependencies so no task can start.
138  CORE_ASSERT( m_tasks.empty() || !pending.empty(), "No free tasks." );
139 
140  while ( !pending.empty() ) {
141  TaskId id = pending.top();
142  pending.pop();
143 
144  // The task has already been visited. It means there is a cycle in the task graph.
145  CORE_ASSERT( !( visited[id] ), "Cycle detected in tasks !" );
146 
147  visited[id] = true;
148  for ( const auto& dep : m_dependencies[id] ) {
149  pending.push( dep );
150  }
151  }
152 #endif
153 }

178 void TaskQueue::runTasksInThisThread() {
179 
180  // lock task queue so no other worker can start working while this thread do the job.
181  std::lock_guard<std::mutex> lock( m_taskQueueMutex );
182 
183  // Add pending dependencies.
184  resolveDependencies();
185 
186  // Do a debug check
187  detectCycles();
188 
189  // Enqueue all tasks with no dependencies.
190  for ( uint t = 0; t < m_tasks.size(); ++t ) {
191  if ( m_remainingDependencies[t] == 0 ) { queueTask( TaskId { t } ); }
192  }
193  while ( !m_taskQueue.empty() ) {
194  TaskId task { m_taskQueue.back() };
195  m_taskQueue.pop_back();
196 
197  // Run task
198  m_timerData[task].start = Utils::Clock::now();
199  m_timerData[task].threadId = 0;
200  m_tasks[task]->process();
201  m_timerData[task].end = Utils::Clock::now();
202 
203  for ( auto t : m_dependencies[task] ) {
204  uint& nDepends = m_remainingDependencies[t];
205  CORE_ASSERT( nDepends > 0, "Inconsistency in dependencies" );
206  --nDepends;
207  if ( nDepends == 0 ) { queueTask( t ); }
208  }
209  }
210  flushTaskQueue();
211 }

241 void TaskQueue::runThread( uint id ) {
242  while ( true ) {
243  TaskId task;
244 
245  // Acquire mutex.
246  {
247  std::unique_lock<std::mutex> lock( m_taskQueueMutex );
248 
249  // Wait for a new task
250  m_threadNotifier.wait( lock,
251  [this]() { return m_shuttingDown || !m_taskQueue.empty(); } );
252  // If the task queue is shutting down we quit, releasing
253  // the lock.
254  if ( m_shuttingDown ) { return; }
255 
256  // If we are here it means we got a task
257  task = m_taskQueue.back();
258  m_taskQueue.pop_back();
259  ++m_processingTasks;
260  CORE_ASSERT( task.isValid() && task < m_tasks.size(), "Invalid task" );
261  }
262  // Release mutex.
263 
264  // Run task
265  m_timerData[task].start = Utils::Clock::now();
266  m_timerData[task].threadId = id;
267  m_tasks[task]->process();
268  m_timerData[task].end = Utils::Clock::now();
269 
270  // Critical section : mark task as finished and en-queue dependencies.
271  uint newTasks = 0;
272  {
273  std::unique_lock<std::mutex> lock( m_taskQueueMutex );
274  for ( auto t : m_dependencies[task] ) {
275  uint& nDepends = m_remainingDependencies[t];
276  CORE_ASSERT( nDepends > 0, "Inconsistency in dependencies" );
277  --nDepends;
278  if ( nDepends == 0 ) {
279  queueTask( t );
280  ++newTasks;
281  }
282  // TODO :Easy optimization : grab one of the new task and process it immediately.
283  }
284  --m_processingTasks;
285  }
286  // If we added new tasks, we wake up one thread to execute it.
287  if ( newTasks > 0 ) { m_threadNotifier.notify_one(); }
288  } // End of while(true)
289 }
