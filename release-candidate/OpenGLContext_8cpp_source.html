43  glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE );
44  m_glfwContext =
45  glfwCreateWindow( size[0], size[1], "Radium CommandLine Context", nullptr, nullptr );
46  }
47  const char* description;
48  int code = glfwGetError( &description );
49  if ( code == GLFW_VERSION_UNAVAILABLE ) {
50  // if the requested opengl version is not available, try to get the recommended version
51  // (4.1)
52  glfwWindowHint( GLFW_CONTEXT_VERSION_MAJOR, 4 );
53  glfwWindowHint( GLFW_CONTEXT_VERSION_MINOR, 1 );
54  m_glfwContext =
55  glfwCreateWindow( size[0], size[1], "Radium CommandLine Context", nullptr, nullptr );
56  code = glfwGetError( &description );
57  }
58  if ( code != GLFW_NO_ERROR ) {
59  std::cerr << "OpenGL context creation failed. Terminate execution." << std::endl;
60  error( code, description );
61  glfwTerminate();
62  std::exit( -1 );
63  }
64  else {
65  // Initialize globjects (internally initializes glbinding, and registers the current
66  // context)
67  glfwMakeContextCurrent( m_glfwContext );
68  globjects::init( []( const char* name ) { return glfwGetProcAddress( name ); } );
69  glfwSetWindowUserPointer( m_glfwContext, this );

190 void OpenGLContext::renderLoop( std::function<void( float )> render ) {
191  double prevFrameDate = glfwGetTime();
192  double curFrameDate;
193 
194  int width, height;
195  glfwGetFramebufferSize( m_glfwContext, &width, &height );
196  glViewport( 0, 0, width, height );
197 
198  while ( !glfwWindowShouldClose( m_glfwContext ) ) {
199  curFrameDate = glfwGetTime();
200  render( curFrameDate - prevFrameDate );
201  prevFrameDate = curFrameDate;
202 
203  glfwSwapBuffers( m_glfwContext );
204  processEvents();
205  }
