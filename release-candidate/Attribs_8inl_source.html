270 template <typename T>
271 AttribHandle<T> AttribManager::addAttrib( const std::string& name ) {
272  // does the attrib already exist?
273  AttribHandle<T> h = findAttrib<T>( name );
274  if ( isValid( h ) ) return h;
275 
276  // create the attrib
277  smart_pointer_type attrib = std::make_unique<Attrib<T>>( name );
278 
279  // look for a free slot
280  auto it = std::find_if(
281  m_attribs.begin(), m_attribs.end(), []( const auto& attr ) { return !attr; } );
282  if ( it != m_attribs.end() ) {
283  it->swap( attrib );
284  h.m_idx = std::distance( m_attribs.begin(), it );
285  }
286  else {
287  m_attribs.push_back( std::move( attrib ) );
288  h.m_idx = m_attribs.size() - 1;
289  }
290  m_attribsIndex[name] = h.m_idx;
291  h.m_name = name;
292  ++m_numAttribs;
293 
294  notify( name );
295  return h;
296 }
