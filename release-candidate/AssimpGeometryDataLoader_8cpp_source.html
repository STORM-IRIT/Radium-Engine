102  const Core::Transform& parentFrame,
103  const std::map<uint, size_t>& indexTable,
104  std::vector<std::unique_ptr<GeometryData>>& data ) const {
105  const uint child_size = node->mNumChildren;

107  if ( ( child_size == 0 ) && ( mesh_size == 0 ) ) { return; }

116  for ( uint i = 0; i < child_size; ++i ) {
117  loadMeshFrame( node->mChildren[i], frame, indexTable, data );

259 void AssimpGeometryDataLoader::loadMaterial( const aiMaterial& material,
260  GeometryData& data ) const {
261 
262  std::string matName;
263  aiString assimpName;
264  if ( AI_SUCCESS == material.Get( AI_MATKEY_NAME, assimpName ) ) {
265  matName = assimpName.C_Str();
266  }
267  // Radium V2 : use AI_MATKEY_SHADING_MODEL to select the apropriate model
268  // (http://assimp.sourceforge.net/lib_html/material_8h.html#a93e23e0201d6ed86fb4287e15218e4cf)
269  auto blinnPhongMaterial = new BlinnPhongMaterialData( matName );
270  if ( AI_DEFAULT_MATERIAL_NAME != matName ) {
271  aiColor4D color;
272  float shininess;
273  float opacity;
274  aiString name;
275 
276  if ( AI_SUCCESS == material.Get( AI_MATKEY_COLOR_DIFFUSE, color ) ) {
277  blinnPhongMaterial->m_hasDiffuse = true;
278  blinnPhongMaterial->m_diffuse = assimpToCore( color );
279  }
280 
281  if ( AI_SUCCESS == material.Get( AI_MATKEY_COLOR_SPECULAR, color ) ) {
282  blinnPhongMaterial->m_hasSpecular = true;
283  blinnPhongMaterial->m_specular = assimpToCore( color );
284  }
285 
286  if ( AI_SUCCESS == material.Get( AI_MATKEY_SHININESS, shininess ) ) {
287  blinnPhongMaterial->m_hasShininess = true;
288  // Assimp gives the Phong exponent, we use the Blinn-Phong exponent
289  blinnPhongMaterial->m_shininess = shininess * 4;
290  }
291 
292  if ( AI_SUCCESS == material.Get( AI_MATKEY_OPACITY, opacity ) ) {
293  blinnPhongMaterial->m_hasOpacity = true;
294  // NOTE(charly): Due to collada way of handling objects that have an alpha map, we must
295  // ensure
296  // we do not have zeros in here.
297  blinnPhongMaterial->m_opacity = opacity < 1e-5 ? 1 : opacity;
298  }
299 
300  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_DIFFUSE, 0 ), name ) ) {
301  blinnPhongMaterial->m_texDiffuse = m_filepath + "/" + assimpToCore( name );
302  blinnPhongMaterial->m_hasTexDiffuse = true;
303  }
304 
305  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_SPECULAR, 0 ), name ) ) {
306  blinnPhongMaterial->m_texSpecular = m_filepath + "/" + assimpToCore( name );
307  blinnPhongMaterial->m_hasTexSpecular = true;
308  }
309 
310  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_SHININESS, 0 ), name ) ) {
311  blinnPhongMaterial->m_texShininess = m_filepath + "/" + assimpToCore( name );
312  blinnPhongMaterial->m_hasTexShininess = true;
313  }
314 
315  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_NORMALS, 0 ), name ) ) {
316  blinnPhongMaterial->m_texNormal = m_filepath + "/" + assimpToCore( name );
317  blinnPhongMaterial->m_hasTexNormal = true;
318  }
319 
320  // Assimp loads objs bump maps as height maps, gj bro
321  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_HEIGHT, 0 ), name ) ) {
322  blinnPhongMaterial->m_texNormal = m_filepath + "/" + assimpToCore( name );
323  blinnPhongMaterial->m_hasTexNormal = true;
324  }
325 
326  if ( AI_SUCCESS == material.Get( AI_MATKEY_TEXTURE( aiTextureType_OPACITY, 0 ), name ) ) {
327  blinnPhongMaterial->m_texOpacity = m_filepath + "/" + assimpToCore( name );
328  blinnPhongMaterial->m_hasTexOpacity = true;

337 void AssimpGeometryDataLoader::loadGeometryData(
338  const aiScene* scene,
339  std::vector<std::unique_ptr<GeometryData>>& data ) {
340  const uint size = scene->mNumMeshes;
341  std::map<uint, std::size_t> indexTable;
342  std::set<std::string> usedNames;
343  for ( uint i = 0; i < size; ++i ) {
344  aiMesh* mesh = scene->mMeshes[i];
345  if ( mesh->HasPositions() ) {
346  auto geometry = new GeometryData();
347  loadMeshData( *mesh, *geometry, usedNames );
348  // This returns always true (see assimp documentation)
349  if ( scene->HasMaterials() ) {
350  const uint matID = mesh->mMaterialIndex;
351  if ( matID < scene->mNumMaterials ) {
352  aiMaterial* material = scene->mMaterials[matID];
353  loadMaterial( *material, *geometry );
354  }
355  }
356  data.push_back( std::unique_ptr<GeometryData>( geometry ) );
357  indexTable[i] = data.size() - 1;
358 
359  if ( m_verbose ) { geometry->displayInfo(); }
360  }
361  }
362  loadMeshFrame( scene->mRootNode, Core::Transform::Identity(), indexTable, data );
363 }
