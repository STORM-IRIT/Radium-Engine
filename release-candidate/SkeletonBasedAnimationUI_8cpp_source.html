140 void SkeletonBasedAnimationUI::postLoadFile( Engine::Scene::Entity* entity ) {
141  // reset current data and ui
142  m_selection = Engine::Scene::ItemEntry();
143  m_currentSkeleton = nullptr;
144  m_currentSkinnings.clear();
145  ui->tabWidget->setEnabled( false );
146  ui->m_skinning->setEnabled( false );
147  if ( !m_timeline ) { return; }
148  // register the animation keyframes of the first animation into the timeline
149  auto c = std::find_if(
150  entity->getComponents().begin(), entity->getComponents().end(), []( const auto& cmpt ) {
151  return ( dynamic_cast<Ra::Engine::Scene::SkeletonComponent*>( cmpt.get() ) != nullptr );
152  } );
153  if ( c != entity->getComponents().end() ) {
154  auto skel = static_cast<Ra::Engine::Scene::SkeletonComponent*>( ( *c ).get() );
155  auto& anim = skel->getAnimation( skel->getAnimationId() );
156  const auto& boneMap = skel->getBoneRO2idx();
157  for ( size_t j = 0; j < anim.size(); ++j ) {
158  auto it = std::find_if(
159  boneMap->begin(), boneMap->end(), [j]( const auto& b ) { return b.second == j; } );
160  if ( it == boneMap->end() ) { continue; } // end bone
161 
162  m_timeline->registerKeyFramedValue(
163  it->first,
164  KeyFramedValueController(
165  &anim[j],
166  "Animation_" + skel->getSkeleton()->getLabel( uint( j ) ),
167  [&anim, j, skel]( const Scalar& t ) {
168  anim[j].insertKeyFrame(
169  t, skel->getSkeleton()->getPose( HandleArray::SpaceType::LOCAL )[j] );
170  } ) ); // no update callback here
171  }
172  }
173  // update the timeline display interval to the bounding interval of all anims
174  Scalar startTime = std::numeric_limits<Scalar>::max();
175  Scalar endTime = 0;
176  for ( const auto& animComp : entity->getComponents() ) {
177  if ( auto skel = dynamic_cast<Ra::Engine::Scene::SkeletonComponent*>( animComp.get() ) ) {
178  auto [s, e] = skel->getAnimationTimeInterval();
179  startTime = std::min( startTime, s );
180  endTime = std::max( endTime, e );
181  }
182  }
183  m_timeline->onChangeStart( startTime );
184  m_timeline->onChangeEnd( endTime );
185 }
