35 struct memory_buffer : public std::streambuf {
36  char* p_start { nullptr };
37  char* p_end { nullptr };
38  size_t size;
39 
40  memory_buffer( char const* first_elem, size_t size_ ) :
41  p_start( const_cast<char*>( first_elem ) ), p_end( p_start + size_ ), size( size_ ) {
42  setg( p_start, p_start, p_end );
43  }
44 
45  pos_type seekoff( off_type off, std::ios_base::seekdir dir, std::ios_base::openmode ) override {
46  if ( dir == std::ios_base::cur )
47  gbump( static_cast<int>( off ) );
48  else
49  setg( p_start, ( dir == std::ios_base::beg ? p_start : p_end ) + off, p_end );
50  return gptr() - p_start;
51  }
52 
53  pos_type seekpos( pos_type pos, std::ios_base::openmode which ) override {
54  return seekoff( pos, std::ios_base::beg, which );
55  }

58 inline std::vector<uint8_t> read_file_binary( const std::string& pathToFile ) {
59  std::ifstream file( pathToFile, std::ios::binary );
60  std::vector<uint8_t> fileBufferBytes;
61 
62  if ( file.is_open() ) {
63  file.seekg( 0, std::ios::end );
64  size_t sizeBytes = file.tellg();
65  file.seekg( 0, std::ios::beg );
66  fileBufferBytes.resize( sizeBytes );
67  if ( file.read( (char*)fileBufferBytes.data(), sizeBytes ) ) return fileBufferBytes;
68  }
69  else
70  throw std::runtime_error( "could not open binary ifstream to path " + pathToFile );
71  return fileBufferBytes;
72 }

101 template <typename ContainerType>
102 void copyBufferToContainer( const std::shared_ptr<tinyply::PlyData>& buffer,
103  ContainerType& container ) {
104  if ( buffer && buffer->count != 0 ) {
105  switch ( buffer->t ) {
106  case tinyply::Type::FLOAT32: {
107  copyArrayToContainer<float>( buffer->buffer.get(), container, buffer->count );
108  } break;
109  case tinyply::Type::FLOAT64: {
110  copyArrayToContainer<double>( buffer->buffer.get(), container, buffer->count );
111  } break;
112  default:
113  LOG( logWARNING ) << "[TinyPLY] copyBufferToContainer - unsupported buffer type ..."
114  << tinyply::PropertyTable[buffer->t].str;
115  }
116  }

119 FileData* TinyPlyFileLoader::loadFile( const std::string& filename ) {
120 
121  std::unique_ptr<std::istream> file_stream;
122  std::vector<uint8_t> byte_buffer;
123 
124  // from https://github.com/ddiakopoulos/tinyply/blob/master/source/example.cpp
125  // For most files < 1gb, pre-loading the entire file upfront and wrapping it into a
126  // stream is a net win for parsing speed, about 40% faster.
127  const bool preload_into_memory = true;
128  if ( preload_into_memory ) {
129  byte_buffer = read_file_binary( filename );
130  file_stream.reset( new memory_stream( (char*)byte_buffer.data(), byte_buffer.size() ) );
131  }
132  else {
133  file_stream.reset( new std::ifstream( filename, std::ios::binary ) );
134  }
135 
136  if ( !file_stream || file_stream->fail() ) {
137  LOG( logINFO ) << "[TinyPLY] Could not open file [" << filename << "] Aborting"
138  << std::endl;
139  return nullptr;
140  }
141 
142  // Parse the ASCII header fields
143  tinyply::PlyFile file;
144  file.parse_header( *file_stream );
145 
146  auto elements = file.get_elements();
147  if ( std::any_of( elements.begin(), elements.end(), []( const auto& e ) -> bool {
148  return e.name == "face" && e.size != 0;
149  } ) ) {
150  // Mesh found. Let the other loaders handle it
151  LOG( logINFO ) << "[TinyPLY] Faces found. Aborting" << std::endl;
152  return nullptr;
153  }
154 
155  // we are now sure to have a point-cloud
156  FileData* fileData = new FileData( filename );
157  fileData->setVerbose( true );
158 
159  if ( !fileData->isInitialized() ) {
160  delete fileData;
161  LOG( logINFO ) << "[TinyPLY] Filedata cannot be initialized...";
162  return nullptr;
163  }
164 
165  if ( fileData->isVerbose() ) {
166  LOG( logINFO ) << "[TinyPLY] File Loading begin...";
167  LOG( logINFO ) << "....................................................................";
168  for ( auto c : file.get_comments() )
169  LOG( logINFO ) << "Comment: " << c;
170  for ( auto e : file.get_elements() ) {
171  LOG( logINFO ) << "element - " << e.name << " (" << e.size << ")";
172  for ( auto p : e.properties )
173  LOG( logINFO ) << "\tproperty - " << p.name << " ("
174  << tinyply::PropertyTable[p.propertyType].str << ")";
175  }
176  LOG( logINFO ) << "....................................................................";
177  }
178 
179  auto initBuffer = [&file]( const std::string& elementKey,
180  const std::vector<std::string> propertyKeys ) {
181  std::shared_ptr<tinyply::PlyData> ret;
182  try {
183  ret = file.request_properties_from_element( elementKey, propertyKeys );
184  }
185  catch ( const std::exception& e ) {
186  ret = nullptr;
187  LOG( logINFO ) << "[TinyPLY] " << e.what();
188  }
189  return ret;
190  };
191 
192  auto startTime { std::clock() };
193 
194  // a unique name is required by the component messaging system
195  static int nameId { 0 };
196  auto geomData = std::make_unique<GeometryData>( "PC_" + std::to_string( ++nameId ),
197  GeometryData::POINT_CLOUD );
198  geomData->setFrame( Core::Transform::Identity() );
199 
201  auto vertBuffer { initBuffer( "vertex", { "x", "y", "z" } ) };
202  // if there is no vertex prop, or their count is 0, then quit.
203  if ( !vertBuffer || vertBuffer->count == 0 ) {
204  delete fileData;
205  LOG( logINFO ) << "[TinyPLY] No vertex found";
206  return nullptr;
207  }
210  auto normalBuffer { initBuffer( "vertex", { "nx", "ny", "nz" } ) };
211  auto alphaBuffer { initBuffer( "vertex", { "alpha" } ) };
212  auto colorBuffer { initBuffer( "vertex", { "red", "green", "blue" } ) };
213 
215  std::vector<std::pair<std::string, std::shared_ptr<tinyply::PlyData>>> allAttributes;
216  const std::set<std::string> usedAttributes {
217  "x", "y", "z", "nx", "ny", "nz", "alpha", "red", "green", "blue" };
218  for ( const auto& e : file.get_elements() ) {
219  if ( e.name == "vertex" ) {
220  for ( const auto& p : e.properties ) {
221  bool exists = usedAttributes.find( p.name ) != usedAttributes.end();
222  if ( !exists ) {
223  allAttributes.emplace_back( p.name, initBuffer( "vertex", { p.name } ) );
224  }
225  }
226  break;
227  }
228  }
229 
230  // read requested buffers (and only those) from file content
231  file.read( *file_stream );
232  {
233  auto unlocker = geomData->getGeometry().vertexAttribs().getScopedLockState();
234  copyBufferToContainer( vertBuffer, geomData->getGeometry().verticesWithLock() );
235  copyBufferToContainer( normalBuffer, geomData->getGeometry().normalsWithLock() );
236  }
237 
238  auto& vertexAttribs = geomData->getGeometry().vertexAttribs();
239 
240  size_t colorCount = colorBuffer ? colorBuffer->count : 0;
241  if ( colorCount != 0 ) {
242  auto handle = vertexAttribs.addAttrib<Core::Vector4>(
243  Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::MeshAttrib::VERTEX_COLOR ) );
244  auto& container = vertexAttribs.getDataWithLock( handle );
245 
246  if ( alphaBuffer && alphaBuffer->count == colorCount ) {
247  uint8_t* al = alphaBuffer->buffer.get();
248  uint8_t* colors = colorBuffer->buffer.get();
249  for ( size_t i = 0; i < colorCount; ++i, al++, colors += 3 ) {
250  container.emplace_back( Scalar( colors[0] ) / 255_ra,
251  Scalar( colors[1] ) / 255_ra,
252  Scalar( colors[2] ) / 255_ra,
253  Scalar( *al ) / 255_ra );
254  }
255  }
256  else {
257  uint8_t* colors = colorBuffer->buffer.get();
258  for ( size_t i = 0; i < colorCount; ++i, colors += 3 ) {
259  container.emplace_back( Scalar( colors[0] ) / 255_ra,
260  Scalar( colors[1] ) / 255_ra,
261  Scalar( colors[2] ) / 255_ra,
262  1_ra );
263  }
264  }
265  vertexAttribs.unlock( handle );
266  }
267 
269  for ( const auto& a : allAttributes ) {
270  // For now manage scalar properties only
271  if ( a.second->isList ) {
272  LOG( logWARNING ) << "[TinyPLY] unmanaged vector attribute " << a.first;
273  continue;
274  }
276  auto attribName { a.first };
277  std::replace( attribName.begin(), attribName.end(), '-', '_' );
278  LOG( logINFO ) << "[TinyPLY] Adding custom attrib with name " << attribName << " (was "
279  << a.first << ")";
280  auto handle = vertexAttribs.addAttrib<Scalar>( attribName );
281  auto& container = vertexAttribs.getDataWithLock( handle );
282  copyBufferToContainer( a.second, container );
283  vertexAttribs.unlock( handle );
284  }
285 
286  fileData->m_geometryData.clear();
287  fileData->m_geometryData.reserve( 1 );
288  fileData->m_geometryData.push_back( std::move( geomData ) );
289 
290  fileData->m_loadingTime = ( std::clock() - startTime ) / Scalar( CLOCKS_PER_SEC );
291 
292  if ( fileData->isVerbose() ) {
293  LOG( logINFO ) << "[TinyPLY] File Loading end.";
294  fileData->displayInfo();
295  }
296 
297  fileData->m_processed = true;
298 
299  return fileData;
300 }
