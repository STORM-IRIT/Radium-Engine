44 inline RA_CORE_API PointToTriangleOutput pointToTriSq( const Vector3& q,
45  const Vector3& a,
46  const Vector3& b,
47  const Vector3& c ) {
48  /*
49  * This function projects the query point Q on the triangle plane to solve the
50  * planar problem described by the following schema of Voronoi zones :
51  * 3 /
52  * -- C
53  * |\\ 6
54  * | \\
55  * 5 | \\ /
56  * | \\ /
57  * -- A ------ B 2
58  * | |
59  * 1 | 4 |
60  *
61  * It first checks if the nearest point is a vertex (zones 1,2,3), then if
62  * it belongs to an edge (zone, 4,5,6). If not the nearest point lies on the
63  * triangle.
64  * Reference : Fast Distance Queries for Triangles, Lines and Points using SSE instructions
65  * Evan Shellshear, Robin Ytterlid.
66  * http://jcgt.org/published/0003/04/05/paper.pdf
67  * Journal of Computer Graphics Techniques, Vol.3 No. 4 (2014)
68  */
69 
70  PointToTriangleOutput output;
71 
72  const Vector3 ab = b - a;
73  const Vector3 ac = c - a;
74  const Vector3 aq = q - a;
75 
76  CORE_ASSERT( ab.cross( ac ).squaredNorm() > 0, "Triangle ABC is degenerate" );
77 
78  const Scalar d1 = ab.dot( aq );
79  const Scalar d2 = ac.dot( aq );
80 
81  // Closest point is A. (zone 1)
82  const bool m1 = d1 <= 0 && d2 <= 0;
83  if ( m1 ) {
84  output.meshPoint = a;
85  output.distanceSquared = aq.squaredNorm();
86  output.flags = FlagsInternal::HIT_A;
87  return output;
88  }
89 
90  const Vector3 bq = q - b;
91  const Scalar d3 = ab.dot( bq );
92  const Scalar d4 = ac.dot( bq );
93 
94  // Closest point is B (zone 2)
95  const bool m2 = d3 >= 0 && d4 <= d3;
96  if ( m2 ) {
97  output.meshPoint = b;
98  output.distanceSquared = bq.squaredNorm();
99  output.flags = FlagsInternal::HIT_B;
100  return output;
101  }
102 
103  const Vector3 cq = q - c;
104  const Scalar d5 = ab.dot( cq );
105  const Scalar d6 = ac.dot( cq );
106 
107  // Closest point is C (zone 3)
108  const bool m3 = ( d6 >= 0 && d5 <= d6 );
109  if ( m3 ) {
110  output.meshPoint = c;
111  output.distanceSquared = cq.squaredNorm();
112  output.flags = FlagsInternal::HIT_C;
113  return output;
114  }
115 
116  const Scalar vc = d1 * d4 - d3 * d2;
117  const Scalar v1 = d1 / ( d1 - d3 );
118 
119  // Closest point is on AB (zone 4)
120  const bool m4 = vc <= 0 && d1 >= 0 && d3 <= 0;
121  if ( m4 ) {
122  output.meshPoint = a + v1 * ab;
123  output.distanceSquared = ( output.meshPoint - q ).squaredNorm();
124  output.flags = FlagsInternal::HIT_AB;
125  return output;
126  }
127 
128  const Scalar vb = d5 * d2 - d1 * d6;
129  const Scalar w1 = d2 / ( d2 - d6 );
130 
131  // Closest point is on AC (zone 5)
132  const bool m5 = vb <= 0 && d2 >= 0 && d6 <= 0;
133  if ( m5 ) {
134  output.meshPoint = a + w1 * ac;
135  output.distanceSquared = ( output.meshPoint - q ).squaredNorm();
136  output.flags = FlagsInternal::HIT_CA;
137  return output;
138  }
139 
140  const Scalar va = d3 * d6 - d5 * d4;
141  const Scalar w2 = ( d4 - d3 ) / ( ( d4 - d3 ) + ( d5 - d6 ) );
142 
143  // Closest point is on BC (zone 6)
144  const bool m6 = va <= 0 && ( d4 - d3 ) >= 0 && ( d5 - d6 ) >= 0;
145  if ( m6 ) {
146  output.meshPoint = b + w2 * ( c - b );
147  output.distanceSquared = ( output.meshPoint - q ).squaredNorm();
148  output.flags = FlagsInternal::HIT_BC;
149  return output;
150  }
151 
152  // Closest point is on the triangle
153  const Scalar d = Scalar( 1 ) / ( va + vb + vc );
154  const Scalar v2 = vb * d;
155  const Scalar w3 = vc * d;
156 
157  output.meshPoint = a + v2 * ab + w3 * ac;
158  output.distanceSquared = ( output.meshPoint - q ).squaredNorm();
159  output.flags = 0;
160  return output;
161 }

163 inline RA_CORE_API LineToSegmentOutput lineToSegSq( const Vector3& lineOrigin,
164  Vector3 lineDirection,
165  const Vector3& segCenter,
166  Vector3 segDirection,
167  const Scalar segExtent ) {
168  // Reference : Geometric Tools,
169  // https://github.com/pmjoniak/GeometricTools/blob/master/GTEngine/Include/GteDistLineSegment.h
170 
171  LineToSegmentOutput output;
172 
173  segDirection.normalize();
174  lineDirection.normalize();
175 
176  Vector3 diff = lineOrigin - segCenter;
177  Scalar a01 = -lineDirection.dot( segDirection );
178  Scalar b0 = diff.dot( lineDirection );
179  Scalar s0, s1;
180 
181  if ( std::abs( a01 ) < (Scalar)1 ) {
182  // The line and the segment are not parallel
183  Scalar det = (Scalar)1 - a01 * a01;
184  Scalar extDet = segExtent * det;
185  Scalar b1 = -diff.dot( segDirection );
186  s1 = a01 * b0 - b1;
187 
188  if ( s1 >= -extDet ) {
189  if ( s1 <= extDet ) {
190  // 2 interior points are closest, one on the line and one on the segment
191  s0 = ( a01 * b1 - b0 ) / det;
192  s1 /= det;
193  }
194  else {
195  // The endpoint e1 of the segment and an interior point of the line are closest
196  s1 = segExtent;
197  s0 = -( a01 * s1 + b0 );
198  }
199  }
200  else {
201  // The endpoint e0 of the segment and an interior point of the line are closest
202  s1 = -segExtent;
203  s0 = -( a01 * s1 + b0 );
204  }
205  }
206  else {
207  // The line and the segment are parallel
208  // We choose the closest pair so that one point is at the segment origin
209  s1 = (Scalar)0;
210  s0 = -b0;
211  }
212 
213  output.parameter[0] = s0;
214  output.parameter[1] = s1;
215  output.closestPoint[0] = lineOrigin + s0 * lineDirection;
216  output.closestPoint[1] = segCenter + s1 * segDirection;
217  diff = output.closestPoint[0] - output.closestPoint[1];
218  output.sqrDistance = diff.dot( diff );
219  output.distance = std::sqrt( output.sqrDistance );
220  return output;
221 }

223 inline RA_CORE_API LineToTriangleOutput lineToTriSq( const Vector3& lineOrigin,
224  Vector3 lineDirection,
225  const Vector3 v[3] ) {
226  // Reference : GeometricTools,
227  // https://github.com/pmjoniak/GeometricTools/blob/master/GTEngine/Include/GteDistLine3Triangle3.h
228 
229  LineToTriangleOutput output;
230 
231  lineDirection.normalize();
232 
233  Vector3 edge0 = v[1] - v[0];
234  Vector3 edge1 = v[2] - v[0];
235  Vector3 normal = edge0.cross( edge1 );
236  normal.normalize();
237  Scalar nd = normal.dot( lineDirection );
238  if ( std::abs( nd ) > (Scalar)0 ) {
239  // The line intersects the plane of the triangle or the triangle itself
240  // (if the intersection point isn't needed, it possible to use the boolean function
241  // vsTriangle of RayCast.hpp)
242  Vector3 diff = lineOrigin - v[0];
243  Vector3 basis[3]; // {D, U, V}
244  basis[0] = lineDirection;
245  if ( std::abs( basis[0][0] ) > std::abs( basis[0][1] ) ) {
246  basis[1] = { -basis[0][2], (Scalar)0, basis[0][0] };
247  }
248  else {
249  basis[1] = { (Scalar)0, basis[0][2], -basis[0][1] };
250  }
251  basis[2] = basis[0].cross( basis[1] );
252  // Orthonormalize basis
253  // Normalize basis[0]
254  basis[0] /= std::sqrt( basis[0].dot( basis[0] ) );
255  basis[1] -= basis[0] * ( basis[1].dot( basis[0] ) );
256  // Normalize basis[1]
257  basis[1] /= std::sqrt( basis[1].dot( basis[1] ) );
258  basis[2] -= basis[0] * ( basis[2].dot( basis[0] ) );
259  basis[2] -= basis[1] * ( basis[2].dot( basis[1] ) );
260  // Normalize basis[2]
261  basis[2] /= std::sqrt( basis[2].dot( basis[2] ) );
262 
263  Scalar UdE0 = basis[1].dot( edge0 );
264  Scalar UdE1 = basis[1].dot( edge1 );
265  Scalar UdDiff = basis[1].dot( diff );
266  Scalar VdE0 = basis[2].dot( edge0 );
267  Scalar VdE1 = basis[2].dot( edge1 );
268  Scalar VdDiff = basis[2].dot( diff );
269  Scalar invDet = (Scalar)1 / ( UdE0 * VdE1 - UdE1 * VdE0 );
270 
271  // Barycentric coordinates for the point of intersection
272  Scalar b1 = ( VdE1 * UdDiff - UdE1 * VdDiff ) * invDet;
273  Scalar b2 = ( UdE0 * VdDiff - VdE0 * UdDiff ) * invDet;
274  Scalar b0 = (Scalar)1 - b1 - b2;
275 
276  if ( b0 >= (Scalar)0 && b1 >= (Scalar)0 && b2 >= (Scalar)0 ) {
277  // The line intersects the triangle itself
278  // Distance between the origin of the line and the intersection point with the triangle
279  Scalar DdE0 = lineDirection.dot( edge0 );
280  Scalar DdE1 = lineDirection.dot( edge1 );
281  Scalar DdDiff = lineDirection.dot( diff );
282  output.lineParameter = b1 * DdE0 + b2 * DdE1 - DdDiff;
283 
284  // Barycentric coordinates for the point of intersection
285  output.triangleParameter[0] = b0;
286  output.triangleParameter[1] = b1;
287  output.triangleParameter[2] = b2;
288 
289  // The intersection point is inside or on the triangle
290  output.closestPoint[0] = lineOrigin + output.lineParameter * lineDirection;
291  output.closestPoint[1] = v[0] + b1 * edge0 + b2 * edge1;
292 
293  output.distance = (Scalar)0;
294  output.sqrDistance = (Scalar)0;
295  return output;
296  }
297  }
298 
299  // Either the line intersects the plane of the triangle but not the triangle itself, or the line
300  // is parallel to the triangle. Regardless, the closest point on the triangle is on an edge of
301  // the triangle, so we compare the line to all 3 edges of the triangle.
302  output.distance = std::numeric_limits<Scalar>::max();
303  output.sqrDistance = std::numeric_limits<Scalar>::max();
304  for ( int i0 = 2, i1 = 0; i1 < 3; i0 = i1++ ) {
305  Vector3 segCenter = ( (Scalar)0.5 ) * ( v[i0] + v[i1] );
306  Vector3 segDirection = v[i1] - v[i0];
307  Scalar segExtent = ( (Scalar)0.5 ) * std::sqrt( segDirection.dot( segDirection ) );
308 
309  // Distance line-segment is computed
310  LineToSegmentOutput lsOutput =
311  lineToSegSq( lineOrigin, lineDirection, segCenter, segDirection, segExtent );
312  if ( lsOutput.sqrDistance < output.sqrDistance ) {
313  output.sqrDistance = lsOutput.sqrDistance;
314  output.distance = lsOutput.distance;
315  output.lineParameter = lsOutput.parameter[0];
316  output.triangleParameter[i0] =
317  ( (Scalar)0.5 ) * ( (Scalar)1 - lsOutput.parameter[0] / segExtent );
318  output.triangleParameter[i1] = (Scalar)1 - output.triangleParameter[i0];
319  output.triangleParameter[3 - i0 - i1] = (Scalar)0;
320  output.closestPoint[0] = lsOutput.closestPoint[0];
321  output.closestPoint[1] = lsOutput.closestPoint[1];
322  }
323  }
324  return output;

328  Vector3 segDirection,
329  Scalar segExtent,
330  const Vector3 v[3] ) {
331  // Reference : Geometric Tools,
332  // https://github.com/pmjoniak/GeometricTools/blob/master/GTEngine/Include/GteDistSegment3Triangle3.h
333 
334  SegmentToTriangleOutput output;
335 
336  segDirection.normalize();
337 
338  // Distance line-triangle is computed
339  LineToTriangleOutput ltOutput = lineToTriSq( segCenter, segDirection, v );
340 
341  if ( ltOutput.lineParameter >= -segExtent ) {
342  // The closest point on the line is on the segment or at its right
343  if ( ltOutput.lineParameter <= segExtent ) {
344  // The closest point on the line is on the segment
345  // The segment intersects the triangle
346  output.distance = ltOutput.distance;
347  output.sqrDistance = ltOutput.sqrDistance;
348  output.segmentParameter = ltOutput.lineParameter;
349  output.triangleParameter[0] = ltOutput.triangleParameter[0];
350  output.triangleParameter[1] = ltOutput.triangleParameter[1];
351  output.triangleParameter[2] = ltOutput.triangleParameter[2];
352  output.closestPoint[0] = ltOutput.closestPoint[0];
353  output.closestPoint[1] = ltOutput.closestPoint[1];
354  }
355  else {
356  // The closest point on the line is at the segment's right
357  // We compute the distance between the right endpoint of the segment and the triangle
358  Vector3 point = segCenter + segExtent * segDirection;
359  PointToTriangleOutput ptOutput = pointToTriSq( point, v[0], v[1], v[2] );
360  output.sqrDistance = ptOutput.distanceSquared;
361  output.distance = std::sqrt( output.sqrDistance );
362  output.segmentParameter = segExtent;
363  output.closestPoint[0] = point;
364  output.closestPoint[1] = ptOutput.meshPoint;
365  }
366  }
367  else {
368  // The closest point on the line is at the segment's left
369  // We compute the distance between the left endpoint of the segment and the triangle
370  Vector3 point = segCenter - segExtent * segDirection;
371  PointToTriangleOutput ptOutput = pointToTriSq( point, v[0], v[1], v[2] );
372  output.sqrDistance = ptOutput.distanceSquared;
373  output.distance = std::sqrt( output.sqrDistance );
374  output.segmentParameter = segExtent;
375  output.closestPoint[0] = point;
376  output.closestPoint[1] = ptOutput.meshPoint;
377  }
