293 void TrackballCameraManipulator::fitScene( const Core::Aabb& aabb ) {
294 
295  Scalar f = m_camera->getFOV();
296  Scalar a = m_camera->getAspect();
297 
298  const Scalar r = ( aabb.max() - aabb.min() ).norm() / 2_ra;
299  const Scalar x = r / std::sin( f / 2_ra );
300  const Scalar y = r / std::sin( f * a / 2_ra );
301  Scalar d = std::max( std::max( x, y ), 0.001_ra );
302 
303  m_camera->setFrame( Core::Transform::Identity() );
304  Core::Vector3 camPos { aabb.center().x(), aabb.center().y(), aabb.center().z() + d };
305  m_camera->setPosition( camPos );
306  Core::Vector3 camDir { aabb.center() - camPos };
307  m_distFromCenter = camDir.norm();
308  m_camera->setDirection( camDir / m_distFromCenter );
309 
310  // no ref camera here, use wolrd frame to align with
311  m_referenceFrame.setIdentity();
312  m_referenceFrame.translation() = aabb.center();
313 
314  updatePhiTheta();
315 
316  if ( m_light != nullptr ) {
317  m_light->setPosition( m_camera->getPosition() );
318  m_light->setDirection( m_camera->getDirection() );
319  }
320 }

322 void TrackballCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
323  Scalar dphi = m_phiDir * dx * m_cameraSensitivity * m_quickCameraModifier;
324  Scalar dtheta = -dy * m_cameraSensitivity * m_quickCameraModifier;
325 
326  Scalar phi = m_phi + dphi;
327  Scalar theta = m_theta + dtheta;
328  Core::Vector3 dir { std::sin( phi ) * std::sin( theta ),
329  std::cos( theta ),
330  std::cos( phi ) * std::sin( theta ) };
331 
332  Core::Vector3 right { -dir[2], 0, dir[0] };
333  right.normalize();
334  if ( ( m_referenceFrame.linear().inverse() * m_camera->getRightVector() ).dot( right ) < 0 )
335  right = -right;
336 
337  Core::Vector3 up = dir.cross( right ).normalized();
338 
339  dir = m_referenceFrame.linear() * dir;
340  right = m_referenceFrame.linear() * right;
341  up = m_referenceFrame.linear() * up;
342 
343  Core::Matrix3 m;
344  // clang-format off
345  m << right[0], up[0], dir[0], //
346  right[1], up[1], dir[1], //
347  right[2], up[2], dir[2]; //
348  // clang-format on
349  Core::Transform t;
350 
351  t.setIdentity();
352  t.linear() = m;
353  Core::Vector3 pos = m_referenceFrame.translation() + m_distFromCenter * dir;
354  t.translation() = pos;
355  m_camera->setFrame( t );
356 
357  m_phi = phi;
358  m_theta = theta;
359 
360  clampThetaPhi();
361 }

363 void TrackballCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {
364  Scalar x = dx * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
365  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
366  // Move camera and trackball center, keep the distance to the center
367  Core::Vector3 R = -m_camera->getRightVector();
368  Core::Vector3 U = m_camera->getUpVector();
369 
370  Core::Transform T( Core::Transform::Identity() );
371  Core::Vector3 t = x * R + y * U;
372  T.translate( t );
373 
374  m_camera->applyTransform( T );
375  m_referenceFrame.translation() += t;
376 }

403 void TrackballCameraManipulator::updatePhiTheta() {
404  using Core::Math::areApproxEqual;
405  const Core::Vector3 R = m_referenceFrame.linear().inverse() * ( -m_camera->getDirection() );
406 
407  m_theta = std::acos( R.y() );
408 
409  // unlikely to have z and x to 0, unless direction is perfectly aligned with
410  // m_referenceFrame.z() in this case phi is given by the relative orientation of right/up in
411  // the z/x plane of m_reference frame.
412  if ( UNLIKELY( areApproxEqual( R.z(), 0_ra ) && areApproxEqual( R.x(), 0_ra ) ) ) {
413  Scalar fx = m_referenceFrame.matrix().block<3, 1>( 0, 2 ).dot( m_camera->getRightVector() );
414  Scalar fy = m_referenceFrame.matrix().block<3, 1>( 0, 2 ).dot( m_camera->getUpVector() );
415  m_phi = std::atan2( fx, fy );
416  }
417  else {
418  m_phi = std::atan2( R.x(), R.z() );
419  }
420 
421  // no need to clamp, atan2 is by def \in [-pi,pi]
422  // acos in [0, pi]
423  // clampThetaPhi();
424  CORE_ASSERT( std::isfinite( m_theta ) && std::isfinite( m_phi ), "Error in trackball camera" );
425 }

427 void TrackballCameraManipulator::clampThetaPhi() {
428  // Keep phi between 0 and 2pi
429  if ( m_phi < 0_ra ) { m_phi += 2_ra * Pi; }
430  // Keep theta in [-pi, pi] (instead of [0,pi]) to allows scene flip
431  if ( m_theta < -Pi ) { m_theta += 2_ra * Pi; }
432  if ( m_theta > Pi ) { m_theta -= 2_ra * Pi; }
433 }

435 bool TrackballCameraManipulator::checkIntegrity( const std::string& mess ) const {
436  Core::Vector3 c = m_camera->getPosition() + m_distFromCenter * m_camera->getDirection();
437  Scalar d = ( m_referenceFrame.translation() - c ).norm();
438  if ( d > 0.001_ra ) {
439  LOG( logWARNING ) << "TrackballCameraManipulator Integrity problem : " << mess;
440  LOG( logWARNING ) << "\t Position " << m_camera->getPosition().transpose();
441  LOG( logWARNING ) << "\t Ref "
442  << ( m_referenceFrame.translation() +
443  m_distFromCenter * ( -m_camera->getDirection() ) )
444  .transpose();
445  LOG( logWARNING ) << "\t Direction " << m_camera->getDirection().transpose();
446  LOG( logWARNING ) << "\t Center " << c.transpose();
447  LOG( logWARNING ) << "\t Distance " << d;
448  LOG( logWARNING ) << "\t angles " << m_phi << " " << m_theta;
449  }
450  return d < 0.001_ra;
451 }
