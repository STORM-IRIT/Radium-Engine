270 void TrackballCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
271  Scalar dphi = m_phiDir * dx * m_cameraSensitivity * m_quickCameraModifier;
272  Scalar dtheta = -dy * m_cameraSensitivity * m_quickCameraModifier;
273 
274  Scalar phi = m_phi + dphi;
275  Scalar theta = m_theta + dtheta;
276  Core::Vector3 dir { std::sin( phi ) * std::sin( theta ),
277  std::cos( theta ),
278  std::cos( phi ) * std::sin( theta ) };
279 
280  Core::Vector3 right { -dir[2], 0, dir[0] };
281  right.normalize();
282  if ( ( m_referenceFrame.linear().inverse() * m_camera->getRightVector() ).dot( right ) < 0 )
283  right = -right;
284 
285  Core::Vector3 up = dir.cross( right ).normalized();
286 
287  dir = m_referenceFrame.linear() * dir;
288  right = m_referenceFrame.linear() * right;
289  up = m_referenceFrame.linear() * up;
290 
291  Core::Matrix3 m;
292  // clang-format off
293  m << right[0], up[0], dir[0], //
294  right[1], up[1], dir[1], //
295  right[2], up[2], dir[2]; //
296  // clang-format on
297  Core::Transform t;
298 
299  t.setIdentity();
300  t.linear() = m;
301  Core::Vector3 pos = m_referenceFrame.translation() + m_distFromCenter * dir;
302  t.translation() = pos;
303  m_camera->setFrame( t );
304 
305  m_phi = phi;
306  m_theta = theta;
307 
308  clampThetaPhi();
309 }

351 void TrackballCameraManipulator::updatePhiTheta() {
352  using Core::Math::areApproxEqual;
353  const Core::Vector3 R = m_referenceFrame.linear().inverse() * ( -m_camera->getDirection() );
354 
355  m_theta = std::acos( R.y() );
356 
357  // unlikely to have z and x to 0, unless direction is perfectly aligned with
358  // m_referenceFrame.z() in this case phi is given by the relative orientation of right/up in the
359  // z/x plane of m_reference frame.
360  if ( UNLIKELY( areApproxEqual( R.z(), 0_ra ) && areApproxEqual( R.x(), 0_ra ) ) ) {
361  Scalar fx = m_referenceFrame.matrix().block<3, 1>( 0, 2 ).dot( m_camera->getRightVector() );
362  Scalar fy = m_referenceFrame.matrix().block<3, 1>( 0, 2 ).dot( m_camera->getUpVector() );
363  m_phi = std::atan2( fx, fy );
364  }
365  else {
366  m_phi = std::atan2( R.x(), R.z() );
367  }
368 
369  // no need to clamp, atan2 is by def \in [-pi,pi]
370  // acos in [0, pi]
371  // clampThetaPhi();
372  CORE_ASSERT( std::isfinite( m_theta ) && std::isfinite( m_phi ), "Error in trackball camera" );
373 }

375 void TrackballCameraManipulator::clampThetaPhi() {
376  // Keep phi between 0 and 2pi
377  if ( m_phi < 0_ra ) { m_phi += 2_ra * Pi; }
378  // Keep theta in [-pi, pi] (instead of [0,pi]) to allows scene flip
379  if ( m_theta < -Pi ) { m_theta += 2_ra * Pi; }
380  if ( m_theta > Pi ) { m_theta -= 2_ra * Pi; }
381 }
