157 void ControlPanel::addColorInput(
158  const std::string& name,
159  const std::function<void( const Ra::Core::Utils::Color& clr )>& callback,
160  Ra::Core::Utils::Color color,
161  bool withAlpha,
162  const std::string& tooltip ) {
163  auto button = new QPushButton( name.c_str(), this );
164  auto srgbColor = Ra::Core::Utils::Color::linearRGBTosRGB( color );
165  auto clrBttn = QColor::fromRgbF( srgbColor[0], srgbColor[1], srgbColor[2], srgbColor[3] );
166  auto clrDlg = [callback, clrBttn, withAlpha, button, name]() mutable {
167  clrBttn = QColorDialog::getColor(
168  clrBttn,
169  nullptr,
170  name.c_str(),
171  ( withAlpha ? QColorDialog::ShowAlphaChannel : QColorDialog::ColorDialogOptions() )
172 #ifndef OS_MACOS
173  | QColorDialog::DontUseNativeDialog
174 #endif
175  );
176  if ( clrBttn.isValid() ) {
177  // update the background color of the viewer
178  auto lum = 0.2126_ra * Scalar( clrBttn.redF() ) +
179  0.7151_ra * Scalar( clrBttn.greenF() ) +
180  0.0721_ra * Scalar( clrBttn.blueF() );
181  auto bgk = Ra::Core::Utils::Color::sRGBToLinearRGB(
182  Ra::Core::Utils::Color( Scalar( clrBttn.redF() ),
183  Scalar( clrBttn.greenF() ),
184  Scalar( clrBttn.blueF() ),
185  Scalar( clrBttn.alphaF() ) ) );
186  callback( bgk );
187  QString qss = QString( "background-color: %1" ).arg( clrBttn.name() );
188  if ( lum > 1_ra / 3_ra ) { qss += QString( "; color: #000000" ); }
189  else {
190  qss += QString( "; color: #FFFFFF" );
191  }
192  button->setStyleSheet( qss );
193  }
194  };
195 
196  if ( !tooltip.empty() ) {
197  button->setToolTip(
198  QString( "<qt>%1</qt>" ).arg( QString( tooltip.c_str() ).toHtmlEscaped() ) );
199  }
200  // for contrast computation : see
201  // https://www.w3.org/TR/UNDERSTANDING-WCAG20/visual-audio-contrast-contrast.html
202  auto lum = 0.2126_ra * srgbColor[0] + 0.7151_ra * srgbColor[1] + 0.0721_ra * srgbColor[2];
203  QString qss = QString( "background-color: %1" ).arg( clrBttn.name() );
204  if ( lum > 1_ra / 3_ra ) { qss += QString( "; color: #000000" ); }
205  else {
206  qss += QString( "; color: #FFFFFF" );
207  }
208  button->setStyleSheet( qss );
209 
210  connect( button, &QPushButton::clicked, clrDlg );
211  m_currentLayout->addWidget( button );
212 }
