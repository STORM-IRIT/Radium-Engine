678 template <typename T>
679 auto VariableSet::addVariableType() -> Utils::optional<VariableContainer<T>*> {
680  auto storage = createVariableStorage<T>();
681 
682  auto tidx = std::type_index( typeid( T ) );
683  auto it = std::find( m_vtable->m_storedType.begin(), m_vtable->m_storedType.end(), tidx );
684 
685  // remember the stored type and its rank for the instance
686  m_typeIndexToVtableIndex[tidx] = it - m_vtable->m_storedType.begin();
687  m_storedType.push_back( tidx );
688 
689  // If the type is not already there, add type operations and identifier in the global vtable
690  if ( it == m_vtable->m_storedType.end() ) {
691  // remember the stored type in the global vtable
692  m_vtable->m_storedType.emplace_back( tidx );
693  // used to merge (keep) the stored data from container "from" to container "to"
694  m_vtable->m_mergeKeepFunctions.emplace_back(
695  []( const VariableSet& from, VariableSet& to ) {
696  auto toStorageAccess = to.existsVariableType<T>();
697  if ( !toStorageAccess ) { toStorageAccess = to.addVariableType<T>(); }
698  auto& toStorage = *( toStorageAccess.value() );
699  auto& fromStorage = from.getVariableStorage<T>();
700  for ( const auto& t : fromStorage ) {
701  toStorage.insert( t );
702  }
703  } );
704  // used to merge (replace) the stored data from container "from" to container "to"
705  m_vtable->m_mergeReplaceFunctions.emplace_back(
706  []( const VariableSet& from, VariableSet& to ) {
707  auto toStorageAccess = to.existsVariableType<T>();
708  if ( !toStorageAccess ) { toStorageAccess = to.addVariableType<T>(); }
709  auto& toStorage = *( toStorageAccess.value() );
710  auto& fromStorage = from.getVariableStorage<T>();
711  for ( const auto& t : fromStorage ) {
712  toStorage.insert_or_assign( t.first, t.second );
713  }
714  } );
715  // use to compute gauge on the stored data
716  m_vtable->m_sizeFunctions.emplace_back(
717  []( const VariableSet& c ) { return c.getVariableStorage<T>().size(); } );
718  // used to visit the variableSet with a dynamic visitor
719  m_vtable->m_visitFunctions.emplace_back(
720  []( const VariableSet& c, const DynamicVisitorBase& v )
721  -> std::pair<bool, std::function<void( DynamicVisitorBase&, std::any && )>> {
722  auto id = getVariableVisitTypeIndex<T>();
723  if ( v.accept( id ) ) {
724  auto& storage = c.getVariableStorage<T>();
725  auto coll = std::ref( storage );
726  return { true, [coll]( DynamicVisitorBase& visitor, std::any&& userParam ) {
727  for ( auto&& t : coll.get() ) {
728  visitor( { std::ref( t ) },
729  std::forward<std::any>( userParam ) );
730  }
731  } };
732  }
733  else {
734  return { false, nullptr };
735  }
736  } );
737  }
738  return storage;
739 }
