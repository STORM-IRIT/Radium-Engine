58void angleWeightedNormal( const VectorArray<Vector3>& p,
59 const AlignedStdVector<Vector3ui>& T,
60 VectorArray<Vector3>& normal ) {
61 const size_t N = p.size();
62 normal.clear();
63 normal.resize( N, Vector3::Zero() );
64 for ( const auto& t : T ) {
65 uint i = t( 0 );
66 uint j = t( 1 );
67 uint k = t( 2 );
68 const Vector3 triN = triangleNormal( p[i], p[j], p[k] );
69 const Scalar theta_i = Math::angle( ( p[j] - p[i] ), ( p[k] - p[i] ) );
70 const Scalar theta_j = Math::angle( ( p[i] - p[j] ), ( p[k] - p[j] ) );
71 const Scalar theta_k = Math::angle( ( p[i] - p[k] ), ( p[j] - p[k] ) );
72 normal[i] += theta_i * triN;
73 normal[j] += theta_j * triN;
74 normal[k] += theta_k * triN;
75 }
76 for ( auto& n : normal ) {
77 n.normalize();
78 }

106Vector3 uniformNormal( const Vector3& v, const VectorArray<Vector3>& one_ring ) {
107 Vector3 normal;
108 normal.setZero();
109 size_t N = one_ring.size();
110 Utils::CircularIndex i;
111 i.setSize( N );
112 for ( uint j = 0; j < N; ++j ) {
113 i.setValue( j );
114 normal += triangleNormal( v, one_ring[i], one_ring[i - 1] );
115 }
116 if ( !normal.isApprox( Vector3::Zero() ) ) { return normal.normalized(); }
117 return Vector3::Zero();
118}
