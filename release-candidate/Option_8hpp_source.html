761  std::string get_name( bool positional = false,
762  bool all_options = false
763  ) const {
764  if ( get_group().empty() ) return {}; // Hidden
765 
766  if ( all_options ) {
767 
768  std::vector<std::string> name_list;
769 
771  if ( ( positional && ( !pname_.empty() ) ) || ( snames_.empty() && lnames_.empty() ) ) {
772  name_list.push_back( pname_ );
773  }
774  if ( ( get_items_expected() == 0 ) && ( !fnames_.empty() ) ) {
775  for ( const std::string& sname : snames_ ) {
776  name_list.push_back( "-" + sname );
777  if ( check_fname( sname ) ) {
778  name_list.back() += "{" + get_flag_value( sname, "" ) + "}";
779  }
780  }
781 
782  for ( const std::string& lname : lnames_ ) {
783  name_list.push_back( "--" + lname );
784  if ( check_fname( lname ) ) {
785  name_list.back() += "{" + get_flag_value( lname, "" ) + "}";
786  }
787  }
788  }
789  else {
790  for ( const std::string& sname : snames_ )
791  name_list.push_back( "-" + sname );
792 
793  for ( const std::string& lname : lnames_ )
794  name_list.push_back( "--" + lname );
795  }
796 
797  return detail::join( name_list );
798  }
799 
800  // This returns the positional name no matter what
801  if ( positional ) return pname_;
802 
803  // Prefer long name
804  if ( !lnames_.empty() ) return std::string( 2, '-' ) + lnames_[0];
805 
806  // Or short name if no long name
807  if ( !snames_.empty() ) return std::string( 1, '-' ) + snames_[0];
808 
809  // If positional is the only name, it's okay to use that
810  return pname_;
811  }

818  void run_callback() {
819 
820  if ( current_option_state_ == option_state::parsing ) {
821  _validate_results( results_ );
822  current_option_state_ = option_state::validated;
823  }
824 
825  if ( current_option_state_ < option_state::reduced ) {
826  _reduce_results( proc_results_, results_ );
827  current_option_state_ = option_state::reduced;
828  }
829  if ( current_option_state_ >= option_state::reduced ) {
830  current_option_state_ = option_state::callback_run;
831  if ( !( callback_ ) ) { return; }
832  const results_t& send_results = proc_results_.empty() ? results_ : proc_results_;
833  bool local_result = callback_( send_results );
834 
835  if ( !local_result ) throw ConversionError( get_name(), results_ );
836  }
837  }

840  const std::string& matching_name( const Option& other ) const {
841  static const std::string estring;
842  for ( const std::string& sname : snames_ )
843  if ( other.check_sname( sname ) ) return sname;
844  for ( const std::string& lname : lnames_ )
845  if ( other.check_lname( lname ) ) return lname;
846 
847  if ( ignore_case_ || ignore_underscore_ ) { // We need to do the inverse, in case we are
848  // ignore_case or ignore underscore
849  for ( const std::string& sname : other.snames_ )
850  if ( check_sname( sname ) ) return sname;
851  for ( const std::string& lname : other.lnames_ )
852  if ( check_lname( lname ) ) return lname;
853  }
854  return estring;
855  }
857  bool operator==( const Option& other ) const { return !matching_name( other ).empty(); }

860  bool check_name( const std::string& name ) const {
861 
862  if ( name.length() > 2 && name[0] == '-' && name[1] == '-' )
863  return check_lname( name.substr( 2 ) );
864  if ( name.length() > 1 && name.front() == '-' ) return check_sname( name.substr( 1 ) );
865  if ( !pname_.empty() ) {
866  std::string local_pname = pname_;
867  std::string local_name = name;
868  if ( ignore_underscore_ ) {
869  local_pname = detail::remove_underscore( local_pname );
870  local_name = detail::remove_underscore( local_name );
871  }
872  if ( ignore_case_ ) {
873  local_pname = detail::to_lower( local_pname );
874  local_name = detail::to_lower( local_name );
875  }
876  if ( local_name == local_pname ) { return true; }
877  }
878 
879  if ( !envname_.empty() ) {
880  // this needs to be the original since envname_ shouldn't match on case insensitivity
881  return ( name == envname_ );
882  }
883  return false;
884  }

1168  void _validate_results( results_t& res ) const {
1169  // Run the Validators (can change the string)
1170  if ( !validators_.empty() ) {
1171  if ( type_size_max_ > 1 ) { // in this context index refers to the index in the type
1172  int index = 0;
1173  if ( get_items_expected_max() < static_cast<int>( res.size() ) &&
1174  multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast ) {
1175  // create a negative index for the earliest ones
1176  index = get_items_expected_max() - static_cast<int>( res.size() );
1177  }
1178 
1179  for ( std::string& result : res ) {
1180  if ( detail::is_separator( result ) && type_size_max_ != type_size_min_ &&
1181  index >= 0 ) {
1182  index = 0; // reset index for variable size chunks
1183  continue;
1184  }
1185  auto err_msg =
1186  _validate( result, ( index >= 0 ) ? ( index % type_size_max_ ) : index );
1187  if ( !err_msg.empty() ) throw ValidationError( get_name(), err_msg );
1188  ++index;
1189  }
1190  }
1191  else {
1192  int index = 0;
1193  if ( expected_max_ < static_cast<int>( res.size() ) &&
1194  multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast ) {
1195  // create a negative index for the earliest ones
1196  index = expected_max_ - static_cast<int>( res.size() );
1197  }
1198  for ( std::string& result : res ) {
1199  auto err_msg = _validate( result, index );
1200  ++index;
1201  if ( !err_msg.empty() ) throw ValidationError( get_name(), err_msg );
1202  }
1203  }

1210  void _reduce_results( results_t& res, const results_t& original ) const {
1211 
1212  // max num items expected or length of vector, always at least 1
1213  // Only valid for a trimming policy
1214 
1215  res.clear();
1216  // Operation depends on the policy setting
1217  switch ( multi_option_policy_ ) {
1218  case MultiOptionPolicy::TakeAll:
1219  break;
1220  case MultiOptionPolicy::TakeLast: {
1221  // Allow multi-option sizes (including 0)
1222  std::size_t trim_size = std::min<std::size_t>(
1223  static_cast<std::size_t>( std::max<int>( get_items_expected_max(), 1 ) ),
1224  original.size() );
1225  if ( original.size() != trim_size ) {
1226  res.assign( original.end() - static_cast<results_t::difference_type>( trim_size ),
1227  original.end() );
1228  }
1229  } break;
1230  case MultiOptionPolicy::TakeFirst: {
1231  std::size_t trim_size = std::min<std::size_t>(
1232  static_cast<std::size_t>( std::max<int>( get_items_expected_max(), 1 ) ),
1233  original.size() );
1234  if ( original.size() != trim_size ) {
1235  res.assign( original.begin(),
1236  original.begin() +
1237  static_cast<results_t::difference_type>( trim_size ) );
1238  }
1239  } break;
1240  case MultiOptionPolicy::Join:
1241  if ( results_.size() > 1 ) {
1242  res.push_back( detail::join(
1243  original, std::string( 1, ( delimiter_ == '\0' ) ? '\n' : delimiter_ ) ) );
1244  }
1245  break;
1246  case MultiOptionPolicy::Throw:
1247  default: {
1248  auto num_min = static_cast<std::size_t>( get_items_expected_min() );
1249  auto num_max = static_cast<std::size_t>( get_items_expected_max() );
1250  if ( num_min == 0 ) { num_min = 1; }
1251  if ( num_max == 0 ) { num_max = 1; }
1252  if ( original.size() < num_min ) {
1253  throw ArgumentMismatch::AtLeast(
1254  get_name(), static_cast<int>( num_min ), original.size() );
1255  }
1256  if ( original.size() > num_max ) {
1257  throw ArgumentMismatch::AtMost(
1258  get_name(), static_cast<int>( num_max ), original.size() );
1259  }
1260  break;
1261  }
1262  }
1263  }
