36 void AdjacencyList::pruneLeaves( std::vector<uint>& pruned, std::vector<bool>& delete_flag ) {
37  pruned.clear();
38  delete_flag.clear();
39  delete_flag.resize( this->size(), false );
40  std::vector<bool> prune_flag = delete_flag;
41  for ( uint i = 0; i < this->size(); ++i ) {
42  if ( this->isLeaf( i ) && !this->isRoot( i ) ) {
43  delete_flag[i] = true;
44  prune_flag[i] = true;
45  }
46  else {
47  pruned.push_back( i );
48  }
49  }
50 
51  for ( uint j = this->size(); j > 0; --j ) {
52  const uint i = j - 1;
53  if ( prune_flag[i] ) {
54  this->m_parent.erase( this->m_parent.begin() + i );
55  this->m_child.erase( this->m_child.begin() + i );
56  prune_flag.erase( prune_flag.begin() + i );
57  ++j;
58  }
59  }
60 
61  for ( uint i = 0; i < this->size(); ++i ) {
62  this->m_parent[i] =
63  ( ( this->m_parent[i] == -1 ) || ( delete_flag[i] ) ) ? -1 : pruned[this->m_parent[i]];
64  for ( auto it = this->m_child[i].begin(); it != this->m_child[i].end(); ++it ) {
65  if ( delete_flag[( *it )] ) {
66  this->m_child[i].erase( it );
67  --it;
68  }
69  else {
70  *it = pruned[*it];
71  }
72  }

105 AdjacencyList::ConsistencyStatus AdjacencyList::computeConsistencyStatus() const {
106  // Note: we abort at the first error found to keep the error backtrace
107  if ( m_parent.size() != m_child.size() ) {
108  return ConsistencyStatus::IncompatibleChildrenAndParentList;
109  }
110 
111  for ( uint node = 0; node < size(); ++node ) {
112  if ( m_parent.at( node ) >= int( node ) ) { return ConsistencyStatus::WrongParentOrdering; }
113  if ( m_parent.at( node ) < -1 ) { return ConsistencyStatus::WrongParentIndex; }
114 
115  for ( const auto& child : m_child.at( node ) ) {
116  if ( m_parent.at( child ) != int( node ) ) {
117  return ConsistencyStatus::InconsistentParentIndex;
118  }
119  }
120 
121  if ( isLeaf( node ) != ( m_child.at( node ).size() == 0 ) ) {
122  return ConsistencyStatus::NonLeafNodeWithoutChild;
123  }
124  }
125  return ConsistencyStatus::Valid;
