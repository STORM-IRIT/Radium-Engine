321 void CoreGeometryDisplayable<CoreGeometry>::updateGL() {
322  if ( m_isDirty ) {
323  // Check that our dirty bits are consistent.
324  ON_ASSERT( bool dirtyTest = false; for ( auto d
325  : m_dataDirty ) { dirtyTest = dirtyTest || d; } );
326  CORE_ASSERT( dirtyTest == m_isDirty, "Dirty flags inconsistency" );
327  CORE_ASSERT( !( m_mesh.vertices().empty() ), "No vertex." );
328 
329  updateGL_specific_impl();
330 #ifdef CORE_USE_DOUBLE
331  // need convserion
332  auto func = [this]( Ra::Core::Utils::AttribBase* b ) {
333  auto idx = m_handleToBuffer[b->getName()];
334 
335  if ( m_dataDirty[idx] ) {
336  if ( !m_vbos[idx] ) { m_vbos[idx] = globjects::Buffer::create(); }
337 
338  auto stride = b->getStride();
339  auto eltSize = b->getNumberOfComponents();
340  auto size = b->getSize();
341  auto data = std::make_unique<float[]>( size * eltSize );
342  const void* ptr = b->dataPtr();
343  const char* cptr = reinterpret_cast<const char*>( ptr );
344 
345  for ( size_t i = 0; i < size; i++ ) {
346  auto tptr = reinterpret_cast<const Scalar*>( cptr + i * stride );
347  for ( size_t j = 0; j < eltSize; ++j ) {
348  data[i * eltSize + j] = tptr[j];
349  }
350  }
351 
352  m_vbos[idx]->setData(
353  size * eltSize * sizeof( float ), data.get(), GL_DYNAMIC_DRAW );
354 
355  m_dataDirty[idx] = false;
356  }
357  };
358 
359 #else
360  auto func = [this]( Ra::Core::Utils::AttribBase* b ) {
361  auto idx = m_handleToBuffer[b->getName()];
362 
363  if ( m_dataDirty[idx] ) {
364  if ( !m_vbos[idx] ) { m_vbos[idx] = globjects::Buffer::create(); }
365  m_vbos[idx]->setData( b->getBufferSize(), b->dataPtr(), GL_DYNAMIC_DRAW );
366  m_dataDirty[idx] = false;
367  }
368  };
369 #endif
370  m_mesh.vertexAttribs().for_each_attrib( func );
371 
372  // cleanup removed attrib
373  for ( auto buffer : m_handleToBuffer ) {
374  // do not remove name from handleToBuffer to keep index ...
375  // we could also update handleToBuffer, m_vbos, m_dataDirty
376  if ( !m_mesh.hasAttrib( buffer.first ) && m_vbos[buffer.second] ) {
377  m_vbos[buffer.second].reset( nullptr );
378  m_dataDirty[buffer.second] = false;
379  }
380  }

382  GL_CHECK_ERROR;
383  m_isDirty = false;

581 template <typename T>
582 void GeneralMesh<T>::triangulate() {
583  m_triangleIndices.clear();
584  m_triangleIndices.reserve( this->m_mesh.getIndices().size() );
585  for ( const auto& face : this->m_mesh.getIndices() ) {
586  if ( face.size() == 3 ) { m_triangleIndices.push_back( face ); }
587  else {
589  int minus { int( face.size() ) - 1 };
590  int plus { 0 };
591  while ( plus + 1 < minus ) {
592  if ( ( plus - minus ) % 2 ) {
593  m_triangleIndices.emplace_back( face[plus], face[plus + 1], face[minus] );
594  ++plus;
595  }
596  else {
597  m_triangleIndices.emplace_back( face[minus], face[plus], face[minus - 1] );
598  --minus;
599  }
600  }
601  }
602  }
