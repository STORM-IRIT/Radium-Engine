172 void KeyMappingManager::setActionBinding( const Context& contextIndex,
173  const EventBinding& binding,
174  const KeyMappingAction& actionIndex ) {
175 
176  CORE_ASSERT( size_t( contextIndex ) < m_contextNameToIndex.size(),
177  "contextIndex is out of range" );
178 
179  // search if an action already correspond to this binding.
180  auto f = m_bindingToAction[contextIndex].find( binding );
181  if ( f != m_bindingToAction[contextIndex].end() ) {
182 
183  // if yes, search for its name
184  auto findResult = std::find_if(
185  std::begin( m_actionNameToIndex[contextIndex] ),
186  std::end( m_actionNameToIndex[contextIndex] ),
187  [&]( const ActionNameMap::value_type& pair ) { return pair.second == actionIndex; } );
188 
189  // if the name is not present, something bad happens.
190  if ( findResult == std::end( m_actionNameToIndex[contextIndex] ) ) {
191  LOG( logERROR ) << "Corrupted call to bindKeyToAction, index " << actionIndex
192  << " must have been inserted before !\n";
193  return;
194  }
195 
196  // if name is present, find the other action's name
197  auto findResult2 = std::find_if(
198  std::begin( m_actionNameToIndex[contextIndex] ),
199  std::end( m_actionNameToIndex[contextIndex] ),
200  [&]( const ActionNameMap::value_type& pair ) { return pair.second == f->second; } );
201 
202  // if the name is not present, something bad happens.
203  if ( findResult2 == std::end( m_actionNameToIndex[contextIndex] ) ) {
204  LOG( logERROR ) << "Corrupted call to bindKeyToAction, index " << actionIndex
205  << " must have been inserted before !\n";
206  return;
207  }
208 
209  LOG( logWARNING ) << "Binding action " << findResult->first << " to "
210  << "buttons [" << enumNamesFromMouseButtons( binding.m_buttons ) << "] "
211  << "modifiers [" << enumNamesFromKeyboardModifiers( binding.m_modifiers )
212  << "]"
213  << " keycode [" << binding.m_key << "]"
214  << " wheel [" << binding.m_wheel << "]"
215  << ", which is already used for action " << findResult2->first << ".";
216  }
217 
218  LOG( logDEBUG2 ) << "In context " << getContextName( contextIndex ) << " [" << contextIndex
219  << "]"
220  << " binding action " << getActionName( contextIndex, actionIndex ) << " ["
221  << actionIndex << "]"
222  << " buttons [" << enumNamesFromMouseButtons( binding.m_buttons ) << "]"
223  << " modifiers [" << enumNamesFromKeyboardModifiers( binding.m_modifiers )
224  << "]"
225  << " keycode [" << binding.m_key << "]"
226  << " wheel [" << binding.m_wheel << "]";
227 
228  m_bindingToAction[contextIndex][binding] = actionIndex;
229 }

300 void KeyMappingManager::saveKeymap( QXmlStreamWriter& stream ) {
301 
302  // helper functor to write attrib
303  auto saveAttrib = [&stream]( const QString& attribName,
304  const QString& attribValue,
305  const QString& attribDefault ) {
306  if ( attribValue != attribDefault ) stream.writeAttribute( attribName, attribValue );
307  };
308 
309  if ( stream.hasError() ) { return; }
310 
311  stream.writeStartElement( "keymaps" );
312 
313  for ( const auto& contextPair : m_contextNameToIndex ) {
314  const auto& contextName = contextPair.first;
315  const auto& context = contextPair.second;
316  for ( const auto& actionPair : m_bindingToAction[context] ) {
317  const auto& binding = actionPair.first;
318  const auto& action = actionPair.second;
319  const auto& actionNames = m_actionNameToIndex[context];
320  auto actionItr =
321  std::find_if( actionNames.begin(), actionNames.end(), [&action]( const auto& a ) {
322  return a.second == action;
323  } );
324  if ( actionItr != actionNames.end() ) {
325  const auto& actionName = actionItr->first;
326  stream.writeStartElement( "keymap" );
327  stream.writeAttribute( "context", QString::fromStdString( contextName ) );
328  stream.writeAttribute( "action", QString::fromStdString( actionName ) );
329 
330  saveAttrib(
331  "buttons",
332  QString::fromStdString( enumNamesFromMouseButtons( binding.m_buttons ) ),
333  QString::fromStdString( enumNamesFromMouseButtons( Qt::NoButton ) ) );
334  saveAttrib(
335  "modifiers",
336  QString::fromStdString( enumNamesFromKeyboardModifiers( binding.m_modifiers ) ),
337  QString::fromStdString( enumNamesFromKeyboardModifiers( Qt::NoModifier ) ) );
338  saveAttrib( "key", "Key_" + QKeySequence( binding.m_key ).toString(), "Key_" );
339  saveAttrib( "wheel", binding.m_wheel ? "true" : "false", "false" );
340 
341  stream.writeEndElement();
342  }
343  }
344  }
