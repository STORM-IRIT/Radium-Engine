26  protected: \
27  name( std::string ename, std::string msg, int exit_code ) : \
28  parent( std::move( ename ), std::move( msg ), exit_code ) {} \
29  name( std::string ename, std::string msg, ExitCodes exit_code ) : \

32  public: \

34  name( std::string msg, int exit_code ) : parent( #name, std::move( msg ), exit_code ) {}

253 class RequiredError : public ParseError
254 {
255  CLI11_ERROR_DEF( ParseError, RequiredError )
256  explicit RequiredError( std::string name ) :
257  RequiredError( name + " is required", ExitCodes::RequiredError ) {}
258  static RequiredError Subcommand( std::size_t min_subcom ) {
259  if ( min_subcom == 1 ) { return RequiredError( "A subcommand" ); }
260  return RequiredError( "Requires at least " + std::to_string( min_subcom ) + " subcommands",
261  ExitCodes::RequiredError );
262  }
263  static RequiredError Option( std::size_t min_option,
264  std::size_t max_option,
265  std::size_t used,
266  const std::string& option_list ) {
267  if ( ( min_option == 1 ) && ( max_option == 1 ) && ( used == 0 ) )
268  return RequiredError( "Exactly 1 option from [" + option_list + "]" );
269  if ( ( min_option == 1 ) && ( max_option == 1 ) && ( used > 1 ) ) {
270  return RequiredError( "Exactly 1 option from [" + option_list + "] is required and " +
271  std::to_string( used ) + " were given",
272  ExitCodes::RequiredError );
273  }
274  if ( ( min_option == 1 ) && ( used == 0 ) )
275  return RequiredError( "At least 1 option from [" + option_list + "]" );
276  if ( used < min_option ) {
277  return RequiredError( "Requires at least " + std::to_string( min_option ) +
278  " options used and only " + std::to_string( used ) +
279  "were given from [" + option_list + "]",
280  ExitCodes::RequiredError );
281  }
282  if ( max_option == 1 )
283  return RequiredError( "Requires at most 1 options be given from [" + option_list + "]",
284  ExitCodes::RequiredError );
285 
286  return RequiredError( "Requires at most " + std::to_string( max_option ) +
287  " options be used and " + std::to_string( used ) +
288  "were given from [" + option_list + "]",
289  ExitCodes::RequiredError );
290  }
