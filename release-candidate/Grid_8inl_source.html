11template <typename T, uint D>
12inline typename Grid<T, D>::IdxVector
13linearToIdxVector( uint linIdx, const typename Grid<T, D>::IdxVector& size ) {
14 typename Grid<T, D>::IdxVector result = Grid<T, D>::IdxVector::Zero();
15
16 for ( uint i = 0; i < D; ++i ) {
17 result[i] = linIdx % size[i];
18 linIdx = linIdx / size[i];
19 }
20 return result;
21}
22
23template <typename T, uint D>
24inline uint idxVectorToLinear( const typename Grid<T, D>::IdxVector& vecIdx,
25 const typename Grid<T, D>::IdxVector& size ) {
26 uint result = 0;
27 uint dimProd = 1;
28 for ( uint i = 0; i < D; ++i ) {
29 result += vecIdx[i] * dimProd;
30 dimProd *= size[i];
31 }
32 return result;
33}
