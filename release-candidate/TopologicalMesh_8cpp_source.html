68 bool TopologicalMesh::checkIntegrity() const {
69  std::vector<unsigned int> count( m_wedges.size(), 0 );
70  bool ret = true;
71  for ( auto he_itr { halfedges_begin() }; he_itr != halfedges_end(); ++he_itr ) {
72  auto widx = property( m_wedgeIndexPph, *he_itr );
73 
74  if ( status( *he_itr ).deleted() ) { continue; }
75 
76  if ( is_boundary( *he_itr ) != widx.isInvalid() ) {
77  LOG( logWARNING ) << "topological mesh wedge inconsistency, boundary he ("
78  << ( is_boundary( *he_itr ) ? "true," : "false," ) << he_itr->idx()
79  << ") != invalid Wedge (" << ( widx.isInvalid() ? "true," : "false," )
80  << widx << ") ref "
81  << ( widx.isValid() ? m_wedges.getWedge( widx ).getRefCount() : 0 );
82  ret = false;
83  }
84 
85  if ( widx.isValid() ) // i.e. non boudnary
86  {
87  count[widx]++;
88 
89  if ( m_wedges.getWedgeData( widx ).m_position !=
90  point( to_vertex_handle( *he_itr ) ) ) {
91  LOG( logWARNING ) << "topological mesh wedge inconsistency, wedge and to position "
92  "differ for widx "
93  << widx << ", have ("
94  << m_wedges.getWedgeData( widx ).m_position.transpose()
95  << ") instead of ("
96  << point( to_vertex_handle( *he_itr ) ).transpose() << ")";
97  ret = false;
98  }
99  }
100  }
101 
102  for ( int widx = 0; widx < int( m_wedges.size() ); ++widx ) {
103  if ( m_wedges.getWedge( WedgeIndex { widx } ).getRefCount() != count[widx] ) {
104  LOG( logWARNING ) << "topological mesh wedge count inconsistency, topo count [ "
105  << count[widx] << " ] != wedge count [ "
106  << m_wedges.getWedge( WedgeIndex { widx } ).getRefCount()
107  << " ] for id " << widx;
108  ret = false;
109  }
110  }
111  return ret;
112 }

114 void TopologicalMesh::triangulate() {
115 
116  auto fix = [this]( HalfedgeHandle next_he, const std::vector<HalfedgeHandle>& old_heh ) {
117  // tagged if already fixed
118  auto to_vh = to_vertex_handle( next_he );
119  // find ref in old_he to copy wedge idx
120 
121  auto ref = std::find_if(
122  old_heh.begin(), old_heh.end(), [this, to_vh]( const HalfedgeHandle& he ) {
123  return to_vertex_handle( he ) == to_vh;
124  } );
125  if ( ref != old_heh.end() ) {
126  property( m_wedgeIndexPph, next_he ) =
127  m_wedges.newReference( property( m_wedgeIndexPph, *ref ) );
128  }
129  else {
130  LOG( logERROR ) << "triangulate::fix reference halfedge not found";
131  }
132  status( next_he ).set_tagged( true );
133  };
134 
135  FaceIter f_it( faces_begin() ), f_end( faces_end() );
136  for ( ; f_it != f_end; ++f_it ) {
137  // save original halfedge of the face
138  std::vector<HalfedgeHandle> old_heh;
139  ConstFaceHalfedgeIter fh_itr = cfh_iter( *f_it );
140  for ( ; fh_itr.is_valid(); ++fh_itr ) {
141  old_heh.push_back( *fh_itr );
142  }
143  auto size = old_heh.size();
144  // if ( size <= 3 ) continue;
145 
146  // base openmesh triangulate
147  base::triangulate( *f_it );
148 
149  // fix newly created he
150  for ( size_t i = 0; i < size; ++i ) {
151  auto next_he = next_halfedge_handle( old_heh[i] );
152  // if next_he is not the same as next in old_heh, then it's a new one.
153  // fix tag halfedge so that it is not fixed two times (in case opposite halfedge is also
154  // parsed in this loop.
155  if ( !status( next_he ).tagged() && next_he != old_heh[( i + 1 ) % size] ) {
156  fix( next_he, old_heh );
157  fix( opposite_halfedge_handle( next_he ), old_heh );
158  }
159  }
160  }
161  // untag everything
162  for ( auto& he : halfedges() ) {
163  status( he ).set_tagged( false );
164  }
165 }

649 bool TopologicalMesh::splitEdge( TopologicalMesh::EdgeHandle eh, Scalar f ) {
650  // Global schema of operation
651  /*
652  TRIANGLES ONLY
653  before after
654 
655  / \ / | \
656  / \ / | \
657  /h2 h1\ /r2 r1|s2 s1\
658  / h0 -->\ / r0 | s0=h0 \
659  V0 -------- V1 V0-(1-f)-V---f---V1
660  <-- o0 / \ u0 | t0=o0 /
661  \o1 o2/ \u1 u2|t1 t2/
662  \ / \ | /
663  \ / \ | /
664 
665 
666  */
667 
668  // incorrect factor
669  if ( f < 0 || f > 1 ) { return false; }
670 
671  const auto h0 = halfedge_handle( eh, 0 );
672  const auto o0 = halfedge_handle( eh, 1 );
673 
674  const auto v0 = to_vertex_handle( o0 );
675  const auto v1 = to_vertex_handle( h0 );
676 
677  const auto h1 = next_halfedge_handle( h0 );
678  const auto h2 = next_halfedge_handle( h1 );
679  const auto o1 = next_halfedge_handle( o0 );
680  const auto o2 = next_halfedge_handle( o1 );
681 
682  // add the new point
683  const Point p = Point( f * point( v1 ) + ( 1_ra - f ) * point( v0 ) );
684  VertexHandle vh = add_vertex( p );
685 
686  // compute interpolated wedge, or the two wedges if not the same wedges
687  // around the two vertices of the edge (we always compute for two wedges,
688  // even if add will return the same wedge.
689  WedgeIndex ovwidx;
690  WedgeIndex hvwidx;
691  if ( !is_boundary( h0 ) ) {
692 
693  const auto hw0idx = property( m_wedgeIndexPph, h2 );
694  const auto hw1idx = property( m_wedgeIndexPph, h0 );
695  const auto hw0 = m_wedges.getWedgeData( hw0idx );
696  const auto hw1 = m_wedges.getWedgeData( hw1idx );
697  auto hvw = interpolateWedgeAttributes( hw0, hw1, f );
698  hvw.m_position = p;
699  hvwidx = m_wedges.add( hvw );
700  }
701  if ( !is_boundary( o0 ) ) {
702 
703  const auto ow0idx = property( m_wedgeIndexPph, o2 );
704  const auto ow1idx = property( m_wedgeIndexPph, o0 );
705  const auto ow0 = m_wedges.getWedgeData( ow0idx );
706  const auto ow1 = m_wedges.getWedgeData( ow1idx );
707  auto ovw = interpolateWedgeAttributes( ow1, ow0, f );
708  ovw.m_position = p;
709  ovwidx = m_wedges.add( ovw );
710  }
711 
712  split_copy( eh, vh );
713 
714  auto r0 = find_halfedge( v0, vh );
715  auto s0 = find_halfedge( vh, v1 );
716  auto t0 = find_halfedge( v1, vh );
717  auto u0 = find_halfedge( vh, v0 );
718 
719  auto r1 = next_halfedge_handle( r0 );
720  auto r2 = next_halfedge_handle( r1 );
721 
722  auto s1 = next_halfedge_handle( s0 );
723  auto s2 = next_halfedge_handle( s1 );
724 
725  auto t1 = next_halfedge_handle( t0 );
726  auto t2 = next_halfedge_handle( t1 );
727 
728  auto u1 = next_halfedge_handle( u0 );
729  auto u2 = next_halfedge_handle( u1 );
730 
731  CORE_ASSERT( s0 == h0, "TopologicalMesh: splitEdgeWedge inconsistency" );
732  CORE_ASSERT( t0 == o0, "TopologicalMesh: splitEdgeWedge inconsistency" );
733 
734  auto updateWedgeIndex1 = [this]( WedgeIndex widx_,
735  HalfedgeHandle r0_,
736  HalfedgeHandle r1_,
737  HalfedgeHandle r2_,
738  HalfedgeHandle h1_,
739  HalfedgeHandle h2_ ) {
740  CORE_UNUSED( r2_ );
741  CORE_UNUSED( h2_ );
742 
743  if ( !is_boundary( r0_ ) ) {
744  CORE_ASSERT( r2_ == h2_, "TopologicalMesh: splitEdgeWedge inconsistency" );
745 
746  // Increment here, the first reference is for the other he
747  property( this->m_wedgeIndexPph, r0_ ) = this->m_wedges.newReference( widx_ );
748  property( this->m_wedgeIndexPph, r1_ ) =
749  this->m_wedges.newReference( property( this->m_wedgeIndexPph, h1_ ) );
750  }
751  else {
752  property( this->m_wedgeIndexPph, r0_ ) = WedgeIndex {};
753  }
754  };
755 
756  auto updateWedgeIndex2 = [this]( WedgeIndex widx_,
757  HalfedgeHandle s0_,
758  HalfedgeHandle s1_,
759  HalfedgeHandle s2_,
760  HalfedgeHandle h0_,
761  HalfedgeHandle h1_ ) {
762  CORE_UNUSED( s1_ );
763  CORE_UNUSED( h1_ );
764 
765  if ( !is_boundary( s0_ ) ) {
766  CORE_ASSERT( s1_ == h1_, "TopologicalMesh: splitEdgeWedge inconsistency" );
767  // do not increment here, since add has set ref to 1
768  property( this->m_wedgeIndexPph, s2_ ) = widx_;
769  // "steal" ref from previous he (actually s0 is h0, u0 is the steal
770  // from o0.
771  property( this->m_wedgeIndexPph, s0_ ) = property( this->m_wedgeIndexPph, h0_ );
772  }
773  else {
774  property( this->m_wedgeIndexPph, s0_ ) = WedgeIndex {};
775  }
776  };
777 
778  // this update read from o0, must be done before t which reads from o0
779  updateWedgeIndex2( ovwidx, u0, u1, u2, o0, o1 );
780  // those update might be in any order
781  updateWedgeIndex2( hvwidx, s0, s1, s2, h0, h1 );
782  updateWedgeIndex1( hvwidx, r0, r1, r2, h1, h2 );
783  updateWedgeIndex1( ovwidx, t0, t1, t2, o1, o2 );
784 
785  return true;
786 }
