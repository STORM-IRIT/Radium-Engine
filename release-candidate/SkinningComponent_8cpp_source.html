43 TriangleMesh triangulate( const PolyMesh& polyMesh ) {
44  TriangleMesh res;
45  res.setVertices( polyMesh.vertices() );
46  res.setNormals( polyMesh.normals() );
47  res.copyAllAttributes( polyMesh );
48  VectorArray<Vector3ui> indices;
49  // using the same triangulation as in Ra::Engine::GeneralMesh::triangulate
50  for ( const auto& face : polyMesh.getIndices() ) {
51  if ( face.size() == 3 ) { indices.push_back( face ); }
52  else {
53  int minus { int( face.size() ) - 1 };
54  int plus { 0 };
55  while ( plus + 1 < minus ) {
56  if ( ( plus - minus ) % 2 ) {
57  indices.emplace_back( face[plus], face[plus + 1], face[minus] );
58  ++plus;
59  }
60  else {
61  indices.emplace_back( face[minus], face[plus], face[minus - 1] );
62  --minus;
63  }
64  }
65  }
66  }
67  res.setIndices( std::move( indices ) );
68  return res;
69 }

110  else if ( m_meshIsQuad ) {

114  m_refData.m_referenceMesh = triangulate( *m_polyMeshWriter() );

118  !m_refData.m_referenceMesh.hasAttrib( bitangentName ) ) {
119  const auto& normals = m_refData.m_referenceMesh.normals();
120  Vector3Array tangents( normals.size() );
121  Vector3Array bitangents( normals.size() );
122 #pragma omp parallel for
123  for ( int i = 0; i < int( normals.size() ); ++i ) {
124  Core::Math::getOrthogonalVectors( normals[i], tangents[i], bitangents[i] );
125  }

130  const auto& normals = m_refData.m_referenceMesh.normals();

132  const auto& bitangents = m_refData.m_referenceMesh.getAttrib( bH ).data();
133  Vector3Array tangents( normals.size() );
134 #pragma omp parallel for
135  for ( int i = 0; i < int( normals.size() ); ++i ) {
136  tangents[i] = bitangents[i].cross( normals[i] );
137  }
138  m_refData.m_referenceMesh.addAttrib( tangentName, std::move( tangents ) );
139  }
140  else if ( !m_refData.m_referenceMesh.hasAttrib( bitangentName ) ) {
141  const auto& normals = m_refData.m_referenceMesh.normals();
142  const auto& tH = m_refData.m_referenceMesh.getAttribHandle<Vector3>( tangentName );
143  const auto& tangents = m_refData.m_referenceMesh.getAttrib( tH ).data();
144  Vector3Array bitangents( normals.size() );
145 #pragma omp parallel for
146  for ( int i = 0; i < int( normals.size() ); ++i ) {
147  bitangents[i] = normals[i].cross( tangents[i] );
148  }
149  m_refData.m_referenceMesh.addAttrib( bitangentName, std::move( bitangents ) );
150  }

153

155  // get other data
156  m_refData.m_meshTransformInverse = ro->getLocalTransform().inverse();
157  m_refData.m_skeleton = *m_skeletonGetter();
158  createWeightMatrix();
159 
160  // initialize frame data
161  m_frameData.m_skeleton = m_refData.m_skeleton;
162  m_frameData.m_currentPosition = m_refData.m_referenceMesh.vertices();
163  m_frameData.m_currentNormal = m_refData.m_referenceMesh.normals();
164  const auto& tH = m_refData.m_referenceMesh.getAttribHandle<Vector3>( tangentName );
165  m_frameData.m_currentTangent = m_refData.m_referenceMesh.getAttrib( tH ).data();
166  const auto& bH = m_refData.m_referenceMesh.getAttribHandle<Vector3>( bitangentName );
167  m_frameData.m_currentBitangent = m_refData.m_referenceMesh.getAttrib( bH ).data();
168  m_frameData.m_frameCounter = 0;
169  m_frameData.m_doSkinning = true;
170  m_frameData.m_doReset = false;
171 
172  // setup comp data
173  m_isReady = true;
174  m_forceUpdate = true;
175 
176  // prepare RO for skinning weights display
177  m_baseMaterial = ro->getMaterial();
178 
179  // prepare UV
180  auto attrUV = Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_TEXCOORD );
181  AttribArrayGeometry* geom;
182  if ( hasTriMesh ) { geom = const_cast<TriangleMesh*>( m_triMeshWriter() ); }
183  else {
184  if ( m_meshIsPoly ) { geom = const_cast<PolyMesh*>( m_polyMeshWriter() ); }
185  else {
186  geom = const_cast<QuadMesh*>( m_quadMeshWriter() );
187  }
188  }
189  if ( geom->hasAttrib( attrUV ) ) {
190  auto handle = geom->getAttribHandle<Vector3>( attrUV );
191  m_baseUV = geom->getAttrib( handle ).data();

193 
194  // prepare weights material
195  auto mat = new Data::BlinnPhongMaterial( "SkinningWeights_Mat" );
196  mat->m_kd = Color::Skin();

210 void SkinningComponent::skin() {
211  CORE_ASSERT( m_isReady, "Skinning is not setup" );
212 
213  const Skeleton* skel = m_skeletonGetter();
214 
215  bool reset = ComponentMessenger::getInstance()->get<bool>( getEntity(), m_skelName );
216 
217  // Reset the skin if it wasn't done before
218  if ( reset && !m_frameData.m_doReset ) {
219  m_frameData.m_doReset = true;
220  m_frameData.m_frameCounter = 0;
221  m_forceUpdate = true;
222  }
223  const auto prevPose = m_frameData.m_skeleton.getPose( SpaceType::MODEL );
224  m_frameData.m_skeleton = *skel;
225  auto currentPose = m_frameData.m_skeleton.getPose( SpaceType::MODEL );
226  if ( !areEqual( currentPose, prevPose ) || m_forceUpdate ) {
227  m_forceUpdate = false;
228  m_frameData.m_doSkinning = true;
229  m_frameData.m_frameCounter++;
230 
231  const auto tH = m_refData.m_referenceMesh.getAttribHandle<Vector3>( tangentName );
232  const Vector3Array& tangents = m_refData.m_referenceMesh.getAttrib( tH ).data();
233  const auto bH = m_refData.m_referenceMesh.getAttribHandle<Vector3>( bitangentName );
234  const Vector3Array& bitangents = m_refData.m_referenceMesh.getAttrib( bH ).data();
235 
236  switch ( m_skinningType ) {
237  case DQS: {
238  dualQuaternionSkinning( m_refData, tangents, bitangents, m_frameData );
239  break;
240  }
241  case COR: {
242  centerOfRotationSkinning( m_refData, tangents, bitangents, m_frameData );
243  break;
244  }
245  case LBS:
246  default: {
247  linearBlendSkinning( m_refData, tangents, bitangents, m_frameData );
248  break;
249  }
250  }
251 
252  if ( m_normalSkinning == GEOMETRIC ) {
253  m_topoMesh.updatePositions( m_frameData.m_currentPosition );
254  m_topoMesh.updateWedgeNormals();
255  m_topoMesh.updateTriangleMeshNormals( m_frameData.m_currentNormal );
256 #pragma omp parallel for
257  for ( int i = 0; i < int( m_frameData.m_currentNormal.size() ); ++i ) {
258  Core::Math::getOrthogonalVectors( m_frameData.m_currentNormal[i],
259  m_frameData.m_currentTangent[i],
260  m_frameData.m_currentBitangent[i] );
261  }
262  }
263  }

317 void SkinningComponent::createWeightMatrix() {
318  m_refData.m_bindMatrices.resize( m_refData.m_skeleton.size(), Transform::Identity() );
319  m_refData.m_weights.resize( int( m_refData.m_referenceMesh.vertices().size() ),
320  m_refData.m_skeleton.size() );
321  std::vector<Eigen::Triplet<Scalar>> triplets;
322  for ( uint col = 0; col < m_refData.m_skeleton.size(); ++col ) {
323  std::string boneName = m_refData.m_skeleton.getLabel( col );
324  auto it = m_loadedWeights.find( boneName );
325  if ( it != m_loadedWeights.end() ) {
326  const auto& W = it->second;
327  for ( uint i = 0; i < W.size(); ++i ) {
328  const auto& w = W[i];
329  int row { int( w.first ) };
330  CORE_ASSERT( row < m_refData.m_weights.rows(),
331  "Weights are incompatible with mesh." );
332  triplets.push_back( { row, int( col ), w.second } );
333  }
334  m_refData.m_bindMatrices[col] = m_loadedBindMatrices[boneName];
335  }
336  }
337  m_refData.m_weights.setFromTriplets( triplets.begin(), triplets.end() );
338 
339  checkWeightMatrix( m_refData.m_weights, false, true );
340 
341  if ( normalizeWeights( m_refData.m_weights, true ) ) {
342  LOG( logINFO ) << "Skinning weights have been normalized";
343  }

380  m_showingWeights = on;

382  auto attrUV = Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_TEXCOORD );

385  AttribArrayGeometry* geom;
386  if ( !m_meshIsPoly ) {

388  else {

390  }

392  else {
393  geom = const_cast<PolyMesh*>( m_polyMeshWriter() );

398  const auto size = m_frameData.m_currentPosition.size();

414  handle = geom->addAttrib<Vector3>( attrUV );
415  geom->getAttrib( handle ).setData( m_weightsUV );
416  }
417  else {
418  // change the material
419  ro->setMaterial( m_baseMaterial );
420  ro->getRenderTechnique()->setParametersProvider( m_baseMaterial );
421  // if the UV attrib existed before, reset it, otherwise remove it.
422  handle = geom->getAttribHandle<Vector3>( attrUV );
423  if ( m_baseUV.size() > 0 ) { geom->getAttrib( handle ).setData( m_baseUV ); }
424  else {
425  geom->removeAttrib( handle );
426  }
427  }
