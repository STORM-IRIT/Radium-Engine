43 inline std::string Formatter::make_groups( const App* app, AppFormatMode mode ) const {
44  std::stringstream out;
45  std::vector<std::string> groups = app->get_groups();
46 
47  // Options
48  for ( const std::string& group : groups ) {
49  std::vector<const Option*> opts =
50  app->get_options( [app, mode, &group]( const Option* opt ) {
51  return opt->get_group() == group // Must be in the right group
52  && opt->nonpositional() // Must not be a positional
53  && ( mode != AppFormatMode::Sub // If mode is Sub, then
54  || ( app->get_help_ptr() != opt // Ignore help pointer
55  && app->get_help_all_ptr() != opt ) ); // Ignore help all pointer
56  } );
57  if ( !group.empty() && !opts.empty() ) {
58  out << make_group( group, false, opts );
59 
60  if ( group != groups.back() ) out << "\n";
61  }
62  }
63 
64  return out.str();
65 }

67 inline std::string Formatter::make_description( const App* app ) const {
68  std::string desc = app->get_description();
69  auto min_options = app->get_require_option_min();
70  auto max_options = app->get_require_option_max();
71  if ( app->get_required() ) { desc += " REQUIRED "; }
72  if ( ( max_options == min_options ) && ( min_options > 0 ) ) {
73  if ( min_options == 1 ) { desc += " \n[Exactly 1 of the following options is required]"; }
74  else {
75  desc += " \n[Exactly " + std::to_string( min_options ) +
76  "options from the following list are required]";
77  }
78  }
79  else if ( max_options > 0 ) {
80  if ( min_options > 0 ) {
81  desc += " \n[Between " + std::to_string( min_options ) + " and " +
82  std::to_string( max_options ) + " of the follow options are required]";
83  }
84  else {
85  desc += " \n[At most " + std::to_string( max_options ) +
86  " of the following options are allowed]";
87  }
88  }
89  else if ( min_options > 0 ) {
90  desc += " \n[At least " + std::to_string( min_options ) +
91  " of the following options are required]";
92  }
93  return ( !desc.empty() ) ? desc + "\n" : std::string {};
94 }

96 inline std::string Formatter::make_usage( const App* app, std::string name ) const {
97  std::stringstream out;
98 
99  out << get_label( "Usage" ) << ":" << ( name.empty() ? "" : " " ) << name;
100 
101  std::vector<std::string> groups = app->get_groups();
102 
103  // Print an Options badge if any options exist
104  std::vector<const Option*> non_pos_options =
105  app->get_options( []( const Option* opt ) { return opt->nonpositional(); } );
106  if ( !non_pos_options.empty() ) out << " [" << get_label( "OPTIONS" ) << "]";
107 
108  // Positionals need to be listed here
109  std::vector<const Option*> positionals =
110  app->get_options( []( const Option* opt ) { return opt->get_positional(); } );
111 
112  // Print out positionals if any are left
113  if ( !positionals.empty() ) {
114  // Convert to help names
115  std::vector<std::string> positional_names( positionals.size() );
116  std::transform( positionals.begin(),
117  positionals.end(),
118  positional_names.begin(),
119  [this]( const Option* opt ) { return make_option_usage( opt ); } );
120 
121  out << " " << detail::join( positional_names, " " );
122  }
123 
124  // Add a marker if subcommands are expected or optional
125  if ( !app->get_subcommands( []( const CLI::App* subc ) {
126  return ( ( !subc->get_disabled() ) && ( !subc->get_name().empty() ) );
127  } )
128  .empty() ) {
129  out << " " << ( app->get_require_subcommand_min() == 0 ? "[" : "" )
130  << get_label( app->get_require_subcommand_max() < 2 ||
131  app->get_require_subcommand_min() > 1
132  ? "SUBCOMMAND"
133  : "SUBCOMMANDS" )
134  << ( app->get_require_subcommand_min() == 0 ? "]" : "" );
135  }
136 
137  out << std::endl;
138 
139  return out.str();
140 }

148 inline std::string
149 Formatter::make_help( const App* app, std::string name, AppFormatMode mode ) const {
150 
151  // This immediately forwards to the make_expanded method. This is done this way so that
152  // subcommands can have overridden formatters
153  if ( mode == AppFormatMode::Sub ) return make_expanded( app );
154 
155  std::stringstream out;
156  if ( ( app->get_name().empty() ) && ( app->get_parent() != nullptr ) ) {
157  if ( app->get_group() != "Subcommands" ) { out << app->get_group() << ':'; }
158  }
159 
160  out << make_description( app );
161  out << make_usage( app, name );
162  out << make_positionals( app );
163  out << make_groups( app, mode );
164  out << make_subcommands( app, mode );
165  out << '\n' << make_footer( app );
166 
167  return out.str();
168 }

170 inline std::string Formatter::make_subcommands( const App* app, AppFormatMode mode ) const {
171  std::stringstream out;
172 
173  std::vector<const App*> subcommands = app->get_subcommands( {} );
174 
175  // Make a list in definition order of the groups seen
176  std::vector<std::string> subcmd_groups_seen;
177  for ( const App* com : subcommands ) {
178  if ( com->get_name().empty() ) {
179  if ( !com->get_group().empty() ) { out << make_expanded( com ); }
180  continue;
181  }
182  std::string group_key = com->get_group();
183  if ( !group_key.empty() && std::find_if( subcmd_groups_seen.begin(),
184  subcmd_groups_seen.end(),
185  [&group_key]( std::string a ) {
186  return detail::to_lower( a ) ==
187  detail::to_lower( group_key );
188  } ) == subcmd_groups_seen.end() )
189  subcmd_groups_seen.push_back( group_key );
190  }
191 
192  // For each group, filter out and print subcommands
193  for ( const std::string& group : subcmd_groups_seen ) {
194  out << "\n" << group << ":\n";
195  std::vector<const App*> subcommands_group =
196  app->get_subcommands( [&group]( const App* sub_app ) {
197  return detail::to_lower( sub_app->get_group() ) == detail::to_lower( group );
198  } );
199  for ( const App* new_com : subcommands_group ) {
200  if ( new_com->get_name().empty() ) continue;
201  if ( mode != AppFormatMode::All ) { out << make_subcommand( new_com ); }
202  else {
203  out << new_com->help( new_com->get_name(), AppFormatMode::Sub );
204  out << "\n";
205  }
206  }
207  }
208 
209  return out.str();
210 }

219 inline std::string Formatter::make_expanded( const App* sub ) const {
220  std::stringstream out;
221  out << sub->get_display_name( true ) << "\n";
222 
223  out << make_description( sub );
224  if ( sub->get_name().empty() && !sub->get_aliases().empty() ) {
225  detail::format_aliases( out, sub->get_aliases(), column_width_ + 2 );
226  }
227  out << make_positionals( sub );
228  out << make_groups( sub, AppFormatMode::Sub );
229  out << make_subcommands( sub, AppFormatMode::Sub );
230 
231  // Drop blank spaces
232  std::string tmp = detail::find_and_replace( out.str(), "\n\n", "\n" );
233  tmp = tmp.substr( 0, tmp.size() - 1 ); // Remove the final '\n'
234 
235  // Indent all but the first line (the name)
236  return detail::find_and_replace( tmp, "\n", "\n " ) + "\n";
237 }

245 inline std::string Formatter::make_option_opts( const Option* opt ) const {
246  std::stringstream out;
247 
248  if ( !opt->get_option_text().empty() ) { out << " " << opt->get_option_text(); }
249  else {
250  if ( opt->get_type_size() != 0 ) {
251  if ( !opt->get_type_name().empty() ) out << " " << get_label( opt->get_type_name() );
252  if ( !opt->get_default_str().empty() ) out << "=" << opt->get_default_str();
253  if ( opt->get_expected_max() == detail::expected_max_vector_size )
254  out << " ...";
255  else if ( opt->get_expected_min() > 1 )
256  out << " x " << opt->get_expected();
257 
258  if ( opt->get_required() ) out << " " << get_label( "REQUIRED" );
259  }
260  if ( !opt->get_envname().empty() )
261  out << " (" << get_label( "Env" ) << ":" << opt->get_envname() << ")";
262  if ( !opt->get_needs().empty() ) {
263  out << " " << get_label( "Needs" ) << ":";
264  for ( const Option* op : opt->get_needs() )
265  out << " " << op->get_name();
266  }
267  if ( !opt->get_excludes().empty() ) {
268  out << " " << get_label( "Excludes" ) << ":";
269  for ( const Option* op : opt->get_excludes() )
270  out << " " << op->get_name();
271  }
272  }
273  return out.str();
274 }
