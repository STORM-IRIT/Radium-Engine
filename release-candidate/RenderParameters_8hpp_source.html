407void RenderParameters::addEnumConverter(
408 const std::string& name,
409 std::shared_ptr<Core::Utils::EnumConverter<EnumBaseType>> converter ) {
410 auto converterHandle = m_parameterSets.insertOrAssignVariable( name, converter );
411 std::function<void( Core::VariableSet&, const std::string&, const std::string& )>
412 convertingFunction = [converter = converterHandle.first]( Core::VariableSet& vs,
413 const std::string& nm,
414 const std::string& vl ) {
415 vs.insertOrAssignVariable( nm, converter->second->getEnumerator( vl ) );
416 };
417 m_parameterSets.insertOrAssignVariable( name, convertingFunction );
418}
