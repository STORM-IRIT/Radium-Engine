96template <uint D, uint K>
97inline void Spline<D, K>::setNodeToUniform() {
98 const uint n = m_points.size() - 1;
99 m_node.resize( K + n + 1 );
100
101 Scalar step = 1.f / Scalar( n - K + 2 );
102 for ( uint i = 0; i < m_node.size(); ++i ) {
103 m_node[i] = Scalar( i ) * step - step * ( Scalar )( K - 1 );
104 }

109template <uint D, uint K>
110inline void Spline<D, K>::setNodeToOpenUniform() {
111 m_node.resize( K + m_points.size() );
112
113 uint acc = 1;
114 for ( uint i = 0; i < m_node.size(); ++i ) {
115 if ( i < K ) { m_node[i] = 0.f; }
116 else if ( i >= ( m_points.size() + 1 ) ) {
117 m_node[i] = 1.f;
118 }
119 else {
120 m_node[i] = Scalar( acc ) / Scalar( m_points.size() + 1 - K );
121 acc++;
122 }
123 }

158template <uint D, uint K>
159inline typename Spline<D, K>::Vector Spline<D, K>::evalRec( Scalar u,
160 const Core::VectorArray<Vector>& points,
161 const std::vector<Scalar>& node,
162 uint k ) {
163 if ( points.size() == 1 ) { return points[0]; }
164
165 // TODO: use buffers in attributes for better performances ?
166 Core::VectorArray<Vector> pOut( k - 1, Vector::Zero() );
167
168 for ( uint i = 0; i < ( k - 1 ); ++i ) {
169 const Scalar n0 = node[i + k - 1];
170 const Scalar n1 = node[i];
171 const Scalar f0 = ( n0 - u ) / ( n0 - n1 );
172 const Scalar f1 = ( u - n1 ) / ( n0 - n1 );
173
174 pOut[i] = points[i] * f0 + points[i + 1] * f1;
175 }
176
177 std::vector<Scalar> nodeOut( node.size() - 2 );
178
179 for ( uint i = 1; i < node.size() - 1; ++i ) {
180 nodeOut[i - 1] = node[i];
181 }
182 return evalRec( u, pOut, nodeOut, k - 1 );
183}
