594  Option* add_option( std::string option_name,
595  callback_t option_callback,
596  std::string option_description = "",
597  bool defaulted = false,
598  std::function<std::string()> func = {} ) {
599  Option myopt { option_name, option_description, option_callback, this };
600 
601  if ( std::find_if(
602  std::begin( options_ ), std::end( options_ ), [&myopt]( const Option_p& v ) {
603  return *v == myopt;
604  } ) == std::end( options_ ) ) {
605  options_.emplace_back();
606  Option_p& option = options_.back();
607  option.reset( new Option( option_name, option_description, option_callback, this ) );
608 
609  // Set the default string capture function
610  option->default_function( func );
611 
612  // For compatibility with CLI11 1.7 and before, capture the default string here
613  if ( defaulted ) option->capture_default_str();
614 
615  // Transfer defaults to the new option
616  option_defaults_.copy_to( option.get() );
617 
618  // Don't bother to capture if we already did
619  if ( !defaulted && option->get_always_capture_default() ) option->capture_default_str();
620 
621  return option.get();
622  }
623  // we know something matches now find what it is so we can produce more error information
624  for ( auto& opt : options_ ) {
625  auto& matchname = opt->matching_name( myopt );
626  if ( !matchname.empty() ) {
627  throw( OptionAlreadyAdded( "added option matched existing option name: " +
628  matchname ) );
629  }
630  }
631  // this line should not be reached the above loop should trigger the throw
632  throw(
633  OptionAlreadyAdded( "added option matched existing option name" ) ); // LCOV_EXCL_LINE
634  }
