33inline bool TopologicalMesh::WedgeData::operator<( const TopologicalMesh::WedgeData& lhs ) const {
34
35 CORE_ASSERT( ( m_floatAttrib.size() == lhs.m_floatAttrib.size() ) &&
36 ( m_vector2Attrib.size() == lhs.m_vector2Attrib.size() ) &&
37 ( m_vector3Attrib.size() == lhs.m_vector3Attrib.size() ) &&
38 ( m_vector4Attrib.size() == lhs.m_vector4Attrib.size() ),
39 "Could only compare wedge with same number of attributes" );
40
41 {
42 int comp = compareVector( m_position, lhs.m_position );
43 if ( comp == 2 ) return true;
44 if ( comp == 3 ) return false;
45 }
46 for ( size_t i = 0; i < m_floatAttrib.size(); i++ ) {
47 if ( m_floatAttrib[i] < lhs.m_floatAttrib[i] )
48 return true;
49 else if ( m_floatAttrib[i] > lhs.m_floatAttrib[i] )
50 return false;
51 }
52
53 for ( size_t i = 0; i < m_vector2Attrib.size(); i++ ) {
54 int comp = compareVector( m_vector2Attrib[i], lhs.m_vector2Attrib[i] );
55 if ( comp == 2 ) return true;
56 if ( comp == 3 ) return false;
57 }
58 for ( size_t i = 0; i < m_vector3Attrib.size(); i++ ) {
59 int comp = compareVector( m_vector3Attrib[i], lhs.m_vector3Attrib[i] );
60 if ( comp == 2 ) return true;
61 if ( comp == 3 ) return false;
62 }
63 for ( size_t i = 0; i < m_vector4Attrib.size(); i++ ) {
64 int comp = compareVector( m_vector4Attrib[i], lhs.m_vector4Attrib[i] );
65 if ( comp == 2 ) return true;
66 if ( comp == 3 ) return false;
67 }
68 return false;
69}

92// return 1 : equals, 2: strict less, 3: strict greater
93template <typename T>
94int TopologicalMesh::WedgeData::compareVector( const T& a, const T& b ) {
95 for ( int i = 0; i < T::RowsAtCompileTime; i++ ) {
96 if ( a[i] < b[i] ) return 2;
97 if ( a[i] > b[i] ) return 3;
98 }
99 // (a == b)
100 return 1;
101}

363inline void
364TopologicalMesh::InitWedgeAttribsFromMultiIndexedGeometry::operator()( AttribBase* attr ) const {
365 if ( attr->getSize() != m_triMesh.vertices().size() ) {
366 LOG( logWARNING ) << "[TopologicalMesh] Skip badly sized attribute " << attr->getName();
367 }
368 else if ( attr->getName() != getAttribName( MeshAttrib::VERTEX_POSITION ) ) {
369 if ( attr->isFloat() ) {
370 m_topo->m_wedges.m_wedgeFloatAttribHandles.push_back(
371 m_triMesh.template getAttribHandle<Scalar>( attr->getName() ) );
372 m_topo->m_wedges.addAttribName<Scalar>( attr->getName() );
373 }
374 else if ( attr->isVector2() ) {
375 m_topo->m_wedges.m_wedgeVector2AttribHandles.push_back(
376 m_triMesh.template getAttribHandle<Vector2>( attr->getName() ) );
377 m_topo->m_wedges.addAttribName<Vector2>( attr->getName() );
378 }
379 else if ( attr->isVector3() ) {
380 m_topo->m_wedges.m_wedgeVector3AttribHandles.push_back(
381 m_triMesh.template getAttribHandle<Vector3>( attr->getName() ) );
382 m_topo->m_wedges.addAttribName<Vector3>( attr->getName() );
383 }
384 else if ( attr->isVector4() ) {
385 m_topo->m_wedges.m_wedgeVector4AttribHandles.push_back(
386 m_triMesh.template getAttribHandle<Vector4>( attr->getName() ) );
387 m_topo->m_wedges.addAttribName<Vector4>( attr->getName() );
388 }
389 else
390 LOG( logWARNING )
391 << "Warning, mesh attribute " << attr->getName()
392 << " type is not supported (only float, vec2, vec3 nor vec4 are supported)";
393 }

400struct hash_vec {
401 std::size_t operator()( const Vector3& lvalue ) const {
402 size_t hx = std::hash<Scalar>()( lvalue[0] );
403 size_t hy = std::hash<Scalar>()( lvalue[1] );
404 size_t hz = std::hash<Scalar>()( lvalue[2] );
405 return ( hx ^ ( hy << 1 ) ) ^ hz;
406 }
407};

442template <typename NonManifoldFaceCommand>
443void TopologicalMesh::initWithWedge(
444 const Ra::Core::Geometry::MultiIndexedGeometry& mesh,
445 const Ra::Core::Geometry::MultiIndexedGeometry::LayerKeyType& layerKey,
446 NonManifoldFaceCommand command ) {
447
448 const auto& abstractLayer = mesh.getLayer( layerKey );
449
450 if ( !abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) &&
451 !abstractLayer.hasSemantic( QuadIndexLayer::staticSemanticName ) &&
452 !abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
453 LOG( logWARNING ) << "TopologicalMesh: mesh does not contains faces. Aborting conversion";
454 return;
455 }
456
457 clean();
458
459 LOG( logDEBUG ) << "TopologicalMesh: load mesh with " << abstractLayer.getSize()
460 << " faces and " << mesh.vertices().size() << " vertices.";
461 // use a hashmap for fast search of existing vertex position
462 using VertexMap = std::unordered_map<Vector3, TopologicalMesh::VertexHandle, hash_vec>;
463 VertexMap vertexHandles;
464
465 // loop over all attribs and build correspondance pair
466 mesh.vertexAttribs().for_each_attrib( InitWedgeAttribsFromMultiIndexedGeometry { this, mesh } );
467
468 for ( size_t i = 0; i < mesh.vertices().size(); ++i ) {
469 // create an empty wedge, with 0 ref
470 Wedge w;
471
472 WedgeData wd;
473 wd.m_position = mesh.vertices()[i];
474 copyMeshToWedgeData( mesh,
475 i,
476 m_wedges.m_wedgeFloatAttribHandles,
477 m_wedges.m_wedgeVector2AttribHandles,
478 m_wedges.m_wedgeVector3AttribHandles,
479 m_wedges.m_wedgeVector4AttribHandles,
480 &wd );
481 // here ref is not incremented
482 w.setWedgeData( std::move( wd ) );
483 // the newly added wedge is not referenced yet, will be done with `newReference` when
484 // creating faces just below
485 m_wedges.m_data.push_back( w );
486 }
487
488 LOG( logDEBUG ) << "TopologicalMesh: have " << m_wedges.size() << " wedges ";
489
490 const bool hasNormals = !mesh.normals().empty();
491 if ( !hasNormals ) {
492 release_face_normals();
493 release_vertex_normals();
494 release_halfedge_normals();
495 }
496
497 command.initialize( mesh );
498
499 auto processFaces = [&mesh, &vertexHandles, this, hasNormals, &command]( const auto& faces ) {
500 size_t num_triangles = faces.size();
501 for ( unsigned int i = 0; i < num_triangles; i++ ) {
502 const auto& face = faces[i];
503 const size_t num_vert = face.size();
504 std::vector<TopologicalMesh::VertexHandle> face_vhandles( num_vert );
505 std::vector<TopologicalMesh::Normal> face_normals( num_vert );
506 std::vector<WedgeIndex> face_wedges( num_vert );
507
508 for ( size_t j = 0; j < num_vert; ++j ) {
509 unsigned int inMeshVertexIndex = face[j];
510 const Vector3& p = mesh.vertices()[inMeshVertexIndex];
511
512 typename VertexMap::iterator vtr = vertexHandles.find( p );
513 TopologicalMesh::VertexHandle vh;
514 if ( vtr == vertexHandles.end() ) {
515 vh = add_vertex( p );
516 vertexHandles.insert( vtr, typename VertexMap::value_type( p, vh ) );
517 }
518 else {
519 vh = vtr->second;
520 }
521
522 face_vhandles[j] = vh;
523 if ( hasNormals ) face_normals[j] = mesh.normals()[inMeshVertexIndex];
524 face_wedges[j] = WedgeIndex { inMeshVertexIndex };
525 m_wedges.m_data[inMeshVertexIndex].getWedgeData().m_vertexHandle = vh;
526 }
527
528 // remove consecutive equal vertex
529 // first take care of "loop" if begin == *end-1
530 // apply the same modifications on wedges and normals
531 // e.g. 1 2 1 becomes 1 2
532 {
533 auto begin = face_vhandles.begin();
534 if ( face_vhandles.size() > 2 ) {
535 auto end = face_vhandles.end() - 1;
536 auto wedgeEnd = face_wedges.end() - 1;
537 auto normalEnd = face_normals.end() - 1;
538
539 while ( begin != end && *begin == *end ) {
540 end--;
541 wedgeEnd--;
542 normalEnd--;
543 }
544 face_vhandles.erase( end + 1, face_vhandles.end() );
545 face_wedges.erase( wedgeEnd + 1, face_wedges.end() );
546 face_normals.erase( normalEnd + 1, face_normals.end() );
547 }
548 }
549 // then remove duplicates
550 // e.g. 1 2 2 becomes 1 2
551 // equiv of
552 // face_vhandles.erase( std::unique( face_vhandles.begin(), face_vhandles.end() ),
553 // face_vhandles.end() );
554 // but handles wedges and normals
555 // see (https://en.cppreference.com/w/cpp/algorithm/unique)
556 {
557 auto first = face_vhandles.begin();
558 auto wedgeFirst = face_wedges.begin();
559 auto normalFirst = face_normals.begin();
560 auto last = face_vhandles.end();
561
562 if ( first != last ) {
563 auto result = first;
564 auto wedgeResult = wedgeFirst;
565 auto normalResult = normalFirst;
566 while ( ++first != last ) {
567 if ( !( *result == *first ) ) {
568 ++result;
569 ++wedgeResult;
570 ++normalResult;
571 if ( result != first ) {
572 *result = std::move( *first );
573 *wedgeResult = std::move( *wedgeFirst );
574 *normalResult = std::move( *normalFirst );
575 }
576 }
577 }
578 face_vhandles.erase( result + 1, face_vhandles.end() );
579 face_wedges.erase( wedgeResult + 1, face_wedges.end() );
580 face_normals.erase( normalResult + 1, face_normals.end() );
581 }
582 }
583
585 // unique sort size == vhandles size, if not split ...
586
587 TopologicalMesh::FaceHandle fh;
588 // skip 2 vertex face
589 if ( face_vhandles.size() > 2 ) fh = add_face( face_vhandles );
590
591 // In case of topological inconsistancy, face will be invalid (or uninitialized <>
592 // invalid)
593 if ( fh.is_valid() ) {
594 for ( size_t vindex = 0; vindex < face_vhandles.size(); vindex++ ) {
595 TopologicalMesh::HalfedgeHandle heh =
596 halfedge_handle( face_vhandles[vindex], fh );
597 if ( hasNormals ) set_normal( heh, face_normals[vindex] );
598 property( m_wedgeIndexPph, heh ) = m_wedges.newReference( face_wedges[vindex] );
599 }
600 }
601 else {
602 command.process( face_vhandles );
603 }
604 face_vhandles.clear();
605 face_normals.clear();
606 }
607 };
608
609 if ( abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) ) {
610 const auto& faces = static_cast<const TriangleIndexLayer&>( abstractLayer ).collection();
611 LOG( logDEBUG ) << "TopologicalMesh: process " << faces.size() << " triangular faces ";
612 processFaces( faces );
613 }
614 else if ( abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
615 const auto& faces = static_cast<const PolyIndexLayer&>( abstractLayer ).collection();
616 LOG( logDEBUG ) << "TopologicalMesh: process " << faces.size() << " polygonal faces ";
617 processFaces( faces );
618 }
619
620 command.postProcess( *this );
621 if ( hasNormals ) {
622 m_normalsIndex =
623 m_wedges.getWedgeAttribIndex<Normal>( getAttribName( MeshAttrib::VERTEX_NORMAL ) );
624
625 m_vertexFaceWedgesWithSameNormals.clear();
626 m_vertexFaceWedgesWithSameNormals.resize( n_vertices() );
627
628 for ( auto itr = vertices_begin(), stop = vertices_end(); itr != stop; ++itr ) {
629 std::unordered_map<TopologicalMesh::Normal,
630 std::pair<std::set<FaceHandle>, std::set<WedgeIndex>>,
631 hash_vec>
632 normalSharedByWedges;
633
634 auto vh = *itr;
635
636 for ( ConstVertexIHalfedgeIter vh_it = cvih_iter( vh ); vh_it.is_valid(); ++vh_it ) {
637 const auto& widx = property( m_wedgeIndexPph, *vh_it );
638 if ( widx.isValid() && !m_wedges.getWedge( widx ).isDeleted() ) {
639 auto oldNormal = m_wedges.getWedgeData<Normal>( widx, m_normalsIndex );
640 normalSharedByWedges[oldNormal].first.insert( face_handle( *vh_it ) );
641 normalSharedByWedges[oldNormal].second.insert( widx );
642 }
643 }
644
645 for ( const auto& pair : normalSharedByWedges ) {
646 for ( const auto& fh : pair.second.first ) {
647 auto& v = m_vertexFaceWedgesWithSameNormals[vh.idx()][fh.idx()];
648 v.insert( v.end(), pair.second.second.begin(), pair.second.second.end() );
649 }
650 }
651 }
652 }
653 LOG( logDEBUG ) << "TopologicalMesh: load end with " << m_wedges.size() << " wedges ";
654}
