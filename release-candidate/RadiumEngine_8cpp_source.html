169void RadiumEngine::getTasks( Core::TaskQueue* taskQueue, Scalar dt ) {
170 static uint frameCounter = 0;
171
172 if ( m_timeData.m_play || m_timeData.m_singleStep ) {
173 m_timeData.updateTime( dt );
174 m_timeData.m_singleStep = false;
175 }
176
177 FrameInfo frameInfo {
178 m_timeData.m_time, m_timeData.m_realTime ? dt : m_timeData.m_dt, frameCounter++ };
179 for ( auto& syst : m_systems ) {
180 syst.second->generateTasks( taskQueue, frameInfo );
181 }

205Data::Displayable* RadiumEngine::getMesh( const std::string& entityName,
206 const std::string& componentName,
207 const std::string& roName ) const {
208
209 // 1) Get entity
210 if ( m_entityManager->entityExists( entityName ) ) {
211 auto e = m_entityManager->getEntity( entityName );
212
213 // 2) Get component
214 const auto c = e->getComponent( componentName );
215
216 if ( c != nullptr && !c->m_renderObjects.empty() ) {
217 // 3) Get RO
218 if ( roName.empty() ) {
219 return m_renderObjectManager->getRenderObject( c->m_renderObjects.front() )
220 ->getMesh()
221 .get();
222 }
223 else {
224 for ( const auto& idx : c->m_renderObjects ) {
225 const auto& ro = m_renderObjectManager->getRenderObject( idx );
226 if ( ro->getName() == roName ) { return ro->getMesh().get(); }
227 }
228 }
229 }
230 }
231 return nullptr;
232}

234bool RadiumEngine::loadFile( const std::string& filename ) {
235 releaseFile();
236
237 std::string extension = Core::Utils::getFileExt( filename );
238
239 for ( auto& l : m_fileLoaders ) {
240 if ( l->handleFileExtension( extension ) ) {
241 FileData* data = l->loadFile( filename );
242 if ( data != nullptr ) {
243 m_loadedFile.reset( data );
244 break;
245 }
246 }
247 }
248
249 if ( m_loadedFile == nullptr ) {
250 LOG( logERROR ) << "There is no loader to handle \"" << extension
251 << "\" extension ! File can't be loaded.";
252
253 return false;
254 }
255
256 std::string entityName = Core::Utils::getBaseName( filename, false );
257
258 Scene::Entity* entity = m_entityManager->createEntity( entityName );
259
260 for ( auto& system : m_systems ) {
261 system.second->handleAssetLoading( entity, m_loadedFile.get() );
262 }
263
264 if ( !entity->getComponents().empty() ) {
265 for ( auto& comp : entity->getComponents() ) {
266 comp->initialize();
267 }
268 }
269 else {
270 LOG( logWARNING ) << "File \"" << filename << "\" has no usable data. Deleting entity...";
271 m_entityManager->removeEntity( entity );
272 }
273 m_loadingState = true;
274 return true;
275}

342void RadiumEngine::pushFboAndViewport() {
343 int activeFbo;
344 std::array<int, 4> activeViewport;
345
346 glGetIntegerv( GL_VIEWPORT, activeViewport.data() );
347 // save the currently bound FBO
348 GL_ASSERT( glGetIntegerv( GL_FRAMEBUFFER_BINDING, &activeFbo ) );
349 // Set the internal rendering viewport
350 m_fboAndViewportStack.emplace( activeFbo, std::move( activeViewport ) );

353void RadiumEngine::popFboAndViewport() {
354 if ( m_fboAndViewportStack.empty() ) {
355 LOG( logERROR ) << "RadiumEngine: try to pop from an empty Fbo and Viewport stack\n";
356 }
357 else {
358 auto b = m_fboAndViewportStack.top();
359
360 glViewport( b.m_viewport[0], b.m_viewport[1], b.m_viewport[2], b.m_viewport[3] );
361 GL_ASSERT( glBindFramebuffer( GL_FRAMEBUFFER, b.m_fbo ) );
362 m_fboAndViewportStack.pop();
363 }

432void RadiumEngine::TimeData::updateTime( Scalar dt ) {
433 dt += m_realTime ? dt : m_dt;
434 // update the time w.r.t. the time flow policy
435 if ( m_forwardBackward && m_isBackward ) { m_time -= dt; }
436 else {
437 m_time += dt;
438 }
439 // special case: empty time window => forever mode
440 if ( m_endTime < 0 || m_startTime >= m_endTime ) {
441 // just run forever
442 m_isBackward = false;
443 return;
444 }
445 // special case: m_time before time window
446 if ( m_time < m_startTime ) {
447 // reset whatever the mode
448 m_time = m_startTime;
449 // and go forward from now on
450 m_isBackward = false;
451 return;
452 }
453 // special case: m_time after time window in loop mode
454 if ( !m_forwardBackward && m_time > m_endTime ) {
455 // compute the overload of time
456 dt = Scalar( fmod( double( m_time - m_startTime ), double( m_endTime - m_startTime ) ) );
457 // loop around, applying the overload of time
458 m_time = m_startTime + dt;
459 return;
460 }
461 // special case: m_time after time window in forward-backward mode
462 if ( m_forwardBackward && m_time > m_endTime ) {
463 if ( !m_isBackward ) // start backwards
464 {
465 m_time = 2 * m_endTime - m_time;
466 }
467 else // restart backward from the end of the time window
468 {
469 m_time = m_endTime;
470 }
471 m_isBackward = true;
472 return;
473 }
