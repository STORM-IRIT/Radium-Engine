276 void Gui::FlightCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
277  Scalar dphi = dx * m_cameraSensitivity * m_quickCameraModifier;
278  Scalar dtheta = -dy * m_cameraSensitivity * m_quickCameraModifier;
279 
280  Core::Transform R( Core::Transform::Identity() );
281  if ( std::abs( dphi ) > std::abs( dtheta ) ) {
282  R = Core::AngleAxis( -dphi, /*m_camera->getUpVector().normalized()*/ m_fixUpVector );
283  }
284  else {
285  R = Core::AngleAxis( -dtheta, -m_camera->getRightVector().normalized() );
286  }
287 
288  Scalar d = ( m_target - m_camera->getPosition() ).norm();
289 
290  m_camera->applyTransform( R );
291  m_target = m_camera->getPosition() + d * m_camera->getDirection();
292 }

294 void Gui::FlightCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {

296  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier;
