212 void Viewer::startRendering( const Scalar dt ) {
213 
214  CORE_ASSERT( m_glInitialized.load(), "OpenGL needs to be initialized before rendering." );
215 
216  CORE_ASSERT( m_currentRenderer != nullptr, "No renderer found." );
217 
218  m_pickingManager->clear();
219  makeCurrent();
220 
221  // update znear/zfar to fit the scene ...
222  auto entityManager = Engine::RadiumEngine::getInstance()->getEntityManager();
223  if ( entityManager ) {
224 
225  // to fit scene only
226  // auto aabb = Ra::Engine::RadiumEngine::getInstance()->computeSceneAabb();
227  // to fit also debug and system entity aabb
228  Core::Aabb aabb {};
229  for ( const auto& entity : entityManager->getEntities() ) {
230  // entity aabb is in world space
231  aabb.extend( entity->computeAabb() );
232  }
233 
234  if ( !aabb.isEmpty() ) { m_camera->getCamera()->fitZRange( aabb ); }
235  else {
236  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
237  Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
238 
239  // scene is empty, reset to defaults bounds
240  m_camera->setCameraZNear( cameraManager->defaultCamera.getZNear() );
241  m_camera->setCameraZFar( cameraManager->defaultCamera.getZFar() );
242  }
243  }
244 
247  if ( !m_currentRenderer->hasLight() ) {
248  if ( m_camera->hasLightAttached() )
249  m_currentRenderer->addLight( m_camera->getLight() );
250  else
251  LOG( logDEBUG ) << "Unable to attach the head light!";
252  }
253  Engine::Data::ViewingParameters data {
254  m_camera->getCamera()->getViewMatrix(), m_camera->getCamera()->getProjMatrix(), dt };
255  m_currentRenderer->render( data );
256 }

456 bool Viewer::initializeGL() {
457  globjects::init( getProcAddress );
458  // mark openGL as initialized
459  m_glInitialized = true;
460 
461  LOG( logINFO ) << "*** Radium Engine OpenGL context ***";
462  LOG( logINFO ) << "Renderer (glbinding) : " << glbinding::aux::ContextInfo::renderer();
463  LOG( logINFO ) << "Vendor (glbinding) : " << glbinding::aux::ContextInfo::vendor();
464  LOG( logINFO ) << "OpenGL (glbinding) : "
465  << glbinding::aux::ContextInfo::version().toString();
466  LOG( logINFO ) << "GLSL : "
467  << gl::glGetString( gl::GLenum( GL_SHADING_LANGUAGE_VERSION ) );
468 
469  LOG( logINFO ) << "*** Radium Engine Viewer ***";
470 
471  // emit the signal so that the client will initialize the OpenGL part of the Engine
472  // and custom OpenGL properties
473  emit requestEngineOpenGLInitialization();
474 
475  // Configure the viewer services
476  // create default camera interface : trackball
477  m_camera = std::make_unique<TrackballCameraManipulator>();
478  m_camera->getCamera()->setViewport( width(), height() );
479  auto headlight = new Engine::Scene::DirectionalLight(
480  Ra::Engine::Scene::SystemEntity::getInstance(), "headlight" );
481  headlight->setColor( Ra::Core::Utils::Color::Grey( 1.0_ra ) );
482 
483  m_camera->attachLight( headlight );
484 
485  // Register to the camera manager active camera changes
486  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
487  Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
488  cameraManager->activeCameraObservers().attach( [this]( Core::Utils::Index /*idx*/ ) {
489  m_camera->updateCamera();
490  m_camera->getCamera()->setViewport( width(), height() );
491  } );
492 
493  // Initialize renderers added to the viewer before initializeGL
494  for ( auto& rptr : m_pendingRenderers ) {
495  initializeRenderer( rptr.get() );
496  }
497  m_pendingRenderers.clear();
498 
499  // create the gizmo manager (Ui)
500  createGizmoManager();
501 
502  // Signal that OpenGL is initialized
503  emit glInitialized();
504 
505  // If no renderer was added before that (either by slots on requestEngineOpenGLInitialization
506  // or on glInitialized), add default forward renderer
507  if ( m_renderers.empty() ) {
508  LOG( logINFO ) << "[Viewer] No renderer added, adding default (Forward Renderer)";
509  std::shared_ptr<Ra::Engine::Rendering::Renderer> e(
510  new Ra::Engine::Rendering::ForwardRenderer() );
511  addRenderer( e );
512  }
513 
514  if ( m_currentRenderer == nullptr ) { changeRenderer( 0 ); }
515 
516  return m_glInitialized;
517 }
