664 inline bool TopologicalMesh::WedgeData::operator<( const TopologicalMesh::WedgeData& lhs ) const {
665 
666  CORE_ASSERT( ( m_floatAttrib.size() == lhs.m_floatAttrib.size() ) &&
667  ( m_vector2Attrib.size() == lhs.m_vector2Attrib.size() ) &&
668  ( m_vector3Attrib.size() == lhs.m_vector3Attrib.size() ) &&
669  ( m_vector4Attrib.size() == lhs.m_vector4Attrib.size() ),
670  "Could only compare wedge with same number of attributes" );
671 
672  {
673  int comp = compareVector( m_position, lhs.m_position );
674  if ( comp == 2 ) return true;
675  if ( comp == 3 ) return false;
676  }
677  for ( size_t i = 0; i < m_floatAttrib.size(); i++ ) {
678  if ( m_floatAttrib[i] < lhs.m_floatAttrib[i] )
679  return true;
680  else if ( m_floatAttrib[i] > lhs.m_floatAttrib[i] )
681  return false;
682  }
683 
684  for ( size_t i = 0; i < m_vector2Attrib.size(); i++ ) {
685  int comp = compareVector( m_vector2Attrib[i], lhs.m_vector2Attrib[i] );
686  if ( comp == 2 ) return true;
687  if ( comp == 3 ) return false;
688  }
689  for ( size_t i = 0; i < m_vector3Attrib.size(); i++ ) {
690  int comp = compareVector( m_vector3Attrib[i], lhs.m_vector3Attrib[i] );
691  if ( comp == 2 ) return true;
692  if ( comp == 3 ) return false;
693  }
694  for ( size_t i = 0; i < m_vector4Attrib.size(); i++ ) {
695  int comp = compareVector( m_vector4Attrib[i], lhs.m_vector4Attrib[i] );
696  if ( comp == 2 ) return true;
697  if ( comp == 3 ) return false;
698  }
699  return false;
700 }

1031 struct hash_vec {
1032  std::size_t operator()( const Vector3& lvalue ) const {
1033  size_t hx = std::hash<Scalar>()( lvalue[0] );
1034  size_t hy = std::hash<Scalar>()( lvalue[1] );

1073 template <typename NonManifoldFaceCommand>
1074 void TopologicalMesh::initWithWedge(
1075  const Ra::Core::Geometry::MultiIndexedGeometry& mesh,
1076  const Ra::Core::Geometry::MultiIndexedGeometry::LayerKeyType& layerKey,
1077  NonManifoldFaceCommand command ) {
1078 
1079  const auto& abstractLayer = mesh.getLayer( layerKey );
1080 
1081  if ( !abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) &&
1082  !abstractLayer.hasSemantic( QuadIndexLayer::staticSemanticName ) &&
1083  !abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
1084  LOG( logWARNING ) << "TopologicalMesh: mesh does not contains faces. Aborting conversion";
1085  return;
1086  }
1087 
1088  clean();
1089 
1090  LOG( logDEBUG ) << "TopologicalMesh: load mesh with " << abstractLayer.getSize()
1091  << " faces and " << mesh.vertices().size() << " vertices.";
1092  // use a hashmap for fast search of existing vertex position
1093  using VertexMap = std::unordered_map<Vector3, TopologicalMesh::VertexHandle, hash_vec>;
1094  VertexMap vertexHandles;
1095 
1096  // loop over all attribs and build correspondance pair
1097  mesh.vertexAttribs().for_each_attrib( InitWedgeAttribsFromMultiIndexedGeometry { this, mesh } );
1098 
1099  for ( size_t i = 0; i < mesh.vertices().size(); ++i ) {
1100  // create an empty wedge, with 0 ref
1101  Wedge w;
1102 
1103  WedgeData wd;
1104  wd.m_position = mesh.vertices()[i];
1105  copyMeshToWedgeData( mesh,
1106  i,
1107  m_wedges.m_wedgeFloatAttribHandles,
1108  m_wedges.m_wedgeVector2AttribHandles,
1109  m_wedges.m_wedgeVector3AttribHandles,
1110  m_wedges.m_wedgeVector4AttribHandles,
1111  &wd );
1112  // here ref is not incremented
1113  w.setWedgeData( std::move( wd ) );
1114  // the newly added wedge is not referenced yet, will be done with `newReference` when
1115  // creating faces just below
1116  m_wedges.m_data.push_back( w );
1117  }
1118 
1119  LOG( logDEBUG ) << "TopologicalMesh: have " << m_wedges.size() << " wedges ";
1120 
1121  const bool hasNormals = !mesh.normals().empty();
1122  if ( !hasNormals ) {
1123  release_face_normals();
1124  release_vertex_normals();
1125  release_halfedge_normals();
1126  }
1127 
1128  command.initialize( mesh );
1129 
1130  auto processFaces = [&mesh, &vertexHandles, this, hasNormals, &command]( const auto& faces ) {
1131  size_t num_triangles = faces.size();
1132  for ( unsigned int i = 0; i < num_triangles; i++ ) {
1133  const auto& face = faces[i];
1134  const size_t num_vert = face.size();
1135  std::vector<TopologicalMesh::VertexHandle> face_vhandles( num_vert );
1136  std::vector<TopologicalMesh::Normal> face_normals( num_vert );
1137  std::vector<WedgeIndex> face_wedges( num_vert );
1138 
1139  for ( size_t j = 0; j < num_vert; ++j ) {
1140  unsigned int inMeshVertexIndex = face[j];
1141  const Vector3& p = mesh.vertices()[inMeshVertexIndex];
1142 
1143  typename VertexMap::iterator vtr = vertexHandles.find( p );
1144  TopologicalMesh::VertexHandle vh;
1145  if ( vtr == vertexHandles.end() ) {
1146  vh = add_vertex( p );
1147  vertexHandles.insert( vtr, typename VertexMap::value_type( p, vh ) );
1148  }
1149  else {
1150  vh = vtr->second;
1151  }
1152 
1153  face_vhandles[j] = vh;
1154  if ( hasNormals ) face_normals[j] = mesh.normals()[inMeshVertexIndex];
1155  face_wedges[j] = WedgeIndex { inMeshVertexIndex };
1156  m_wedges.m_data[inMeshVertexIndex].getWedgeData().m_vertexHandle = vh;
1157  }
1158 
1159  // remove consecutive equal vertex
1160  // first take care of "loop" if begin == *end-1
1161  // apply the same modifications on wedges and normals
1162  // e.g. 1 2 1 becomes 1 2
1163  {
1164  auto begin = face_vhandles.begin();
1165  if ( face_vhandles.size() > 2 ) {
1166  auto end = face_vhandles.end() - 1;
1167  auto wedgeEnd = face_wedges.end() - 1;
1168  auto normalEnd = face_normals.end() - 1;
1169 
1170  while ( begin != end && *begin == *end ) {
1171  end--;
1172  wedgeEnd--;
1173  normalEnd--;
1174  }
1175  face_vhandles.erase( end + 1, face_vhandles.end() );
1176  face_wedges.erase( wedgeEnd + 1, face_wedges.end() );
1177  face_normals.erase( normalEnd + 1, face_normals.end() );
1178  }
1179  }
1180  // then remove duplicates
1181  // e.g. 1 2 2 becomes 1 2
1182  // equiv of
1183  // face_vhandles.erase( std::unique( face_vhandles.begin(), face_vhandles.end() ),
1184  // face_vhandles.end() );
1185  // but handles wedges and normals
1186  // see (https://en.cppreference.com/w/cpp/algorithm/unique)
1187  {
1188  auto first = face_vhandles.begin();
1189  auto wedgeFirst = face_wedges.begin();
1190  auto normalFirst = face_normals.begin();
1191  auto last = face_vhandles.end();
1192 
1193  if ( first != last ) {
1194  auto result = first;
1195  auto wedgeResult = wedgeFirst;
1196  auto normalResult = normalFirst;
1197  while ( ++first != last ) {
1198  if ( !( *result == *first ) ) {
1199  ++result;
1200  ++wedgeResult;
1201  ++normalResult;
1202  if ( result != first ) {
1203  *result = std::move( *first );
1204  *wedgeResult = std::move( *wedgeFirst );
1205  *normalResult = std::move( *normalFirst );
1206  }
1207  }
1208  }
1209  face_vhandles.erase( result + 1, face_vhandles.end() );
1210  face_wedges.erase( wedgeResult + 1, face_wedges.end() );
1211  face_normals.erase( normalResult + 1, face_normals.end() );
1212  }
1213  }
1214 
1216  // unique sort size == vhandles size, if not split ...
1217 
1218  TopologicalMesh::FaceHandle fh;
1219  // skip 2 vertex face
1220  if ( face_vhandles.size() > 2 ) fh = add_face( face_vhandles );
1221 
1222  // In case of topological inconsistancy, face will be invalid (or uninitialized <>
1223  // invalid)
1224  if ( fh.is_valid() ) {
1225  for ( size_t vindex = 0; vindex < face_vhandles.size(); vindex++ ) {
1226  TopologicalMesh::HalfedgeHandle heh =
1227  halfedge_handle( face_vhandles[vindex], fh );
1228  if ( hasNormals ) set_normal( heh, face_normals[vindex] );
1229  property( m_wedgeIndexPph, heh ) = m_wedges.newReference( face_wedges[vindex] );
1230  }
1231  }
1232  else {
1233  command.process( face_vhandles );
1234  }
1235  face_vhandles.clear();
1236  face_normals.clear();
1237  }
1238  };
1239 
1240  if ( abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) ) {
1241  const auto& faces = static_cast<const TriangleIndexLayer&>( abstractLayer ).collection();
1242  LOG( logDEBUG ) << "TopologicalMesh: process " << faces.size() << " triangular faces ";
1243  processFaces( faces );
1244  }
1245  else if ( abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
1246  const auto& faces = static_cast<const PolyIndexLayer&>( abstractLayer ).collection();
1247  LOG( logDEBUG ) << "TopologicalMesh: process " << faces.size() << " polygonal faces ";
1248  processFaces( faces );
1249  }
1250 
1251  command.postProcess( *this );
1252  if ( hasNormals ) {
1253  m_normalsIndex =
1254  m_wedges.getWedgeAttribIndex<Normal>( getAttribName( MeshAttrib::VERTEX_NORMAL ) );
1255 
1256  m_vertexFaceWedgesWithSameNormals.clear();
1257  m_vertexFaceWedgesWithSameNormals.resize( n_vertices() );
1258 
1259  for ( auto itr = vertices_begin(), stop = vertices_end(); itr != stop; ++itr ) {
1260  std::unordered_map<TopologicalMesh::Normal,
1261  std::pair<std::set<FaceHandle>, std::set<WedgeIndex>>,
1262  hash_vec>
1263  normalSharedByWedges;
1264 
1265  auto vh = *itr;
1266 
1267  for ( ConstVertexIHalfedgeIter vh_it = cvih_iter( vh ); vh_it.is_valid(); ++vh_it ) {
1268  const auto& widx = property( m_wedgeIndexPph, *vh_it );
1269  if ( widx.isValid() && !m_wedges.getWedge( widx ).isDeleted() ) {
1270  auto oldNormal = m_wedges.getWedgeData<Normal>( widx, m_normalsIndex );
1271  normalSharedByWedges[oldNormal].first.insert( face_handle( *vh_it ) );
1272  normalSharedByWedges[oldNormal].second.insert( widx );
1273  }
1274  }
1275 
1276  for ( const auto& pair : normalSharedByWedges ) {
1277  for ( const auto& fh : pair.second.first ) {
1278  auto& v = m_vertexFaceWedgesWithSameNormals[vh.idx()][fh.idx()];
1279  v.insert( v.end(), pair.second.second.begin(), pair.second.second.end() );
1280  }
1281  }
1282  }
1283  }
1284  LOG( logDEBUG ) << "TopologicalMesh: load end with " << m_wedges.size() << " wedges ";
1285 }
