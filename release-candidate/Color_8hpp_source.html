184  static inline ColorBase<_Scalar> fromHSV( const _Scalar hue,
185  const _Scalar saturation = 1.0,
186  const _Scalar value = 1.0,
187  const _Scalar alpha = 1.0 ) {
188  ColorBase<_Scalar> c;
189 
190  if ( saturation == 0.0f ) {
191  c[0] = c[1] = c[2] = value;
192  c[3] = alpha;
193  return c;
194  }
195  _Scalar h = ( ( hue == 1.0f ) ? 0.0f : hue ) * 6.0f;
196  int i = int( std::floor( h ) );
197  _Scalar v1 = value * ( 1.0f - saturation );
198  _Scalar v2 = value * ( 1.0f - ( saturation * ( h - i ) ) );
199  _Scalar v3 = value * ( 1.0f - ( saturation * ( 1.0f - h - i ) ) );
200  switch ( i ) {
201  case 0: {
202  c[0] = value;
203  c[1] = v3;
204  c[2] = v1;
205  } break;
206  case 1: {
207  c[0] = v2;
208  c[1] = value;
209  c[2] = v1;
210  } break;
211  case 2: {
212  c[0] = v1;
213  c[1] = value;
214  c[2] = v3;
215  } break;
216  case 3: {
217  c[0] = v1;
218  c[1] = v2;
219  c[2] = value;
220  } break;
221  case 4: {
222  c[0] = v3;
223  c[1] = v1;
224  c[2] = value;
225  } break;
226  default: {
227  c[0] = value;
228  c[1] = v1;
229  c[2] = v2;
230  } break;
231  }
232  c[3] = alpha;
233  return c;
234  }

250  static inline std::vector<ColorBase<_Scalar>> scatter( const uint size, const _Scalar gamma ) {
251  std::vector<ColorBase<_Scalar>> color( size );
252  if ( size > 1 )
253  for ( uint i = 0; i < size; ++i ) {
254  color[i] = fromHSV( ( _Scalar( i ) / _Scalar( size - 1 ) ) * 0.777 );
255  color[i] = ( color[i] + ColorBase<_Scalar>::Constant( gamma ) ) * 0.5;
256  }
257  else {
258  color[0] = Red();
259  }
260  std::shuffle( color.begin(), color.end(), std::mt19937( std::random_device()() ) );
