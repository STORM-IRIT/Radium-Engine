158  static inline ColorBase<_Scalar> fromHSV( const _Scalar hue,
159  const _Scalar saturation = 1.0,
160  const _Scalar value = 1.0,
161  const _Scalar alpha = 1.0 ) {
162  ColorBase<_Scalar> c;
163 
164  if ( saturation == 0.0f ) {
165  c[0] = c[1] = c[2] = value;
166  c[3] = alpha;
167  return c;
168  }
169  _Scalar h = ( ( hue == 1.0f ) ? 0.0f : hue ) * 6.0f;
170  int i = int( std::floor( h ) );
171  _Scalar v1 = value * ( 1.0f - saturation );
172  _Scalar v2 = value * ( 1.0f - ( saturation * ( h - i ) ) );
173  _Scalar v3 = value * ( 1.0f - ( saturation * ( 1.0f - h - i ) ) );
174  switch ( i ) {
175  case 0: {
176  c[0] = value;
177  c[1] = v3;
178  c[2] = v1;
179  } break;
180  case 1: {
181  c[0] = v2;
182  c[1] = value;
183  c[2] = v1;
184  } break;
185  case 2: {
186  c[0] = v1;
187  c[1] = value;
188  c[2] = v3;
189  } break;
190  case 3: {
191  c[0] = v1;
192  c[1] = v2;
193  c[2] = value;
194  } break;
195  case 4: {
196  c[0] = v3;
197  c[1] = v1;
198  c[2] = value;
199  } break;
200  default: {
201  c[0] = value;
202  c[1] = v1;
203  c[2] = v2;
204  } break;
