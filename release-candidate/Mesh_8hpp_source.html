464 template <typename CoreMeshType>
465 CoreMeshType createCoreMeshFromGeometryData( const Ra::Core::Asset::GeometryData* data ) {
466  CoreMeshType mesh;
467  typename CoreMeshType::PointAttribHandle::Container vertices;
468  typename CoreMeshType::NormalAttribHandle::Container normals;
469  typename CoreMeshType::IndexContainerType indices;
470 
471  vertices.reserve( data->getVerticesSize() );
472  std::copy(
473  data->getVertices().begin(), data->getVertices().end(), std::back_inserter( vertices ) );
474 
475  if ( data->hasNormals() ) {
476  normals.reserve( data->getVerticesSize() );
477  std::copy(
478  data->getNormals().begin(), data->getNormals().end(), std::back_inserter( normals ) );
479  }
480 
481  const auto& faces = data->getFaces();
482  indices.reserve( faces.size() );
483  std::copy( faces.begin(), faces.end(), std::back_inserter( indices ) );
484  mesh.setVertices( std::move( vertices ) );
485  mesh.setNormals( std::move( normals ) );
486 
487  // \todo remove when data will handle all the attributes in a coherent way.
488  if ( data->hasTangents() ) {
489  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_TANGENT ),
490  data->getTangents() );
491  }
492 
493  if ( data->hasBiTangents() ) {
494  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_BITANGENT ),
495  data->getBiTangents() );
496  }
497 
498  if ( data->hasTextureCoordinates() ) {
499  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_TEXCOORD ),
500  data->getTexCoords() );
501  }
502 
503  // add custom attribs
504  // only attributs not handled before are handled by data->getAttribManager()
505  // but futur plan will handle also "usual" attibutes this way
506  mesh.vertexAttribs().copyAllAttributes( data->getAttribManager() );
507 
508  // also this one will be automatically done with the futur behavior
509  // mesh->addData( Data::Mesh::VERTEX_WEIGHTS, meshData.weights );
510 
511  mesh.setIndices( std::move( indices ) );
512 
513  return mesh;
514 }
