35  Ra::Core::Quaternion& rotation,
36  const Ra::Core::Vector3& point ) {
37  Ra::Core::Vector3 t = cam->getPosition();
38  Scalar l = ( target - t ).norm();
39  Ra::Core::Transform inverseCamRotateAround;
40  Ra::Core::AngleAxis aa0 { rotation };
41  Ra::Core::AngleAxis aa1 { aa0.angle(), ( cam->getFrame().linear() * aa0.axis() ).normalized() };

47  cam->setPosition( trans );
48  target = cam->getPosition() + l * cam->getDirection();

112  Ra::Core::Vector3 x( 1_ra, 0_ra, 0_ra );
113  Ra::Core::Vector3 y( 0_ra, 1_ra, 0_ra );
114  Ra::Core::Vector3 z( 0_ra, 0_ra, 1_ra );

116  Ra::Core::Vector3 newUpVector = m_camera->getUpVector();

119 
120  auto updateMaxAndAxis =

122  Scalar d = ref.dot( axis );
123  if ( d > max ) {
124  max = d;

132  updateMaxAndAxis( ref, axis, newDirection, max );

135

169 Scalar RotateAroundCameraManipulator::projectOnBall( Scalar x, Scalar y ) {
170  const Scalar size = 1.0;
171  const Scalar size2 = size * size;
172  const Scalar size_limit = size2 * 0.5;
173 
174  const Scalar d = x * x + y * y;
175  return d < size_limit ? std::sqrt( size2 - d ) : size_limit / std::sqrt( d );
176 }

178 Ra::Core::Quaternion
179 RotateAroundCameraManipulator::deformedBallQuaternion( Scalar x, Scalar y, Scalar cx, Scalar cy ) {
180  // Points on the deformed ball
181  Scalar px = m_cameraSensitivity * ( m_lastMouseX - cx ) / m_camera->getWidth();
182  Scalar py = m_cameraSensitivity * ( cy - m_lastMouseY ) / m_camera->getHeight();
183  Scalar dx = m_cameraSensitivity * ( x - cx ) / m_camera->getWidth();
184  Scalar dy = m_cameraSensitivity * ( cy - y ) / m_camera->getHeight();
185 
186  const Ra::Core::Vector3 p1( px, py, projectOnBall( px, py ) );
187  const Ra::Core::Vector3 p2( dx, dy, projectOnBall( dx, dy ) );
188  // Approximation of rotation angle
189  // Should be divided by the projectOnBall size, but it is 1.0
190  Ra::Core::Vector3 axis = p2.cross( p1 );
191 
192  if ( axis.norm() > 10_ra * Ra::Core::Math::machineEps ) {
193  const Scalar angle =
194  5.0_ra *
195  std::asin( std::sqrt( axis.squaredNorm() / p1.squaredNorm() / p2.squaredNorm() ) );
196  return Ra::Core::Quaternion( Ra::Core::AngleAxis( angle, axis.normalized() ) );
197  }
198  return Ra::Core::Quaternion {
199  Ra::Core::AngleAxis( 0_ra, Ra::Core::Vector3( 0_ra, 0_ra, 1_ra ) ) };
200 }

210 void RotateAroundCameraManipulator::handleCameraPan( Scalar dx, Scalar dy ) {
211  Scalar x = dx * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
212  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;
