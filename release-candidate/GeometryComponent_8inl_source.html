63  if ( data != nullptr ) {

67  }
68  else {

70  mat->m_renderAsSplat = m_displayMesh->getNumFaces() == 0;

72  Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_COLOR ) );
73  roMaterial.reset( mat );
74  }
75  // initialize with a default rendertechique that draws nothing

77  auto ro = Rendering::RenderObject::createRenderObject( roName,
78  this,
79  Rendering::RenderObjectType::Geometry,

81  Rendering::RenderTechnique {} );
82  ro->setTransparent( roMaterial->isTransparent() );
