13 template <typename NonManifoldFaceCommand>
14 inline TopologicalMesh::TopologicalMesh( const TriangleMesh& triMesh,
15  NonManifoldFaceCommand command ) :
16  TopologicalMesh() {
17 
18  LOG( logINFO ) << "TopologicalMesh: load triMesh with " << triMesh.getIndices().size()
19  << " faces and " << triMesh.vertices().size() << " vertices.";
20 
21  struct hash_vec {
22  size_t operator()( const Vector3& lvalue ) const {
23  size_t hx = std::hash<Scalar>()( lvalue[0] );
24  size_t hy = std::hash<Scalar>()( lvalue[1] );
25  size_t hz = std::hash<Scalar>()( lvalue[2] );
26  return ( hx ^ ( hy << 1 ) ) ^ hz;
27  }
28  };
29  // use a hashmap for fast search of existing vertex position
30  using VertexMap = std::unordered_map<Vector3, TopologicalMesh::VertexHandle, hash_vec>;
31  VertexMap vertexHandles;
32 
33  std::vector<PropPair<Scalar>> vprop_float;
34  std::vector<std::pair<AttribHandle<Vector2>, OpenMesh::HPropHandleT<Vector2>>> vprop_vec2;
35  std::vector<std::pair<AttribHandle<Vector3>, OpenMesh::HPropHandleT<Vector3>>> vprop_vec3;
36  std::vector<std::pair<AttribHandle<Vector4>, OpenMesh::HPropHandleT<Vector4>>> vprop_vec4;
37 
38  // loop over all attribs and build correspondance pair
39  triMesh.vertexAttribs().for_each_attrib(
40  [&triMesh, this, &vprop_float, &vprop_vec2, &vprop_vec3, &vprop_vec4]( const auto& attr ) {
41  // skip builtin attribs
42  if ( attr->getName() != std::string( getAttribName( MeshAttrib::VERTEX_POSITION ) ) &&
43  attr->getName() != std::string( getAttribName( MeshAttrib::VERTEX_NORMAL ) ) ) {
44  if ( attr->isFloat() )
45  addAttribPairToTopo( triMesh, attr, vprop_float, m_floatPph );
46  else if ( attr->isVector2() )
47  addAttribPairToTopo( triMesh, attr, vprop_vec2, m_vec2Pph );
48  else if ( attr->isVector3() )
49  addAttribPairToTopo( triMesh, attr, vprop_vec3, m_vec3Pph );
50  else if ( attr->isVector4() )
51  addAttribPairToTopo( triMesh, attr, vprop_vec4, m_vec4Pph );
52  else
53  LOG( logWARNING )
54  << "Warning, mesh attribute " << attr->getName()
55  << " type is not supported (only float, vec2, vec3 nor vec4 are supported)";
56  }
57  } );
58 
59  size_t num_triangles = triMesh.getIndices().size();
60 
61  command.initialize( triMesh );
62 
63  const bool hasNormals = !triMesh.normals().empty();
64  if ( !hasNormals ) {
65  release_face_normals();
66  release_vertex_normals();
67  release_halfedge_normals();
68  }
69  for ( unsigned int i = 0; i < num_triangles; i++ ) {
70  std::vector<TopologicalMesh::VertexHandle> face_vhandles( 3 );
71  std::vector<TopologicalMesh::Normal> face_normals( 3 );
72  std::vector<unsigned int> face_vertexIndex( 3 );
73  const auto& triangle = triMesh.getIndices()[i];
74  for ( size_t j = 0; j < 3; ++j ) {
75  unsigned int inMeshVertexIndex = triangle[j];
76  const Vector3& p = triMesh.vertices()[inMeshVertexIndex];
77 
78  typename VertexMap::iterator vtr = vertexHandles.find( p );
79 
80  TopologicalMesh::VertexHandle vh;
81  if ( vtr == vertexHandles.end() ) {
82  vh = add_vertex( p );
83  vertexHandles.insert( vtr, typename VertexMap::value_type( p, vh ) );
84  }
85  else {
86  vh = vtr->second;
87  }
88 
89  face_vhandles[j] = vh;
90  face_vertexIndex[j] = inMeshVertexIndex;
91  if ( hasNormals ) face_normals[j] = triMesh.normals()[inMeshVertexIndex];
92  }
93 
94  TopologicalMesh::FaceHandle fh;
95 
96  // skip 2 vertex face
97  if ( face_vhandles.size() > 2 ) fh = add_face( face_vhandles );
98  // x-----------------------------------------------------------------------------------x
99 
100  if ( fh.is_valid() ) {
101  for ( size_t vindex = 0; vindex < face_vhandles.size(); vindex++ ) {
102  TopologicalMesh::HalfedgeHandle heh = halfedge_handle( face_vhandles[vindex], fh );
103  if ( hasNormals ) set_normal( heh, face_normals[vindex] );
104  property( m_inputTriangleMeshIndexPph, heh ) = face_vertexIndex[vindex];
105  copyAttribToTopo( triMesh, vprop_float, heh, face_vertexIndex[vindex] );
106  copyAttribToTopo( triMesh, vprop_vec2, heh, face_vertexIndex[vindex] );
107  copyAttribToTopo( triMesh, vprop_vec3, heh, face_vertexIndex[vindex] );
108  copyAttribToTopo( triMesh, vprop_vec4, heh, face_vertexIndex[vindex] );
109  }
110  }
111  else {
112  command.process( face_vhandles );
113  }
114  face_vhandles.clear();
115  face_normals.clear();
116  face_vertexIndex.clear();
117  }
118  command.postProcess( *this );
119 
120  // grabage collect since some wedge might already be deleted
121  garbage_collection();
122 }

263 inline void TopologicalMesh::interpolateNormalOnFaces( FaceHandle fh,
264  OpenMesh::FPropHandleT<Normal> fProp ) {
265  if ( !has_halfedge_normals() ) {
266  LOG( logERROR ) << "TopologicalMesh has no normals, nothing set";
267  return;
268  }
269  auto nph = halfedge_normals_pph();
270 
271  // init sum to first
272  auto heh = halfedge_handle( fh );
273  property( fProp, fh ) = property( nph, heh );
274  heh = next_halfedge_handle( heh );
275 
276  // sum others
277  for ( size_t i = 1; i < valence( fh ); ++i ) {
278  property( fProp, fh ) += property( nph, heh );
279  heh = next_halfedge_handle( heh );
280  }
281 
282  // normalize
283  property( fProp, fh ) = property( fProp, fh ).normalized();

338 template <typename T>
339 void TopologicalMesh::interpolatePropsOnFaces(
340  FaceHandle fh,
341  const std::vector<OpenMesh::HPropHandleT<T>>& hProps,
342  const std::vector<OpenMesh::FPropHandleT<T>>& fProps ) {
343  auto heh = halfedge_handle( fh );
344  const size_t v = valence( fh );
345 
346  // init sum to first
347  for ( size_t j = 0; j < fProps.size(); ++j ) {
348  auto hp = hProps[j];
349  auto fp = fProps[j];
350  property( fp, fh ) = property( hp, heh );
351  }
352  heh = next_halfedge_handle( heh );
353  // sum others
354  for ( size_t i = 1; i < v; ++i ) {
355  for ( size_t j = 0; j < fProps.size(); ++j ) {
356  auto hp = hProps[j];
357  auto fp = fProps[j];
358  property( fp, fh ) += property( hp, heh );
359  }
360  heh = next_halfedge_handle( heh );
361  }
362  // normalize
363  for ( size_t j = 0; j < fProps.size(); ++j ) {
364  auto fp = fProps[j];
365  property( fp, fh ) = property( fp, fh ) / v;
366  }
367 }
