26 inline std::string convert_arg_for_ini( const std::string& arg ) {
27  if ( arg.empty() ) { return std::string( 2, '"' ); }
28  // some specifically supported strings
29  if ( arg == "true" || arg == "false" || arg == "nan" || arg == "inf" ) { return arg; }
30  // floating point conversion can convert some hex codes, but don't try that here
31  if ( arg.compare( 0, 2, "0x" ) != 0 && arg.compare( 0, 2, "0X" ) != 0 ) {
32  double val;
33  if ( detail::lexical_cast( arg, val ) ) { return arg; }
34  }
35  // just quote a single non numeric character
36  if ( arg.size() == 1 ) { return std::string( "'" ) + arg + '\''; }
37  // handle hex, binary or octal arguments
38  if ( arg.front() == '0' ) {
39  if ( arg[1] == 'x' ) {
40  if ( std::all_of( arg.begin() + 2, arg.end(), []( char x ) {
41  return ( x >= '0' && x <= '9' ) || ( x >= 'A' && x <= 'F' ) ||
42  ( x >= 'a' && x <= 'f' );
43  } ) ) {
44  return arg;
45  }
46  }
47  else if ( arg[1] == 'o' ) {
48  if ( std::all_of( arg.begin() + 2, arg.end(), []( char x ) {
49  return ( x >= '0' && x <= '7' );
50  } ) ) {
51  return arg;
52  }
53  }
54  else if ( arg[1] == 'b' ) {
55  if ( std::all_of( arg.begin() + 2, arg.end(), []( char x ) {
56  return ( x == '0' || x == '1' );
57  } ) ) {
58  return arg;
59  }
60  }
61  }
62  if ( arg.find_first_of( '"' ) == std::string::npos ) { return std::string( "\"" ) + arg + '"'; }
63  else {
64  return std::string( "'" ) + arg + '\'';
65  }
66 }

69 inline std::string ini_join( const std::vector<std::string>& args,
70  char sepChar = ',',
71  char arrayStart = '[',
72  char arrayEnd = ']' ) {
73  std::string joined;
74  if ( args.size() > 1 && arrayStart != '\0' ) { joined.push_back( arrayStart ); }
75  std::size_t start = 0;
76  for ( const auto& arg : args ) {
77  if ( start++ > 0 ) {
78  joined.push_back( sepChar );
79  if ( isspace( sepChar ) == 0 ) { joined.push_back( ' ' ); }
80  }
81  joined.append( convert_arg_for_ini( arg ) );
82  }
83  if ( args.size() > 1 && arrayEnd != '\0' ) { joined.push_back( arrayEnd ); }
84  return joined;
85 }

87 inline std::vector<std::string> generate_parents( const std::string& section, std::string& name ) {
88  std::vector<std::string> parents;
89  if ( detail::to_lower( section ) != "default" ) {
90  if ( section.find( '.' ) != std::string::npos ) { parents = detail::split( section, '.' ); }
91  else {
92  parents = { section };
93  }
94  }
95  if ( name.find( '.' ) != std::string::npos ) {
96  std::vector<std::string> plist = detail::split( name, '.' );
97  name = plist.back();
98  detail::remove_quotes( name );
99  plist.pop_back();
100  parents.insert( parents.end(), plist.begin(), plist.end() );
101  }
102 
103  // clean up quotes on the parents
104  for ( auto& parent : parents ) {
105  detail::remove_quotes( parent );
106  }
107  return parents;
108 }

112 inline void checkParentSegments( std::vector<ConfigItem>& output,
113  const std::string& currentSection ) {
114 
115  std::string estring;
116  auto parents = detail::generate_parents( currentSection, estring );
117  if ( !output.empty() && output.back().name == "--" ) {
118  std::size_t msize = ( parents.size() > 1U ) ? parents.size() : 2;
119  while ( output.back().parents.size() >= msize ) {
120  output.push_back( output.back() );
121  output.back().parents.pop_back();
122  }
123 
124  if ( parents.size() > 1 ) {
125  std::size_t common = 0;
126  std::size_t mpair = ( std::min )( output.back().parents.size(), parents.size() - 1 );
127  for ( std::size_t ii = 0; ii < mpair; ++ii ) {
128  if ( output.back().parents[ii] != parents[ii] ) { break; }
129  ++common;
130  }
131  if ( common == mpair ) { output.pop_back(); }
132  else {
133  while ( output.back().parents.size() > common + 1 ) {
134  output.push_back( output.back() );
135  output.back().parents.pop_back();
136  }
137  }
138  for ( std::size_t ii = common; ii < parents.size() - 1; ++ii ) {
139  output.emplace_back();
140  output.back().parents.assign(
141  parents.begin(), parents.begin() + static_cast<std::ptrdiff_t>( ii ) + 1 );
142  output.back().name = "++";
143  }
144  }
145  }
146  else if ( parents.size() > 1 ) {
147  for ( std::size_t ii = 0; ii < parents.size() - 1; ++ii ) {
148  output.emplace_back();
149  output.back().parents.assign( parents.begin(),
150  parents.begin() + static_cast<std::ptrdiff_t>( ii ) + 1 );
151  output.back().name = "++";
152  }
153  }
154 
155  // insert a section end which is just an empty items_buffer
156  output.emplace_back();
157  output.back().parents = std::move( parents );
158  output.back().name = "++";
159 }

162 inline std::vector<ConfigItem> ConfigBase::from_config( std::istream& input ) const {
163  std::string line;
164  std::string section = "default";
165 
166  std::vector<ConfigItem> output;
167  bool isDefaultArray = ( arrayStart == '[' && arrayEnd == ']' && arraySeparator == ',' );
168  bool isINIArray = ( arrayStart == '\0' || arrayStart == ' ' ) && arrayStart == arrayEnd;
169  char aStart = ( isINIArray ) ? '[' : arrayStart;
170  char aEnd = ( isINIArray ) ? ']' : arrayEnd;
171  char aSep = ( isINIArray && arraySeparator == ' ' ) ? ',' : arraySeparator;
172 
173  while ( getline( input, line ) ) {
174  std::vector<std::string> items_buffer;
175  std::string name;
176 
177  detail::trim( line );
178  std::size_t len = line.length();
179  if ( len > 1 && line.front() == '[' && line.back() == ']' ) {
180  if ( section != "default" ) {
181  // insert a section end which is just an empty items_buffer
182  output.emplace_back();
183  output.back().parents = detail::generate_parents( section, name );
184  output.back().name = "--";
185  }
186  section = line.substr( 1, len - 2 );
187  // deal with double brackets for TOML
188  if ( section.size() > 1 && section.front() == '[' && section.back() == ']' ) {
189  section = section.substr( 1, section.size() - 2 );
190  }
191  if ( detail::to_lower( section ) == "default" ) { section = "default"; }
192  else {
193  detail::checkParentSegments( output, section );
194  }
195  continue;
196  }
197  if ( len == 0 ) { continue; }
198  // comment lines
199  if ( line.front() == ';' || line.front() == '#' || line.front() == commentChar ) {
200  continue;
201  }
202 
203  // Find = in string, split and recombine
204  auto pos = line.find( valueDelimiter );
205  if ( pos != std::string::npos ) {
206  name = detail::trim_copy( line.substr( 0, pos ) );
207  std::string item = detail::trim_copy( line.substr( pos + 1 ) );
208  if ( item.size() > 1 && item.front() == aStart ) {
209  for ( std::string multiline;
210  item.back() != aEnd && std::getline( input, multiline ); ) {
211  detail::trim( multiline );
212  item += multiline;
213  }
214  items_buffer = detail::split_up( item.substr( 1, item.length() - 2 ), aSep );
215  }
216  else if ( ( isDefaultArray || isINIArray ) &&
217  item.find_first_of( aSep ) != std::string::npos ) {
218  items_buffer = detail::split_up( item, aSep );
219  }
220  else if ( ( isDefaultArray || isINIArray ) &&
221  item.find_first_of( ' ' ) != std::string::npos ) {
222  items_buffer = detail::split_up( item );
223  }
224  else {
225  items_buffer = { item };
226  }
227  }
228  else {
229  name = detail::trim_copy( line );
230  items_buffer = { "true" };
231  }
232  if ( name.find( '.' ) == std::string::npos ) { detail::remove_quotes( name ); }
233  // clean up quotes on the items
234  for ( auto& it : items_buffer ) {
235  detail::remove_quotes( it );
236  }
237 
238  std::vector<std::string> parents = detail::generate_parents( section, name );
239 
240  if ( !output.empty() && name == output.back().name && parents == output.back().parents ) {
241  output.back().inputs.insert(
242  output.back().inputs.end(), items_buffer.begin(), items_buffer.end() );
243  }
244  else {
245  output.emplace_back();
246  output.back().parents = std::move( parents );
247  output.back().name = std::move( name );
248  output.back().inputs = std::move( items_buffer );
249  }
250  }
251  if ( section != "default" ) {
252  // insert a section end which is just an empty items_buffer
253  std::string ename;
254  output.emplace_back();
255  output.back().parents = detail::generate_parents( section, ename );
256  output.back().name = "--";
257  while ( output.back().parents.size() > 1 ) {
258  output.push_back( output.back() );
259  output.back().parents.pop_back();
260  }
261  }
262  return output;
263 }

265 inline std::string ConfigBase::to_config( const App* app,
266  bool default_also,
267  bool write_description,
268  std::string prefix ) const {
269  std::stringstream out;
270  std::string commentLead;
271  commentLead.push_back( commentChar );
272  commentLead.push_back( ' ' );
273 
274  std::vector<std::string> groups = app->get_groups();
275  bool defaultUsed = false;
276  groups.insert( groups.begin(), std::string( "Options" ) );
277  if ( write_description ) { out << commentLead << app->get_description() << '\n'; }
278  for ( auto& group : groups ) {
279  if ( group == "Options" || group.empty() ) {
280  if ( defaultUsed ) { continue; }
281  defaultUsed = true;
282  }
283  if ( write_description && group != "Options" && !group.empty() ) {
284  out << '\n' << commentLead << group << " Options\n";
285  }
286  for ( const Option* opt : app->get_options( {} ) ) {
287 
288  // Only process options that are configurable
289  if ( opt->get_configurable() ) {
290  if ( opt->get_group() != group ) {
291  if ( !( group == "Options" && opt->get_group().empty() ) ) { continue; }
292  }
293  std::string name = prefix + opt->get_single_name();
294  std::string value = detail::ini_join(
295  opt->reduced_results(), arraySeparator, arrayStart, arrayEnd );
296 
297  if ( value.empty() && default_also ) {
298  if ( !opt->get_default_str().empty() ) {
299  value = detail::convert_arg_for_ini( opt->get_default_str() );
300  }
301  else if ( opt->get_expected_min() == 0 ) {
302  value = "false";
303  }
304  }
305 
306  if ( !value.empty() ) {
307  if ( write_description && opt->has_description() ) {
308  out << '\n';
309  out << commentLead
310  << detail::fix_newlines( commentLead, opt->get_description() ) << '\n';
311  }
312  out << name << valueDelimiter << value << '\n';
313  }
314  }
315  }
316  }
317  auto subcommands = app->get_subcommands( {} );
318  for ( const App* subcom : subcommands ) {
319  if ( subcom->get_name().empty() ) {
320  if ( write_description && !subcom->get_group().empty() ) {
321  out << '\n' << commentLead << subcom->get_group() << " Options\n";
322  }
323  out << to_config( subcom, default_also, write_description, prefix );
324  }
325  }
326 
327  for ( const App* subcom : subcommands ) {
328  if ( !subcom->get_name().empty() ) {
329  if ( subcom->get_configurable() && app->got_subcommand( subcom ) ) {
330  if ( !prefix.empty() || app->get_parent() == nullptr ) {
331  out << '[' << prefix << subcom->get_name() << "]\n";
332  }
333  else {
334  std::string subname = app->get_name() + "." + subcom->get_name();
335  auto p = app->get_parent();
336  while ( p->get_parent() != nullptr ) {
337  subname = p->get_name() + "." + subname;
338  p = p->get_parent();
339  }
340  out << '[' << subname << "]\n";
341  }
342  out << to_config( subcom, default_also, write_description, "" );
343  }
344  else {
345  out << to_config(
346  subcom, default_also, write_description, prefix + subcom->get_name() + "." );
347  }
348  }
349  }
350 
351  return out.str();
352 }
