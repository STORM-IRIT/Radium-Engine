101  struct hash_vec {
102  size_t operator()( const VertexDataInternal& lvalue ) const {
103  size_t hx = std::hash<Scalar>()( lvalue._vertex[0] );
104  size_t hy = std::hash<Scalar>()( lvalue._vertex[1] );
105  size_t hz = std::hash<Scalar>()( lvalue._vertex[2] );
106  return ( hx ^ ( hy << 1 ) ) ^ hz;
107  }
108  };
109 
110  TriangleMesh out;
111 
112  using VertexMap = std::unordered_map<VertexDataInternal, int, hash_vec>;
113 
114  VertexMap vertexHandles;
115 
116  if ( !get_property_handle( m_outputTriangleMeshIndexPph, "OutputTriangleMeshIndices" ) ) {
117  add_property( m_outputTriangleMeshIndexPph, "OutputTriangleMeshIndices" );
118  }
119  std::vector<PropPair<Scalar>> vprop_float;
120  std::vector<PropPair<Vector2>> vprop_vec2;
121  std::vector<PropPair<Vector3>> vprop_vec3;
122  std::vector<PropPair<Vector4>> vprop_vec4;
123 
124  // loop over all attribs and build correspondance pair
125  vprop_float.reserve( m_floatPph.size() );
126  for ( auto oh : m_floatPph )
127  addAttribPairToCore( out, this, oh, vprop_float );
128  vprop_vec2.reserve( m_vec2Pph.size() );
129  for ( auto oh : m_vec2Pph )
130  addAttribPairToCore( out, this, oh, vprop_vec2 );
131  vprop_vec3.reserve( m_vec3Pph.size() );
132  for ( auto oh : m_vec3Pph )
133  addAttribPairToCore( out, this, oh, vprop_vec3 );
134  vprop_vec4.reserve( m_vec4Pph.size() );
135  for ( auto oh : m_vec4Pph )
136  addAttribPairToCore( out, this, oh, vprop_vec4 );
137 
138  // iterator over all faces
139  unsigned int vertexIndex = 0;
140 
141  // out will have at least n_vertices vertices and normals.
142  TriangleMesh::PointAttribHandle::Container vertices;
143  TriangleMesh::NormalAttribHandle::Container normals;
144  TriangleMesh::IndexContainerType indices;
145 
146  vertices.reserve( n_vertices() );
147  normals.reserve( n_vertices() );
148  indices.reserve( n_faces() );
149 
150  for ( TopologicalMesh::FaceIter f_it = faces_sbegin(); f_it != faces_end(); ++f_it ) {
151  int tindices[3];
152  int i = 0;
153 
154  // iterator over vertex (through halfedge to get access to halfedge normals)
155  for ( TopologicalMesh::ConstFaceHalfedgeIter fh_it = cfh_iter( *f_it ); fh_it.is_valid();
156  ++fh_it ) {
157  VertexDataInternal v;
158  CORE_ASSERT( i < 3, "Non-triangular face found." );
159  v._vertex = point( to_vertex_handle( *fh_it ) );
160 
161  if ( has_halfedge_normals() ) {
162  v._normal = normal( to_vertex_handle( *fh_it ), *f_it );
163  }
164  copyAttribToCoreVertex( v._float, this, vprop_float, *fh_it );
165  copyAttribToCoreVertex( v._vec2, this, vprop_vec2, *fh_it );
166  copyAttribToCoreVertex( v._vec3, this, vprop_vec3, *fh_it );
167  copyAttribToCoreVertex( v._vec4, this, vprop_vec4, *fh_it );
168 
169  int vi;
170  VertexMap::iterator vtr = vertexHandles.find( v );
171  if ( vtr == vertexHandles.end() ) {
172  vi = int( vertexIndex++ );
173  vertexHandles.insert( vtr, VertexMap::value_type( v, vi ) );
174  vertices.push_back( v._vertex );
175  if ( has_halfedge_normals() ) { normals.push_back( v._normal ); }
176  copyAttribToCore( out, v._float );
177  copyAttribToCore( out, v._vec2 );
178  copyAttribToCore( out, v._vec3 );
179  copyAttribToCore( out, v._vec4 );
180  }
181  else {
182  vi = vtr->second;
183  }
184  tindices[i] = vi;
185  property( m_outputTriangleMeshIndexPph, *fh_it ) = vi;
186  i++;
187  }
188  indices.emplace_back( tindices[0], tindices[1], tindices[2] );
189  }
190  out.setVertices( std::move( vertices ) );
191  if ( has_halfedge_normals() ) { out.setNormals( std::move( normals ) ); }
192  out.setIndices( std::move( indices ) );
193  CORE_ASSERT( vertexIndex == vertices.size(),
194  "Inconsistent number of faces in generated TriangleMesh." );
195 
196  return out;

199 bool TopologicalMesh::splitEdge( TopologicalMesh::EdgeHandle eh, Scalar f ) {
200  // Global schema of operation
201  /*
202  TRIANGLES ONLY
203  before after
204  A A
205  / F0 \ / F2 | F0 \
206  / \ / | \
207  /h1 h0\ /h1 e2|e0 h0\
208  / he0 \ / he2 | he0 \
209  V1 -------- V0 V1 ------ V ------ V0
210  \ he1 / \ he3 | he1 /
211  \o1 o0/ \o1 e3|e1 o0/
212  \ / \ | /
213  \ F1 / \ F3 | F1 /
214  B B
215 
216  */
217 
218  // incorrect factor
219  if ( f < 0 || f > 1 ) { return false; }
220 
221  // get existing topology data
222  HalfedgeHandle he0 = halfedge_handle( eh, 0 );
223  HalfedgeHandle he1 = halfedge_handle( eh, 1 );
224  VertexHandle v0 = to_vertex_handle( he0 );
225  VertexHandle v1 = to_vertex_handle( he1 );
226  FaceHandle F0 = face_handle( he0 );
227  FaceHandle F1 = face_handle( he1 );
228 
229  // not triangles or holes
230  if ( ( !is_boundary( he0 ) && valence( F0 ) != 3 ) ||
231  ( !is_boundary( he1 ) && valence( F1 ) != 3 ) ) {
232  return false;
233  }
234 
235  // add the new vertex
236  const Point p = Point( f * point( v0 ) + ( Scalar( 1. ) - f ) * point( v1 ) );
237  VertexHandle v = add_vertex( p );
238 
239  // create the new faces and reconnect the topology
240  HalfedgeHandle he3 = new_edge( v, v1 );
241  HalfedgeHandle he2 = opposite_halfedge_handle( he3 );
242  set_halfedge_handle( v, he0 );
243  set_vertex_handle( he1, v );
244 
245  // does F0 exist
246  if ( !is_boundary( he0 ) ) {
247  HalfedgeHandle h0 = next_halfedge_handle( he0 );
248  HalfedgeHandle h1 = next_halfedge_handle( h0 );
249  // create new edge
250  VertexHandle A = to_vertex_handle( h0 );
251  HalfedgeHandle e2 = new_edge( v, A );
252  HalfedgeHandle e0 = opposite_halfedge_handle( e2 );
253  // split F0
254  FaceHandle F2 = new_face();
255  set_halfedge_handle( F0, he0 );
256  set_halfedge_handle( F2, h1 );
257  // update F0
258  set_face_handle( h0, F0 );
259  set_face_handle( e0, F0 );
260  set_face_handle( he0, F0 );
261  set_next_halfedge_handle( he0, h0 );
262  set_next_halfedge_handle( h0, e0 );
263  set_next_halfedge_handle( e0, he0 );
264  // update F2
265  set_face_handle( h1, F2 );
266  set_face_handle( he2, F2 );
267  set_face_handle( e2, F2 );
268  set_next_halfedge_handle( e2, h1 );
269  set_next_halfedge_handle( h1, he2 );
270  set_next_halfedge_handle( he2, e2 );
271  // deal with custom properties
272  // interpolate at he2
273  interpolateAllProps( h1, he0, he2, 0.5 );
274  // copy at e0, and e2
275  copyAllProps( he2, e0 );
276  }
277  else {
278  HalfedgeHandle h1 = prev_halfedge_handle( he0 );
279  set_next_halfedge_handle( h1, he2 );
280  set_next_halfedge_handle( he2, he0 );
281  // next halfedge handle of he0 already is h0
282  // halfedge handle of V already is he0
283  }
284 
285  // does F1 exist
286  if ( !is_boundary( he1 ) ) {
287  HalfedgeHandle o1 = next_halfedge_handle( he1 );
288  HalfedgeHandle o0 = next_halfedge_handle( o1 );
289  // create new edge
290  VertexHandle B = to_vertex_handle( o1 );
291  HalfedgeHandle e1 = new_edge( v, B );
292  HalfedgeHandle e3 = opposite_halfedge_handle( e1 );
293  // split F1
294  FaceHandle F3 = new_face();
295  set_halfedge_handle( F3, o1 );
296  set_halfedge_handle( F1, he1 );
297  // update F1
298  set_face_handle( o1, F3 );
299  set_face_handle( e3, F3 );
300  set_face_handle( he3, F3 );
301  set_next_halfedge_handle( he3, o1 );
302  set_next_halfedge_handle( o1, e3 );
303  set_next_halfedge_handle( e3, he3 );
304  // update F3
305  set_face_handle( o0, F1 );
306  set_face_handle( he1, F1 );
307  set_face_handle( e1, F1 );
308  set_next_halfedge_handle( he1, e1 );
309  set_next_halfedge_handle( e1, o0 );
310  set_next_halfedge_handle( o0, he1 );
311  // deal with custom properties
312  // first copy at he3
313  copyAllProps( he1, he3 );
314  // interpolate at he1
315  interpolateAllProps( o0, he3, he1, 0.5 );
316  // copy at e1, and e3
317  copyAllProps( he1, e3 );
318  copyAllProps( o1, e1 );
319  }
320  else {
321  HalfedgeHandle o1 = next_halfedge_handle( he1 );
322  // next halfedge handle of o0 already is he1
323  set_next_halfedge_handle( he1, he3 );
324  set_next_halfedge_handle( he3, o1 );
325  // halfedge handle of V already is he0
326  }
327 
328  // ensure consistency at v1
329  if ( halfedge_handle( v1 ) == he0 ) { set_halfedge_handle( v1, he2 ); }
330 
331  return true;
332 }

426 void TopologicalMesh::split_copy( EdgeHandle _eh, VertexHandle _vh ) {
427  const VertexHandle v0 = to_vertex_handle( halfedge_handle( _eh, 0 ) );
428  const VertexHandle v1 = to_vertex_handle( halfedge_handle( _eh, 1 ) );
429 
430  const int nf = n_faces();
431 
432  // Split the halfedge ( handle will be preserved)
433  split( _eh, _vh );
434 
435  // Copy the properties of the original edge to all neighbor edges that
436  // have been created
437  for ( VEIter ve_it = ve_iter( _vh ); ve_it.is_valid(); ++ve_it )
438  copy_all_properties( _eh, *ve_it, true );
439 
440  for ( auto vh : { v0, v1 } ) {
441  // get the halfedge pointing from new vertex to old vertex
442  const HalfedgeHandle h = find_halfedge( _vh, vh );
443  // for boundaries there are no faces whose properties need to be copied
444  if ( !is_boundary( h ) ) {
445  FaceHandle fh0 = face_handle( h );
446  FaceHandle fh1 = face_handle( opposite_halfedge_handle( prev_halfedge_handle( h ) ) );
447  // is fh0 the new face?
448  if ( fh0.idx() >= nf ) std::swap( fh0, fh1 );
449 
450  // copy properties from old face to new face
451  copy_all_properties( fh0, fh1, true );
452  }
