238 template <typename T>
239 AttribHandle<T> AttribManager::addAttrib( const std::string& name ) {
240  // does the attrib already exist?
241  AttribHandle<T> h = findAttrib<T>( name );
242  if ( isValid( h ) ) return h;
243 
244  // create the attrib
245  smart_pointer_type attrib = std::make_unique<Attrib<T>>( name );
246 
247  // look for a free slot
248  auto it = std::find_if(
249  m_attribs.begin(), m_attribs.end(), []( const auto& attr ) { return !attr; } );
250  if ( it != m_attribs.end() )
251  {
252  it->swap( attrib );
253  h.m_idx = std::distance( m_attribs.begin(), it );
254  }
255  else
256  {
257  m_attribs.push_back( std::move( attrib ) );
258  h.m_idx = m_attribs.size() - 1;
259  }
260  m_attribsIndex[name] = h.m_idx;
261  h.m_name = name;
262  ++m_numAttribs;
263 
264  notify( name );
265  return h;
266 }
