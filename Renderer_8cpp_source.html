414 void Renderer::doPicking( const ViewingParameters& renderData ) {
415  m_pickingResults.reserve( m_pickingQueries.size() );
416 
417  m_pickingFbo->bind();
418  preparePicking( renderData );
419 
420  // Now read the Picking Texture to address the Picking Requests.
421  GL_ASSERT( glReadBuffer( GL_COLOR_ATTACHMENT0 ) );
422 
423  int pick[4];
424  for ( const PickingQuery& query : m_pickingQueries )
425  {
426  PickingResult result;
427  // fill picking result according to picking mode
428  if ( query.m_mode < C_VERTEX )
429  {
430  // skip query if out of window (can occur when picking while moving outside)
431  if ( query.m_screenCoords.x() < 0 || query.m_screenCoords.x() > m_width - 1 ||
432  query.m_screenCoords.y() < 0 || query.m_screenCoords.y() > m_height - 1 )
433  {
434  result.m_roIdx = -1;
435  m_pickingResults.push_back( result );
436  continue;
437  }
438  GL_ASSERT( glReadPixels( query.m_screenCoords.x(),
439  query.m_screenCoords.y(),
440  1,
441  1,
442  GL_RGBA_INTEGER,
443  GL_INT,
444  pick ) );
445  result.m_roIdx = pick[0]; // RO idx
446  result.m_vertexIdx.emplace_back( pick[1] ); // vertex idx in the element
447  result.m_elementIdx.emplace_back( pick[2] ); // element idx
448  result.m_edgeIdx.emplace_back( pick[3] ); // edge opposite idx for triangles
449  }
450  else
451  {
452  // select the results for the RO with the most representatives
453  // (or first to come if same amount)
454  std::map<int, PickingResult> resultPerRO;
455  for ( auto i = -m_brushRadius; i <= m_brushRadius; i += 3 )
456  {
457  auto h = std::round( std::sqrt( m_brushRadius * m_brushRadius - i * i ) );
458  for ( auto j = -h; j <= +h; j += 3 )
459  {
460  const int x = query.m_screenCoords.x() + i;
461  const int y = query.m_screenCoords.y() - j;
462  // skip query if out of window (can occur when picking while moving outside)
463  if ( x < 0 || x > int( m_width ) - 1 || y < 0 || y > int( m_height ) - 1 )
464  { continue; }
465  GL_ASSERT( glReadPixels( x, y, 1, 1, GL_RGBA_INTEGER, GL_INT, pick ) );
466  resultPerRO[pick[0]].m_roIdx = pick[0];
467  resultPerRO[pick[0]].m_vertexIdx.emplace_back( pick[1] );
468  resultPerRO[pick[0]].m_elementIdx.emplace_back( pick[2] );
469  resultPerRO[pick[0]].m_edgeIdx.emplace_back( pick[3] );
470  }
471  }
472 
473  auto itr = std::max_element(
474  resultPerRO.begin(),
475  resultPerRO.end(),
476  []( const std::map<int, PickingResult>::value_type& a,
477  const std::map<int, PickingResult>::value_type& b ) -> bool {
478  return a.second.m_vertexIdx.size() < b.second.m_vertexIdx.size();
479  } );
480  result = itr->second;
481  }
482  result.m_mode = query.m_mode;
483  m_pickingResults.push_back( result );
484  }
485 
486  m_pickingFbo->unbind();
487 }
488 void Renderer::preparePicking( const ViewingParameters& renderData ) {
489 
490  GL_ASSERT( glDepthMask( GL_TRUE ) );
491  GL_ASSERT( glColorMask( 1, 1, 1, 1 ) );
492  GL_ASSERT( glDrawBuffers( 1, buffers ) );
493 
494  float clearDepth = 1.0;
495  int clearColor[] = {-1, -1, -1, -1};
496 
497  GL_ASSERT( glClearBufferiv( GL_COLOR, 0, clearColor ) );
498  GL_ASSERT( glClearBufferfv( GL_DEPTH, 0, &clearDepth ) );
499 
500  splitRenderQueuesForPicking( renderData );
501 
502  // First draw Geometry Objects
503  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
504  GL_ASSERT( glDepthFunc( GL_LESS ) );
505 
506  renderForPicking( renderData, m_pickingShaders, m_fancyRenderObjectsPicking );
507 
508  // Then draw debug objects
509  GL_ASSERT( glClearBufferfv( GL_DEPTH, 0, &clearDepth ) );
510  if ( m_drawDebug )
511  { renderForPicking( renderData, m_pickingShaders, m_debugRenderObjectsPicking ); }
512 
513  // Then draw xrayed objects on top of normal objects
514  GL_ASSERT( glClearBufferfv( GL_DEPTH, 0, &clearDepth ) );
515  if ( m_drawDebug )
516  { renderForPicking( renderData, m_pickingShaders, m_xrayRenderObjectsPicking ); }
517 
518  // Finally draw ui stuff on top of everything
519  // these have a different way to compute the transform matrices
520  // FIXME (florian): find a way to use renderForPicking()!
521  GL_ASSERT( glClearBufferfv( GL_DEPTH, 0, &clearDepth ) );
522  for ( uint i = 0; i < m_pickingShaders.size(); ++i )
523  {
524  m_pickingShaders[i]->bind();
525  m_pickingShaders[i]->setUniform( "transform.proj", renderData.projMatrix );
526  m_pickingShaders[i]->setUniform( "transform.view", renderData.viewMatrix );
527 
528  for ( const auto& ro : m_uiRenderObjectsPicking[i] )
529  {
530  if ( ro->isVisible() && ro->isPickable() )
531  {
532  m_pickingShaders[i]->setUniform( "objectId", ro->getIndex().getValue() );
533 
534  Core::Matrix4 M = ro->getTransformAsMatrix();
535  Core::Matrix4 MV = renderData.viewMatrix * M;
536  Scalar d = MV.block<3, 1>( 0, 3 ).norm();
537 
538  Core::Matrix4 S = Core::Matrix4::Identity();
539  S( 0, 0 ) = S( 1, 1 ) = S( 2, 2 ) = d;
540 
541  M = M * S;
542  Core::Matrix4 N = M.inverse().transpose();
543 
544  m_pickingShaders[i]->setUniform( "transform.model", M );
545  m_pickingShaders[i]->setUniform( "transform.worldNormal", N );
546 
547  // render
548  ro->getMesh()->render( m_pickingShaders[i] );
549  }
550  }

659 std::unique_ptr<uchar[]> Renderer::grabFrame( size_t& w, size_t& h ) const {
660  Engine::Texture* tex = getDisplayTexture();
661  tex->bind();
662 
663  // Get a buffer to store the pixels of the OpenGL texture (in float format)
664  auto pixels = std::unique_ptr<float[]>( new float[tex->width() * tex->height() * 4] );
665 
666  // Grab the texture data
667  GL_ASSERT( glGetTexImage( GL_TEXTURE_2D, 0, GL_RGBA, GL_FLOAT, pixels.get() ) );
668 
669  // Now we must convert the floats to RGB while flipping the image updisde down.
670  auto writtenPixels = std::unique_ptr<uchar[]>( new uchar[tex->width() * tex->height() * 4] );
671  for ( uint j = 0; j < tex->height(); ++j )
672  {
673  for ( uint i = 0; i < tex->width(); ++i )
674  {
675  auto in = 4 * ( j * tex->width() + i ); // Index in the texture buffer
676  auto ou = 4 * ( ( tex->height() - 1 - j ) * tex->width() +
677  i ); // Index in the final image (note the j flipping).
678 
679  writtenPixels[ou + 0] =
680  (uchar)std::clamp( Scalar( pixels[in + 0] * 255.f ), Scalar( 0 ), Scalar( 255 ) );
681  writtenPixels[ou + 1] =
682  (uchar)std::clamp( Scalar( pixels[in + 1] * 255.f ), Scalar( 0 ), Scalar( 255 ) );
683  writtenPixels[ou + 2] =
684  (uchar)std::clamp( Scalar( pixels[in + 2] * 255.f ), Scalar( 0 ), Scalar( 255 ) );
685  writtenPixels[ou + 3] =
686  (uchar)std::clamp( Scalar( pixels[in + 3] * 255.f ), Scalar( 0 ), Scalar( 255 ) );
687  }
688  }
689  w = tex->width();
690  h = tex->height();
691  return writtenPixels;
692 }
