270 template <typename T>
271 inline bool IndexedGeometry<T>::append( const IndexedGeometry<IndexType>& other ) {
272  const std::size_t verticesBefore = vertices().size();
273  const std::size_t trianglesBefore = m_indices.size();
274 
275  // check same attributes through names
276  if ( !AttribArrayGeometry::append( other ) ) return false;
277 
278  // now we can proceed topology
279  m_indices.insert( m_indices.end(), other.m_indices.cbegin(), other.m_indices.cend() );
280 
281  // Offset the vertex indices in the triangles and faces

283  {
284  for ( uint i = 0; i < IndexType::RowsAtCompileTime; ++i )
