275 void Texture::sRGBToLinearRGB( uint8_t* texels, uint numCommponent, bool hasAlphaChannel ) {
276  if ( !m_isLinear )
277  {
278  m_isLinear = true;
279  // auto linearize = [gamma](float in)-> float {
280  auto linearize = []( uint8_t in ) -> uint8_t {
281  // Constants are described at https://en.wikipedia.org/wiki/SRGB
282  float c = float( in ) / 255;
283  if ( c < 0.04045 ) { c = c / 12.92f; }
284  else
285  { c = std::pow( ( ( c + 0.055f ) / ( 1.055f ) ), 2.4f ); }
286  return uint8_t( c * 255 );
287  };
288  uint numvalues = hasAlphaChannel ? numCommponent - 1 : numCommponent;
289 #pragma omp parallel for
290  for ( int i = 0; i < int( m_textureParameters.width * m_textureParameters.height *
291  m_textureParameters.depth );
292  ++i )
293  {
294  // Convert each R or RGB value while keeping alpha unchanged
295  for ( uint p = i * numCommponent; p < i * numCommponent + numvalues; ++p )
296  {
297  texels[p] = linearize( texels[p] );
298  }
299  }
300  }
301 }
