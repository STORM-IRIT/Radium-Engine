413 void RadiumEngine::TimeData::updateTime( Scalar dt ) {
414  dt += m_realTime ? dt : m_dt;
415  // update the time w.r.t. the time flow policy
416  if ( m_forwardBackward && m_isBackward ) { m_time -= dt; }
417  else
418  { m_time += dt; }
419  // special case: empty time window => forever mode
420  if ( m_endTime < 0 || m_startTime >= m_endTime )
421  {
422  // just run forever
423  m_isBackward = false;
424  return;
425  }
426  // special case: m_time before time window
427  if ( m_time < m_startTime )
428  {
429  // reset whatever the mode
430  m_time = m_startTime;
431  // and go forward from now on
432  m_isBackward = false;
433  return;
434  }
435  // special case: m_time after time window in loop mode
436  if ( !m_forwardBackward && m_time > m_endTime )
437  {
438  // compute the overload of time
439  dt = Scalar( fmod( double( m_time - m_startTime ), double( m_endTime - m_startTime ) ) );
440  // loop around, applying the overload of time
441  m_time = m_startTime + dt;
442  return;
443  }
444  // special case: m_time after time window in forward-backward mode
445  if ( m_forwardBackward && m_time > m_endTime )
446  {
447  if ( !m_isBackward ) // start backwards
448  { m_time = 2 * m_endTime - m_time; }
449  else // restart backward from the end of the time window
450  { m_time = m_endTime; }
451  m_isBackward = true;
452  return;
453  }
454 }
