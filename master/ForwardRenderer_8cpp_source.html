149 void ForwardRenderer::updateStepInternal( const Data::ViewingParameters& renderData ) {
150  CORE_UNUSED( renderData );
151  // TODO : Improve the way RO are distributed in fancy (opaque), transparent and volume
152  // to simplify rendering loop and code maintenance
153  // i.e. Volume should considered as transparent but stored in the volumetric list and
154  // transparent-but-not-volume object should be kept in the fancy list, ...
155  m_transparentRenderObjects.clear();
156  m_volumetricRenderObjects.clear();
157  for ( auto it = m_fancyRenderObjects.begin(); it != m_fancyRenderObjects.end(); ) {
158  if ( ( *it )->isTransparent() ) {
159  m_transparentRenderObjects.push_back( *it );
160  it = m_fancyRenderObjects.erase( it );
161  }
162  else {
163  auto material = ( *it )->getMaterial();
164  if ( material &&
165  material->getMaterialAspect() == Data::Material::MaterialAspect::MAT_DENSITY ) {
166  m_volumetricRenderObjects.push_back( *it );
167  it = m_fancyRenderObjects.erase( it );
168  }
169  else {
170  ++it;
171  }
172  }
173  }
174  m_fancyTransparentCount = m_transparentRenderObjects.size();
175  m_fancyVolumetricCount = m_volumetricRenderObjects.size();
176 
177  // simple hack to clean wireframes ...
178  if ( m_fancyRenderObjects.size() < m_wireframes.size() ) { m_wireframes.clear(); }

181 template <typename IndexContainerType>
182 void computeIndices( Core::Geometry::LineMesh::IndexContainerType& indices,
183  IndexContainerType& other ) {
184 
185  for ( const auto& index : other ) {
186  auto s = index.size();
187  for ( unsigned int i = 0; i < s; ++i ) {
188  int i1 = index[i];
189  int i2 = index[( i + 1 ) % s];
190  if ( i1 > i2 ) std::swap( i1, i2 );
191  indices.emplace_back( i1, i2 );
192  }
193  }
194 
195  std::sort( indices.begin(),
196  indices.end(),
197  []( const Core::Geometry::LineMesh::IndexType& a,
198  const Core::Geometry::LineMesh::IndexType& b ) {
199  return a[0] < b[0] || ( a[0] == b[0] && a[1] < b[1] );
200  } );
201  indices.erase( std::unique( indices.begin(), indices.end() ), indices.end() );
202 }

234 // create a linemesh to draw wireframe given a core mesh
235 template <typename CoreGeometry>
236 void setupLineMesh( std::shared_ptr<Data::LineMesh>& disp, CoreGeometry& core ) {
237 
238  Core::Geometry::LineMesh lines;
239  Core::Geometry::LineMesh::IndexContainerType indices;
240 
241  lines.setVertices( core.vertices() );
242  computeIndices( indices, core.getIndices() );
243  if ( indices.size() > 0 ) {
244  lines.setIndices( std::move( indices ) );
245  disp =
246  Ra::Core::make_shared<Data::LineMesh>( std::string( "wireframe" ), std::move( lines ) );
247  disp->updateGL();
248 
249  // add observer
250  auto handle = core.template getAttribHandle<typename CoreGeometry::Point>(
251  Ra::Core::Geometry::getAttribName( Ra::Core::Geometry::VERTEX_POSITION ) );
252  core.vertexAttribs().getAttrib( handle ).attach( VerticesUpdater( disp, core ) );
253  core.attach( IndicesUpdater( disp, core ) );
254  }
255  else {
256  disp.reset();
257  }

260 void ForwardRenderer::renderInternal( const Data::ViewingParameters& renderData ) {
261 
262  m_fbo->bind();
263 
264  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
265  GL_ASSERT( glDepthMask( GL_TRUE ) );
266  GL_ASSERT( glColorMask( 1, 1, 1, 1 ) );
267 
268  GL_ASSERT( glDrawBuffers( 4, buffers ) );
269  if ( m_wireframe ) {
271  glEnable( GL_POLYGON_OFFSET_FILL );
272  glPolygonOffset( 1.f, 3.f );
273  // GL_ASSERT( glDepthRange( 0.001, 1.0 ) );
274  }
275  else {
276  glDisable( GL_POLYGON_OFFSET_FILL );
277  }
278  static const auto clearZeros = Core::Utils::Color::Black().cast<GL_SCALAR_PLAIN>().eval();
279  static const auto clearOnes = Core::Utils::Color::White().cast<GL_SCALAR_PLAIN>().eval();
280  static const float clearDepth { 1.0f };
281 
283  auto bgColor = getBackgroundColor().cast<GL_SCALAR_PLAIN>().eval();
284  GL_ASSERT( glClearBufferfv( GL_COLOR, 0, bgColor.data() ) ); // Clear color
285  GL_ASSERT( glClearBufferfv( GL_COLOR, 1, clearZeros.data() ) ); // Clear normals
286  GL_ASSERT( glClearBufferfv( GL_COLOR, 2, clearZeros.data() ) ); // Clear diffuse
287  GL_ASSERT( glClearBufferfv( GL_COLOR, 3, clearZeros.data() ) ); // Clear specular
288  GL_ASSERT( glClearBufferfv( GL_DEPTH, 0, &clearDepth ) ); // Clear depth
289 
290  // render background objects (eg sky box)
291  renderBackground( renderData );
292 
293  // Z prepass
294  GL_ASSERT( glDepthFunc( GL_LESS ) );
295  GL_ASSERT( glDisable( GL_BLEND ) );
296  GL_ASSERT( glPointSize( 3.f ) );
297 
298  // Set in RenderParam the configuration about ambiant lighting (instead of hard constant
299  // direclty in shaders)
300  Data::RenderParameters zprepassParams;
301  for ( const auto& ro : m_fancyRenderObjects ) {
302  ro->render( zprepassParams, renderData, DefaultRenderingPasses::Z_PREPASS );
303  }
304  // Transparent objects are rendered in the Z-prepass, but only their fully opaque fragments
305  // (if any) might influence the z-buffer.
306  // Rendering transparent objects assuming that they
307  // discard all their non-opaque fragments
308  for ( const auto& ro : m_transparentRenderObjects ) {
309  ro->render( zprepassParams, renderData, DefaultRenderingPasses::Z_PREPASS );
310  }
311  // Volumetric objects are not rendered in the Z-prepass
312 
313  // Opaque Lighting pass
314  GL_ASSERT( glDepthFunc( GL_LEQUAL ) );
315  GL_ASSERT( glDepthMask( GL_FALSE ) );
316 
317  GL_ASSERT( glEnable( GL_BLEND ) );
318  GL_ASSERT( glBlendFunc( GL_ONE, GL_ONE ) );
319 
320  GL_ASSERT( glDrawBuffers( 1, buffers ) ); // Draw color texture
321 
322  // Radium V2 : this render loop might be greatly improved by inverting light and objects
323  // loop.
324  // Make shaders bounded only once, minimize full stats-changes, ...
325  if ( m_lightmanagers[0]->count() > 0 ) {
326  // for ( const auto& l : m_lights )
327  for ( size_t i = 0; i < m_lightmanagers[0]->count(); ++i ) {
328  const auto l = m_lightmanagers[0]->getLight( i );
329  Data::RenderParameters lightingpassParams;
330  l->getRenderParameters( lightingpassParams );
331 
332  for ( const auto& ro : m_fancyRenderObjects ) {
333  ro->render(
334  lightingpassParams, renderData, DefaultRenderingPasses::LIGHTING_OPAQUE );
335  }
336  // Rendering transparent objects assuming that they discard all their non-opaque
337  // fragments
338  for ( const auto& ro : m_transparentRenderObjects ) {
339  ro->render(
340  lightingpassParams, renderData, DefaultRenderingPasses::LIGHTING_OPAQUE );
341  }
342  }
343  }
344  else {
345  LOG( logINFO ) << "Opaque : no light sources, unable to render";
346  }
347 
348  // Transparency (blending) pass
349  if ( !m_transparentRenderObjects.empty() ) {
350  m_fbo->unbind();
351 
352  m_oitFbo->bind();
353 
354  GL_ASSERT( glDrawBuffers( 2, buffers ) );
355  GL_ASSERT( glClearBufferfv( GL_COLOR, 0, clearZeros.data() ) );
356  GL_ASSERT( glClearBufferfv( GL_COLOR, 1, clearOnes.data() ) );
357 
358  GL_ASSERT( glDepthFunc( GL_LESS ) );
359  GL_ASSERT( glEnable( GL_BLEND ) );
360 
361  GL_ASSERT( glBlendEquation( GL_FUNC_ADD ) );
362  GL_ASSERT( glBlendFunci( 0, GL_ONE, GL_ONE ) );
363  GL_ASSERT( glBlendFunci( 1, GL_ZERO, GL_ONE_MINUS_SRC_ALPHA ) );
364 
365  if ( m_lightmanagers[0]->count() > 0 ) {
366  // for ( const auto& l : m_lights )
367  for ( size_t i = 0; i < m_lightmanagers[0]->count(); ++i ) {
368  const auto l = m_lightmanagers[0]->getLight( i );
369  Data::RenderParameters trasparencypassParams;
370  l->getRenderParameters( trasparencypassParams );
371 
372  for ( const auto& ro : m_transparentRenderObjects ) {
373  ro->render( trasparencypassParams,
374  renderData,
375  DefaultRenderingPasses::LIGHTING_TRANSPARENT );
376  }
377  }
378  }
379  else {
380  LOG( logINFO ) << "Transparent : no light sources, unable to render";
381  }
382 
383  m_oitFbo->unbind();
384 
385  m_fbo->bind();
386  GL_ASSERT( glDrawBuffers( 1, buffers ) );
387  GL_ASSERT( glDisable( GL_DEPTH_TEST ) );
388  GL_ASSERT( glBlendFunc( GL_ONE_MINUS_SRC_ALPHA, GL_SRC_ALPHA ) );
389  {
390  auto shader = m_shaderProgramManager->getShaderProgram( "ComposeOIT" );
391  shader->bind();
392  shader->setUniform( "u_OITSumColor", m_textures[RendererTextures_OITAccum].get(), 0 );
393  shader->setUniform(
394  "u_OITSumWeight", m_textures[RendererTextures_OITRevealage].get(), 1 );
395 
396  m_quadMesh->render( shader );
397  }
398  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
399  }
400  // Volumetric pass
401  // Z-test is enabled but z-write must be disable to allow access to the z-buffer in the
402  // shader.
403  // This pass render in its own FBO and copy the result to the main colortexture
404  if ( !m_volumetricRenderObjects.empty() ) {
405  m_volumeFbo->bind();
406  GL_ASSERT( glDrawBuffers( 1, buffers ) );
407  auto alpha = Core::Utils::Color::Alpha().cast<GL_SCALAR_PLAIN>().eval();
408  GL_ASSERT( glClearBufferfv( GL_COLOR, 0, alpha.data() ) );
409  GL_ASSERT( glDisable( GL_BLEND ) );
410  // for ( const auto& l : m_lights )
411  for ( size_t i = 0; i < m_lightmanagers[0]->count(); ++i ) {
412  const auto l = m_lightmanagers[0]->getLight( i );
413  Data::RenderParameters passParams;
414  l->getRenderParameters( passParams );
415  passParams.addParameter( "imageColor", m_textures[RendererTextures_HDR].get() );
416  passParams.addParameter( "imageDepth", m_textures[RendererTextures_Depth].get() );
417 
418  for ( const auto& ro : m_volumetricRenderObjects ) {
419  ro->render( passParams, renderData, DefaultRenderingPasses::LIGHTING_VOLUMETRIC );
420  }
421  }
422  m_volumeFbo->unbind();
423 
424  m_fbo->bind();
425  GL_ASSERT( glDrawBuffers( 1, buffers ) );
426  GL_ASSERT( glDisable( GL_DEPTH_TEST ) );
427  GL_ASSERT( glEnable( GL_BLEND ) );
428  GL_ASSERT( glBlendFunc( GL_ONE_MINUS_SRC_ALPHA, GL_SRC_ALPHA ) );
429  {
430  auto shader = m_shaderProgramManager->getShaderProgram( "ComposeVolume" );
431  shader->bind();
432  shader->setUniform( "volumeImage", m_textures[RendererTextures_Volume].get(), 0 );
433  m_quadMesh->render( shader );
434  }
435  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
436  }
437 
438  if ( m_wireframe ) {
439 
440  m_fbo->bind();
441 
442  glDisable( GL_POLYGON_OFFSET_FILL );
443  GL_ASSERT( glDepthFunc( GL_LEQUAL ) );
444  GL_ASSERT( glEnable( GL_BLEND ) );
445  glBlendEquationSeparate( GL_FUNC_ADD, GL_FUNC_ADD );
446  glBlendFuncSeparate( GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA, GL_ONE, GL_ZERO );
447  GL_ASSERT( glDrawBuffers( 1, buffers ) ); // Draw color texture
448 
449  auto drawWireframe = [this, &renderData]( const auto& ro ) {
450  std::shared_ptr<Data::Displayable> wro;
451 
452  WireMap::iterator it = m_wireframes.find( ro.get() );
453  if ( it == m_wireframes.end() ) {
454  std::shared_ptr<Data::LineMesh> disp;
455 
456  using trimesh = Ra::Engine::Data::IndexedGeometry<Ra::Core::Geometry::TriangleMesh>;
457  using polymesh = Ra::Engine::Data::IndexedGeometry<Ra::Core::Geometry::PolyMesh>;
458  using quadmesh = Ra::Engine::Data::IndexedGeometry<Ra::Core::Geometry::QuadMesh>;
459 
460  auto displayable = ro->getMesh();
461  auto tm = std::dynamic_pointer_cast<trimesh>( displayable );
462  auto tp = std::dynamic_pointer_cast<polymesh>( displayable );
463  auto tq = std::dynamic_pointer_cast<quadmesh>( displayable );
464 
465  auto processLineMesh = []( auto cm, std::shared_ptr<Data::LineMesh>& lm ) {
466  if ( cm->getRenderMode() ==
467  Data::AttribArrayDisplayable::MeshRenderMode::RM_TRIANGLES ) {
468  setupLineMesh( lm, cm->getCoreGeometry() );
469  }
470  };
471  if ( tm ) { processLineMesh( tm, disp ); }
472  if ( tp ) { processLineMesh( tp, disp ); }
473  if ( tq ) { processLineMesh( tq, disp ); }
474 
475  m_wireframes[ro.get()] = disp;
476  wro = disp;
477  }
478  else {
479  wro = it->second;
480  }
481 
482  const Data::ShaderProgram* shader =
483  m_shaderProgramManager->getShaderProgram( "Wireframe" );
484 
485  if ( shader && wro ) {
486  shader->bind();
487  if ( ro->isVisible() ) {
488  wro->updateGL();
489 
490  Core::Matrix4 modelMatrix = ro->getTransformAsMatrix();
491  shader->setUniform( "transform.proj", renderData.projMatrix );
492  shader->setUniform( "transform.view", renderData.viewMatrix );
493  shader->setUniform( "transform.model", modelMatrix );
494  shader->setUniform( "viewport", Core::Vector2 { m_width, m_height } );
495  wro->render( shader );
496 
497  GL_CHECK_ERROR;
498  }
499  }
500  };
501 
502  for ( const auto& ro : m_fancyRenderObjects ) {
503  drawWireframe( ro );
504  }
505  for ( const auto& ro : m_transparentRenderObjects ) {
506  drawWireframe( ro );
507  }
508  }
509 
510  // Restore state
511  GL_ASSERT( glDepthFunc( GL_LESS ) );
512  GL_ASSERT( glDisable( GL_BLEND ) );
513  m_fbo->unbind();
514 }

550 // Draw UI stuff, always drawn on top of everything else + clear ZMask
551 void ForwardRenderer::uiInternal( const Data::ViewingParameters& renderData ) {
552 
553  m_uiXrayFbo->bind();
554  glDrawBuffers( 1, buffers );
555  // Enable z-test
556  GL_ASSERT( glDepthMask( GL_TRUE ) );
557  GL_ASSERT( glEnable( GL_DEPTH_TEST ) );
558  GL_ASSERT( glDepthFunc( GL_LESS ) );
559  GL_ASSERT( glClear( GL_DEPTH_BUFFER_BIT ) );
560  for ( const auto& ro : m_uiRenderObjects ) {
561  if ( ro->isVisible() ) {
562  auto shader = ro->getRenderTechnique()->getShader();
563 
564  // bind data
565  shader->bind();
566 
567  Core::Matrix4 M = ro->getTransformAsMatrix();
568  Core::Matrix4 MV = renderData.viewMatrix * M;
569  Core::Vector3 V = MV.block<3, 1>( 0, 3 );
570  Scalar d = V.norm();
571 
572  Core::Matrix4 S = Core::Matrix4::Identity();
573  S.coeffRef( 0, 0 ) = S.coeffRef( 1, 1 ) = S.coeffRef( 2, 2 ) = d;
574 
575  M = M * S;
576 
577  shader->setUniform( "transform.proj", renderData.projMatrix );
578  shader->setUniform( "transform.view", renderData.viewMatrix );
579  shader->setUniform( "transform.model", M );

703 /*
704  * Build renderTechnique for Forward Renderer : this is the default in Radium, so create Default
705  * Render Technique
706  */
707 bool ForwardRenderer::buildRenderTechnique( RenderObject* ro ) const {
708  auto material = ro->getMaterial();
709  if ( !material ) {
710  LOG( logWARNING ) << "ForwardRenderer : no material found when building RenderTechnique"
711  << " - adding red Lambertian material";
712  auto defMat = new Data::LambertianMaterial( "ForwardRenderer::Default material" );
713  defMat->m_color = Ra::Core::Utils::Color::Red();
714  material.reset( defMat );
715  }
716  auto builder = EngineRenderTechniques::getDefaultTechnique( material->getMaterialName() );
717  auto rt = Core::make_shared<RenderTechnique>();
718  // define the technique for rendering this RenderObject (here, using the default from
719  // Material name)
720  // NOTE : Setting transparency to true does not hurt performances here. It will just allow to
721  // change the transparency of an object online
722  builder.second( *rt, true /*material->isTransparent()*/ );
723  // If renderObject is a point cloud, add geometry shader for splatting
724  auto RenderedGeometry = ro->getMesh().get();
725 
726  /*
727  * WARNING : this way of managing specific geometries is here only for testing and
728  * experimentation purpose. It will be replace soon by a better management of the way
729  * components could add specific properties to a rendertechnique
730  * TODO : see PR Draft and gist subShaderBlob
731  */
732  if ( RenderedGeometry && RenderedGeometry->getNumFaces() == 0 ) {
733  auto pointCloud = dynamic_cast<Scene::PointCloudComponent*>( ro->getComponent() );
734  if ( pointCloud ) {
735  auto addGeomShader = [&rt]( Core::Utils::Index pass ) {
736  if ( rt->hasConfiguration( pass ) ) {
737  Data::ShaderConfiguration config = rt->getConfiguration( pass );
738  config.addShader( Data::ShaderType_GEOMETRY,
739  RadiumEngine::getInstance()->getResourcesDir() +
740  "Shaders/Points/PointCloud.geom.glsl" );
741  rt->setConfiguration( config, pass );
742  }
743  };
744 
745  addGeomShader( DefaultRenderingPasses::LIGHTING_OPAQUE );
746  addGeomShader( DefaultRenderingPasses::LIGHTING_TRANSPARENT );
747  addGeomShader( DefaultRenderingPasses::Z_PREPASS );
748  // construct the parameter provider for the technique
749  auto pr = std::make_shared<PointCloudParameterProvider>( material, pointCloud );
750  rt->setParametersProvider( pr );
751  }
752  else {
753  rt->setParametersProvider( material );
754  }
755  }
756  else {
757  // make the material the parameter provider for the technique
758  rt->setParametersProvider( material );
759  }
760  ro->setRenderTechnique( rt );
761  return true;
762 }
