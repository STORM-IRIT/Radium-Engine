12 template <uint SLICES, uint STACKS>
13 TriangleMesh
14 makeParametricSphere( Scalar radius, const Utils::optional<Utils::Color>& color, bool gtc ) {
15 
16  const Scalar du = 1_ra / SLICES;
17  const Scalar dv = 1_ra / STACKS;
18  using WAI = TopologicalMesh::WedgeAttribIndex;
19  TopologicalMesh topoMesh;
20  auto whNormal = topoMesh.addWedgeAttrib<Vector3>( getAttribName( MeshAttrib::VERTEX_NORMAL ) );
21 
22  WAI whColor =
23  color ? topoMesh.addWedgeAttrib<Vector4>( getAttribName( MeshAttrib::VERTEX_COLOR ) )
24  : WAI::Invalid();
25  WAI whTexCoord =
26  gtc ? topoMesh.addWedgeAttrib<Vector3>( getAttribName( MeshAttrib::VERTEX_TEXCOORD ) )
27  : WAI::Invalid();
28 
29  TopologicalMesh::VertexHandle vhandles[( STACKS - 1 ) * SLICES + 2];
30  Vector3Array topoTexCoords;
31  topoTexCoords.reserve( ( STACKS - 1 ) * SLICES + 2 + 2 * SLICES );
32  Vector3Array topoNormals;
33  topoNormals.reserve( ( STACKS - 1 ) * SLICES + 2 );
34 
35  uint index = 0;
36  // check https://en.wikipedia.org/wiki/Spherical_coordinate_system
37  // theta \in [0, pi]
38  // phi \in [0, 2pi]
39  const Scalar uFactor = 2_ra / Scalar { SLICES } * Core::Math::Pi;
40  const Scalar vFactor = Core::Math::Pi / STACKS;
41  for ( uint u = 0; u < SLICES; ++u ) {
42  const Scalar phi = u * uFactor;
43  for ( uint v = 1; v < STACKS; ++v ) {
44  // Regular vertices on the sphere.
45  const Scalar theta = v * vFactor;
46  Vector3 p = Vector3( radius * std::cos( phi ) * std::sin( theta ),
47  radius * std::sin( phi ) * std::sin( theta ),
48  radius * std::cos( theta ) );
49  vhandles[index++] = topoMesh.add_vertex( p );
50  topoNormals.push_back( p.normalized() );
51  topoTexCoords.emplace_back( 1_ra - u * du, v * dv, 0_ra );
52  }
53  }
54 
55  // Add the pole vertices.
56  auto topoNorthPoleIdx = ( STACKS - 1 ) * SLICES;
57  auto topoSouthPoleIdx = topoNorthPoleIdx + 1;
58  vhandles[topoNorthPoleIdx] = topoMesh.add_vertex( Vector3( 0, 0, radius ) );
59  vhandles[topoSouthPoleIdx] = topoMesh.add_vertex( Vector3( 0, 0, -radius ) );
60 
61  // Set wedge for fh, vhIndex to n and t.
62  auto wedgeSetter = [&topoMesh, &vhandles, color, whNormal, whTexCoord, whColor, gtc](
63  int vhIndex, TopologicalMesh::FaceHandle fh, Vector3 n, Vector3 t ) {
64  auto heh = topoMesh.halfedge_handle( vhandles[vhIndex], fh );
65  auto wd = topoMesh.newWedgeData( heh );
66  wd.m_vector3Attrib[whNormal] = n;
67  if ( gtc ) wd.m_vector3Attrib[whTexCoord] = t;
68  if ( color ) wd.m_vector4Attrib[whColor] = *color;
69  topoMesh.replaceWedge( heh, wd );
70  };
71 
72  // For general vertices retrieve normals and texCoords from vhIndex
73  auto wedgeSetterGeneric = [wedgeSetter, &topoNormals, &topoTexCoords, whNormal, whTexCoord](
74  int vhIndex, TopologicalMesh::FaceHandle fh ) {
75  wedgeSetter( vhIndex, fh, topoNormals[vhIndex], topoTexCoords[vhIndex] );
76  };
77 
78  // take seams into account when u =1
79  auto wedgeSetterSeam = [wedgeSetter, &topoTexCoords, &topoNormals, whNormal, whTexCoord](
80  int u, int vhIndex, TopologicalMesh::FaceHandle fh ) {
81  Vector3 t = topoTexCoords[vhIndex];
82  if ( u == SLICES - 1 ) t[0] = 0_ra;
83  wedgeSetter( vhIndex, fh, topoNormals[vhIndex], t );
84  };
85 
86  // special for poles
87  auto wedgeSetterPole =
88  [wedgeSetter, &topoTexCoords, whNormal, whTexCoord]( bool north,
89  int id,
90  int baseSlice,
91  int nextSlice,
92  int u,
93  TopologicalMesh::FaceHandle fh ) {
94  // pole vertex use "midpoint" texCoord
95  Scalar bu = topoTexCoords[baseSlice][0];
96  Scalar nu = ( u == SLICES - 1 ) ? 0_ra : topoTexCoords[nextSlice][0];
97  Scalar tu = ( bu + nu ) * .5_ra;
98  wedgeSetter( id, fh, Vector3( 0, 0, north ? 1 : -1 ), Vector3( tu, north ? 0 : 1, 0 ) );
99  };
100 
101  for ( uint u = 0; u < SLICES; ++u ) {
102  for ( uint v = 2; v < STACKS; ++v ) {
103  const uint nextSlice = ( ( u + 1 ) % SLICES ) * ( STACKS - 1 );
104  const uint baseSlice = u * ( STACKS - 1 );
105  std::vector vindices = { baseSlice + v - 2,
106  baseSlice + v - 1,
107  nextSlice + v - 1,
108  baseSlice + v - 2,
109  nextSlice + v - 1,
110  nextSlice + v - 2 };
111 
112  auto fh1 = topoMesh.add_face(
113  { vhandles[vindices[0]], vhandles[vindices[1]], vhandles[vindices[2]] } );
114  auto fh2 = topoMesh.add_face(
115  { vhandles[vindices[3]], vhandles[vindices[4]], vhandles[vindices[5]] } );
116 
117  wedgeSetterGeneric( baseSlice + v - 2, fh1 );
118  wedgeSetterGeneric( baseSlice + v - 1, fh1 );
119  wedgeSetterGeneric( baseSlice + v - 2, fh2 );
120 
121  wedgeSetterSeam( u, nextSlice + v - 1, fh1 );
122  wedgeSetterSeam( u, nextSlice + v - 1, fh2 );
123  wedgeSetterSeam( u, nextSlice + v - 2, fh2 );
124  }
125  }
126  // caps faces
127  for ( uint u = 0; u < SLICES; ++u ) {
128  const uint nextSlice = ( ( u + 1 ) % SLICES ) * ( STACKS - 1 );
129  const uint baseSlice = u * ( STACKS - 1 );
130 
131  auto fh1 = topoMesh.add_face(
132  { vhandles[topoNorthPoleIdx], vhandles[baseSlice], vhandles[nextSlice] } );
133  auto fh2 = topoMesh.add_face( { vhandles[topoSouthPoleIdx],
134  vhandles[nextSlice + STACKS - 2],
135  vhandles[baseSlice + STACKS - 2] } );
136 
137  wedgeSetterGeneric( baseSlice, fh1 );
138  wedgeSetterGeneric( baseSlice + STACKS - 2, fh2 );
139  wedgeSetterSeam( u, nextSlice, fh1 );
140  wedgeSetterSeam( u, nextSlice + STACKS - 2, fh2 );
141 
142  // pole vertex use "midpoint" texCoord
143  wedgeSetterPole( true, topoNorthPoleIdx, baseSlice, nextSlice, u, fh1 );
144  wedgeSetterPole( false, topoSouthPoleIdx, baseSlice, nextSlice, u, fh2 );
145  }
146 
147  topoMesh.mergeEqualWedges();
148  topoMesh.garbage_collection();
149  TriangleMesh result = topoMesh.toTriangleMesh();
150  result.checkConsistency();
151  return result;
152 }

154 template <uint U, uint V>
155 TriangleMesh makeParametricTorus( Scalar majorRadius,
156  Scalar minorRadius,
157  const Utils::optional<Utils::Color>& color,
158  bool generateTexCoord ) {
159  TriangleMesh result;
160  TriangleMesh::PointAttribHandle::Container vertices;
161  TriangleMesh::NormalAttribHandle::Container normals;
162  TriangleMesh::IndexContainerType indices;
163  Ra::Core::Vector3Array texCoords;
164 
165  vertices.reserve( ( U + 1 ) * ( V + 1 ) );
166  normals.reserve( ( U + 1 ) * ( V + 1 ) );
167  indices.reserve( 2 * U * V );
168  texCoords.reserve( ( U + 1 ) * ( V + 1 ) );
169 
170  const Scalar du = 1_ra / U;
171  const Scalar dv = 1_ra / V;
172 
173  for ( uint iu = 0; iu <= U; ++iu ) {
174  Scalar u = Scalar( iu ) * Core::Math::PiMul2 / Scalar( U );
175  Core::Vector3 circleCenter( majorRadius * std::cos( u ), majorRadius * std::sin( u ), 0.f );
176 
177  for ( uint iv = 0; iv <= V; ++iv ) {
178  Scalar v = Scalar( iv ) * Core::Math::PiMul2 / Scalar( V );
179 
180  Core::Vector3 vertex( ( majorRadius + minorRadius * std::cos( v ) ) * std::cos( u ),
181  ( majorRadius + minorRadius * std::cos( v ) ) * std::sin( u ),
182  minorRadius * std::sin( v ) );
183 
184  vertices.push_back( vertex );
185  normals.push_back( ( vertex - circleCenter ).normalized() );
186  texCoords.emplace_back( iu * du, iv * dv, 0_ra );
187 
188  if ( iu != U && iv != V ) {
189  indices.push_back( Vector3ui( iu * ( V + 1 ) + iv,
190  ( iu + 1 ) * ( V + 1 ) + iv,
191  iu * ( V + 1 ) + ( iv + 1 ) ) );
192  indices.push_back( Vector3ui( ( iu + 1 ) * ( V + 1 ) + iv,
193  ( iu + 1 ) * ( V + 1 ) + ( iv + 1 ),
194  iu * ( V + 1 ) + ( iv + 1 ) ) );
195  }
196  }
197  }
198 
199  result.setVertices( std::move( vertices ) );
200  result.setNormals( std::move( normals ) );
201  result.setIndices( std::move( indices ) );
202  if ( generateTexCoord )
203  result.addAttrib( getAttribName( MeshAttrib::VERTEX_TEXCOORD ), std::move( texCoords ) );
204  if ( color ) result.colorize( *color );
205  result.checkConsistency();
206 
207  return result;
208 }
