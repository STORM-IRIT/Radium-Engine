152 void SkeletonComponent::update( Scalar t ) {
153  m_wasReset = Core::Math::areApproxEqual( t, 0_ra );
154  if ( m_wasReset )
155  {
156  m_animationTime = t;
157  m_skel.setPose( m_refPose, SpaceType::LOCAL );
158 
159  updateDisplay();
160  return;
161  }
162 
163  m_animationTime = m_speed * t;
164  Scalar lastTime = 0;
165  for ( auto boneAnim : m_animations[m_animationID] )
166  {
167  lastTime = std::max( lastTime, *boneAnim.getTimes().rbegin() );
168  }
169  if ( m_autoRepeat )
170  {
171  if ( !m_pingPong ) { m_animationTime = std::fmod( m_animationTime, lastTime ); }
172  else
173  {
174  m_animationTime = std::fmod( m_animationTime, 2 * lastTime );
175  if ( m_animationTime > lastTime ) { m_animationTime = 2 * lastTime - m_animationTime; }
176  }
177  }
178  else if ( m_pingPong )
179  {
180  if ( m_animationTime > 2 * lastTime ) { m_animationTime = 0_ra; }
181  else if ( m_animationTime > lastTime )
182  { m_animationTime = 2 * lastTime - m_animationTime; }
183  }
184 
185  // get the current pose from the animation
186  Core::Animation::Pose pose = m_skel.getPose( SpaceType::LOCAL );
187  if ( !m_animations.empty() )
188  {
189 #pragma omp parallel for
190  for ( int i = 0; i < int( m_animations[m_animationID].size() ); ++i )
191  {
192  pose[uint( i )] = m_animations[m_animationID][uint( i )].at(
193  m_animationTime, Core::Animation::linearInterpolate<Core::Transform> );
194  }
195  }
196  else
197  { pose = m_refPose; }
198  m_skel.setPose( pose, SpaceType::LOCAL );
199 
200  updateDisplay();
201 }

360 void SkeletonComponent::updateDisplay() {
361  for ( auto& bone : m_boneDrawables )
362  {
363  uint boneIdx = m_boneMap.at( bone->getIndex() );
364  Core::Vector3 start;
365  Core::Vector3 end;
366  m_skel.getBonePoints( boneIdx, start, end );
367 
368  Core::Transform scale = Core::Transform::Identity();
369  scale.scale( ( end - start ).norm() );
370 
371  Core::Quaternion rot =
372  Core::Quaternion::FromTwoVectors( Core::Vector3::UnitZ(), end - start );
373 
374  Core::Transform boneTransform = m_skel.getTransform( boneIdx, SpaceType::MODEL );
375  Core::Matrix3 rotation = rot.toRotationMatrix();
376  Core::Transform drawTransform;
377  drawTransform.linear() = rotation;
378  drawTransform.translation() = boneTransform.translation();
379 
380  bone->setLocalTransform( drawTransform * scale );
381  }
