220 void RenderObject::render( const Data::RenderParameters& lightParams,
221  const Data::ViewingParameters& viewParams,
222  const Data::ShaderProgram* shader,
223  const Data::RenderParameters& shaderParams ) {
224  if ( !m_visible || !shader ) { return; }
225  // Radium V2 : avoid this temporary
226  Core::Matrix4 modelMatrix = getTransformAsMatrix();
227  Core::Matrix4 normalMatrix = modelMatrix.inverse().transpose();
228  // bind data
229  shader->bind();
230  shader->setUniform( "transform.proj", viewParams.projMatrix );
231  shader->setUniform( "transform.view", viewParams.viewMatrix );
232  shader->setUniform( "transform.model", modelMatrix );
233  shader->setUniform( "transform.worldNormal", normalMatrix );
234  lightParams.bind( shader );
235  shaderParams.bind( shader );
236  // FIXME : find another solution for FrontFacing polygons (depends on the camera matrix)
237  // This is a hack to allow correct face culling
238  // Note that this hack implies the inclusion of OpenGL.h in this file
239  if ( viewParams.viewMatrix.determinant() < 0 ) { glFrontFace( GL_CW ); }
240  else
241  { glFrontFace( GL_CCW ); }
