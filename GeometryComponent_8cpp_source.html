113 void PointCloudComponent::finalizeROFromGeometry( const Core::Asset::MaterialData* data,
114  Core::Transform transform ) {
115  // The technique for rendering this component
116  std::shared_ptr<Material> roMaterial;
117  // First extract the material from asset or create a default one
118  if ( data != nullptr )
119  {
120  auto converter = EngineMaterialConverters::getMaterialConverter( data->getType() );
121  auto mat = converter.second( data );
122  roMaterial.reset( mat );
123  }
124  else
125  {
126  auto mat = new BlinnPhongMaterial( m_contentName + "_DefaultBPMaterial" );
127  mat->m_renderAsSplat = m_displayMesh->getNumFaces() == 0;
128  mat->m_perVertexColor =
129  m_displayMesh->getCoreGeometry().hasAttrib( Mesh::getAttribName( Mesh::VERTEX_COLOR ) );
130  roMaterial.reset( mat );
131  }
132  // initialize with a default rendertechique that draws nothing
133  std::string roName( m_name + "_" + m_contentName + "_RO" );
134  auto ro = RenderObject::createRenderObject(
135  roName, this, RenderObjectType::Geometry, m_displayMesh, RenderTechnique {} );
136  ro->setTransparent( roMaterial->isTransparent() );
137  ro->setMaterial( roMaterial );
138  setupIO( m_contentName );
139  ro->setLocalTransform( transform );
140  m_roIndex = addRenderObject( ro );
