125 void PointCloudComponent::finalizeROFromGeometry( const Core::Asset::MaterialData* data,
126  Core::Transform transform ) {
127  // The technique for rendering this component
128  std::shared_ptr<Data::Material> roMaterial;
129  // First extract the material from asset or create a default one
130  if ( data != nullptr )
131  {
132  auto converter = Data::EngineMaterialConverters::getMaterialConverter( data->getType() );
133  auto mat = converter.second( data );
134  roMaterial.reset( mat );
135  }
136  else
137  {
138  auto mat = new Data::BlinnPhongMaterial( m_contentName + "_DefaultBPMaterial" );
139  mat->m_renderAsSplat = m_displayMesh->getNumFaces() == 0;
140  mat->m_perVertexColor = m_displayMesh->getCoreGeometry().hasAttrib(
141  Data::Mesh::getAttribName( Data::Mesh::VERTEX_COLOR ) );
142  roMaterial.reset( mat );
143  }
144  // initialize with a default rendertechique that draws nothing

147  this,

150  Rendering::RenderTechnique {} );
151  ro->setTransparent( roMaterial->isTransparent() );
152  ro->setMaterial( roMaterial );
