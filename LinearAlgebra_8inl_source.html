132 // Reference : Paolo Baerlocher, Inverse Kinematics techniques for interactive posture control
133 // of articulated figures (PhD Thesis), p.138. EPFL, 2001.
134 // http://www0.cs.ucl.ac.uk/research/equator/papers/Documents2002/Paolo%20Baerlocher_Thesis_2001.pdf
135 inline void getSwingTwist( const Quaternion& in, Quaternion& swingOut, Quaternion& twistOut ) {
136  // singular case.
137  if ( UNLIKELY( in.z() == 0 && in.w() == 0 ) )
138  {
139  twistOut = in;
140  swingOut.setIdentity();
141  }
142  else
143  {
144  const Scalar gamma = std::atan2( in.z(), in.w() );
145  // beta = atan2 ( sqrt ( in.x^2 + in.y^2), sqrt ( in.z^2 + in.w^2))
146  const Scalar beta =
147  std::atan2( in.coeffs().head<2>().norm(), in.coeffs().tail<2>().norm() );
148 
149  // k = 2 / sinc(beta) = 2 / ( sin(beta)/ beta))
150  const Scalar k = 2.f * beta / std::sin( beta );
151 
152  // The parentheses here are important. because otherwise k * rotation2D would
153  // be evaluated first (which would yield the rotation of angle k * gamma).
154  const Vector2 sxy = k * ( Eigen::Rotation2D<Scalar>( gamma ) * in.coeffs().head<2>() );
155 
156  AngleAxis twist( 2 * gamma, Vector3::UnitZ() );
157  twistOut = twist;
158 
159  Vector3 swingAxis( Vector3::Zero() );
160  swingAxis.head<2>() = sxy.normalized();
161 
162  AngleAxis swing( sxy.norm(), swingAxis );
163  swingOut = swing;
164  }
165 }

168 inline Matrix4 lookAt( const Vector3& position, const Vector3& target, const Vector3& up ) {
169  Matrix3 R;
170  R.col( 2 ) = ( position - target ).normalized();
171  R.col( 0 ) = up.cross( R.col( 2 ) ).normalized();
172  R.col( 1 ) = R.col( 2 ).cross( R.col( 0 ) );
173 
174  Matrix4 result = Matrix4::Zero();

176  result.topRightCorner<3, 1>() = -R.transpose() * position;
177  result( 3, 3 ) = 1.0;

179  return result;

183  Scalar theta = fovy * 0.5f;
