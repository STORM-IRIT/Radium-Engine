75 void Camera::updateProjMatrix() {
76 
77  switch ( m_projType )
78  {
79  case ProjType::ORTHOGRAPHIC: {
80  const Scalar dx = m_zoomFactor * .5_ra;
81  const Scalar dy = dx / m_aspect;
82  // ------------
83  // Compute projection matrix as describe in the doc of gluPerspective()
84  const Scalar l = -dx; // left
85  const Scalar r = dx; // right
86  const Scalar t = dy; // top
87  const Scalar b = -dy; // bottom
88 
89  Core::Vector3 tr( -( r + l ) / ( r - l ),
90  -( t + b ) / ( t - b ),
91  -( ( m_zFar + m_zNear ) / ( m_zFar - m_zNear ) ) );
92 
93  m_projMatrix.setIdentity();
94 
95  m_projMatrix.coeffRef( 0, 0 ) = 2_ra / ( r - l );
96  m_projMatrix.coeffRef( 1, 1 ) = 2_ra / ( t - b );
97  m_projMatrix.coeffRef( 2, 2 ) = -2_ra / ( m_zFar - m_zNear );
98  m_projMatrix.block<3, 1>( 0, 3 ) = tr;
99  }
100  break;
101 
102  case ProjType::PERSPECTIVE: {
103  // Compute projection matrix as describe in the doc of gluPerspective()
104  const Scalar f = std::tan( ( PiDiv2 ) - ( m_fov * m_zoomFactor * .5_ra ) );
105  const Scalar diff = m_zNear - m_zFar;
106 
107  m_projMatrix.setZero();
108 
109  m_projMatrix.coeffRef( 0, 0 ) = f / m_aspect;
110  m_projMatrix.coeffRef( 1, 1 ) = f;
111  m_projMatrix.coeffRef( 2, 2 ) = ( m_zFar + m_zNear ) / diff;
112  m_projMatrix.coeffRef( 2, 3 ) = ( 2_ra * m_zFar * m_zNear ) / diff;
113  m_projMatrix.coeffRef( 3, 2 ) = -1_ra;
114  }
115  break;
116 
117  default:
118  break;
119  }

140 void Camera::fitZRange( const Core::Aabb& aabb ) {
141  const auto& minAabb = aabb.min();
142  const auto& maxAabb = aabb.max();
143  const auto& position = m_frame.translation();
144  Ra::Core::Vector3 direction = m_frame.linear() * Ra::Core::Vector3( 0_ra, 0_ra, -1_ra );
145 
146  m_zNear = m_zFar = direction.dot( minAabb - position );
147 
148  auto adaptRange = [position, direction, this]( Scalar x, Scalar y, Scalar z ) {
149  Ra::Core::Vector3 corner( x, y, z );
150  auto d = direction.dot( corner - position );
151  this->m_zNear = std::min( d, this->m_zNear );
152  this->m_zFar = std::max( d, this->m_zFar );
153  };
154 
155  adaptRange( minAabb[0], minAabb[1], maxAabb[2] );
156  adaptRange( minAabb[0], maxAabb[1], minAabb[2] );
157  adaptRange( minAabb[0], maxAabb[1], maxAabb[2] );
158  adaptRange( maxAabb[0], maxAabb[1], maxAabb[2] );
159  adaptRange( maxAabb[0], maxAabb[1], minAabb[2] );
160  adaptRange( maxAabb[0], minAabb[1], maxAabb[2] );
161  adaptRange( maxAabb[0], minAabb[1], minAabb[2] );
162

165  m_zNear = std::max( range, m_zNear - range );
