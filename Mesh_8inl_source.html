505 void PolyMesh::triangulate() {
506  m_triangleIndices.clear();
507  m_triangleIndices.reserve( m_mesh.getIndices().size() );
508  for ( const auto& face : m_mesh.getIndices() )
509  {
510  if ( face.size() == 3 ) { m_triangleIndices.push_back( face ); }
511  else
512  {
514  int minus {int( face.size() ) - 1};
515  int plus {0};
516  while ( plus + 1 < minus )
517  {
518  if ( ( plus - minus ) % 2 )
519  {
520  m_triangleIndices.emplace_back( face[plus], face[plus + 1], face[minus] );
521  ++plus;
522  }
523  else
524  {
525  m_triangleIndices.emplace_back( face[minus], face[plus], face[minus - 1] );
526  --minus;
527  }
