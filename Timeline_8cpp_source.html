371 void Timeline::onMovingKeyFrames( size_t first, Scalar offset ) {
372  if ( m_current.m_value )
373  {
374  auto times = m_current.m_value->getTimes();
375  const Scalar time = times[first];
376  if ( offset < 0 )
377  {
378  for ( int i = 0; i < int( times.size() ); ++i )
379  {
380  const Scalar t = times[i];
381  if ( t >= time ) { m_current.m_value->moveKeyFrame( i, t + offset ); }
382  }
383  }
384  else // go from the end to ensure we do not mess up keyframes
385  {
386  for ( int i = times.size() - 1; i >= 0; --i )
387  {
388  const Scalar t = times[i];
389  if ( t >= time ) { m_current.m_value->moveKeyFrame( i, t + offset ); }
390  }
391  }
392  m_current.updateKeyFrame( time );
393  emit keyFramesMoved( first, offset );
394  }

397 void Timeline::on_comboBox_attribute_currentIndexChanged( const QString& arg1 ) {
398  onClearKeyFrames();
399  std::vector<Ra::Core::Animation::KeyFramedValueController> list;
400  const QStringList names = arg1.split( "::" );
401 
402  // prints a warning message for the attribute and returns
403 #define LOG_AND_RETURN \
404  LOG( logWARNING ) << "[Timeline] Error: attribute \"" << arg1.toStdString() \
405  << "\"'s name is not conform."; \
406  return
407 
408  // Checks if the given vector of KeyFramedValues contains a KeyFramedValue
409  // with the given name.
410  // If so, set the current frame to this KeyFramedValue;
411  // If not, prints a warning message for the attribute and returns.
412 #define GET_KEYFRAMEDVALUE( list, name ) \
413  { \
414  auto it = std::find_if( list.begin(), list.end(), [&name]( const auto& frame ) { \
415  return frame.m_name == name; \
416  } ); \
417  if ( it == list.end() ) { LOG_AND_RETURN; } \
418  m_current = *it; \
419  }
420 
421  // Checks if the given map has an element fulfilling lambda.
422  // If so, set list to the associated vector of KeyFramedValues.
423  // If not, prints a warning message for the attribute and returns.
424 #define GET_KEYFRAMEDVALUE_LIST( map, lambda ) \
425  { \
426  auto keyFramedValues = std::find_if( map.begin(), map.end(), lambda ); \
427  if ( keyFramedValues == map.end() ) { LOG_AND_RETURN; } \
428  list = keyFramedValues->second; \
429  }
430 
431  switch ( names.size() )
432  {
433  case 2: {
434  const std::string entityName = names.at( 0 ).toStdString();
435  const std::string frameName = names.at( 1 ).toStdString();
436  auto lambda = [entityName]( const auto& frame ) {
437  return frame.first->getName() == entityName;
438  };
439  GET_KEYFRAMEDVALUE_LIST( m_entityKeyFrames, lambda );
440  GET_KEYFRAMEDVALUE( list, frameName );
441  }
442  break;
443  case 3: {
444  const std::string compName = names.at( 1 ).toStdString();
445  const std::string frameName = names.at( 2 ).toStdString();
446  auto lambda = [compName]( const auto& frame ) {

448  };

450  GET_KEYFRAMEDVALUE( list, frameName );
451  }

453  case 4: {

455  bool ok;
