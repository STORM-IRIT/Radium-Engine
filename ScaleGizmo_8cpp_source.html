160  getControler( i )->clearState();

163

171  }
172 
173  // T
174  if ( m_selectedAxis == -1 && m_selectedPlane == -1 && !whole ) { return m_transform; }
175 
176  // Get gizmo center and translation axis / plane normal
177  int dir = whole ? 0 : std::max( m_selectedAxis, m_selectedPlane );
178  const Core::Vector3 origin = m_transform.translation();
179  Core::Vector3 scaleDir = Core::Vector3( m_transform.rotation() * Core::Vector3::Unit( dir ) );
180 
181  // Project the clicked points against the axis defined by the scale axis,
182  // or the planes defined by the scale plane.
183  std::vector<Scalar> hits;
184  bool found = false;
185  Core::Vector3 endPoint;
186  if ( whole )

188  else if ( m_selectedAxis > -1 )
189  { found = findPointOnAxis( cam, origin, scaleDir, m_initialPix + nextXY, endPoint, hits ); }
190  else if ( m_selectedPlane >= 0 )
191  { found = findPointOnPlane( cam, origin, scaleDir, m_initialPix + nextXY, endPoint, hits ); }
192  if ( !found ) { return m_transform; }

194  // Initialize scale

196  {
197  m_start = true;
198  m_startPos = m_transform.translation();
199  m_prevScale = Core::Vector3::Ones();
200  m_startPoint = endPoint;
201  }
202 
203  // Prevent scale == 0
204  const Scalar a = ( endPoint - m_startPos ).squaredNorm();
205  if ( a < 1e-3_ra ) { return m_transform; }
206 
207  // Get scale value
208  const Core::Vector3 b = m_startPoint - m_startPos;
209  Scalar scale = std::sqrt( a ) / b.norm();
210  if ( stepped ) { scale = int( scale / step + 1 ) * step; }
211

214  if ( whole ) { m_prevScale = scale * Core::Vector3::Ones(); }
215  else if ( m_selectedAxis > -1 )
216  {
217  m_prevScale = Core::Vector3::Ones();
218  m_prevScale[m_selectedAxis] = scale;
219  }
220  else if ( m_selectedPlane >= 0 )
221  {
222  m_prevScale = Core::Vector3::Ones();
223  m_prevScale[( m_selectedPlane + 1 ) % 3] = scale;
224  m_prevScale[( m_selectedPlane + 2 ) % 3] = scale;
225  }
226  m_transform.scale( m_prevScale );
