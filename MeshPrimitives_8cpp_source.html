648  const Vector3& tip,
649  Scalar radius,
650  uint nFaces,
651  const Utils::optional<Utils::Color>& color ) {
652  TriangleMesh result;
653  TriangleMesh::PointAttribHandle::Container vertices;
654  TriangleMesh::NormalAttribHandle::Container normals;
655  TriangleMesh::IndexContainerType indices;
656  vertices.reserve( 2 + nFaces );
657  normals.reserve( 2 + nFaces );
658  indices.reserve( 2 * nFaces );
659 
660  Vector3 ab = tip - base;
661  Vector3 dir = ab.normalized();
662 
663  // Create two circles normal centered on A and B and normal to ab;
664  Vector3 xPlane, yPlane;
665  Math::getOrthogonalVectors( dir, xPlane, yPlane );
666  xPlane.normalize();
667  yPlane.normalize();
668 
669  vertices.push_back( base );
670  vertices.push_back( tip );
671  normals.push_back( -dir );
672  normals.push_back( dir );
673 
674  const Scalar thetaInc( Core::Math::PiMul2 / Scalar( nFaces ) );
675  for ( uint i = 0; i < nFaces; ++i )
676  {
677  const Scalar theta = i * thetaInc;
678  Vector3 normal = std::cos( theta ) * xPlane + std::sin( theta ) * yPlane;
679 
680  vertices.push_back( base + radius * normal );
681  normals.push_back( ( normal - dir ).normalized() );
682  }
683 
684  for ( uint i = 0; i < nFaces; ++i )
685  {
686  uint bl = i + 2; // bottom left corner of face
687  uint br = ( ( i + 1 ) % nFaces ) + 2; // bottom right corner of face
688 
689  indices.emplace_back( 0, br, bl );
690  indices.emplace_back( 1, bl, br );
691  }
692  result.setVertices( std::move( vertices ) );
693  result.setNormals( std::move( normals ) );
694  result.setIndices( std::move( indices ) );
695  if ( bool( color ) ) result.colorize( *color );
696  result.checkConsistency();
697 
698  return result;

701 TriangleMesh makePlaneGrid( const uint rows,
702  const uint cols,
703  const Vector2& halfExts,
704  const Transform& T,
705  const Utils::optional<Utils::Color>& color ) {
706  TriangleMesh result;
707  TriangleMesh::PointAttribHandle::Container vertices;
708  TriangleMesh::NormalAttribHandle::Container normals;
709  TriangleMesh::IndexContainerType indices;
710 
711  const uint R = ( rows + 1 );
712  const uint C = ( cols + 1 );
713  const uint v_size = C * R;
714  const uint t_size = 2 * cols * rows;
715 
716  vertices.resize( v_size );
717  normals.resize( v_size );
718  indices.reserve( t_size );
719 
720  const Vector3 X = T.linear().col( 0 ).normalized();
721  const Vector3 Y = T.linear().col( 1 ).normalized();
722  const Vector3 Z = T.linear().col( 2 ).normalized();
723 
724  const Vector3 x = ( 2_ra * halfExts[0] * X ) / Scalar( cols );
725  const Vector3 y = ( 2_ra * halfExts[1] * Y ) / Scalar( rows );
726  const Vector3 o = T.translation() - ( halfExts[0] * X ) - ( halfExts[1] * Y );
727 
728  Grid<uint, 2> v( {R, C} );
729  for ( uint i = 0; i < R; ++i )
730  {
731  for ( uint j = 0; j < C; ++j )
732  {
733  const uint id = ( i * C ) + j;
734  v.at( {i, j} ) = id;
735  vertices[id] = o + ( i * y ) + ( j * x );
736  normals[id] = Z;
737  }
738  }
739 
740  for ( uint i = 0; i < rows; ++i )
741  {
742  for ( uint j = 0; j < cols; ++j )
743  {
744  indices.emplace_back( v.at( {i, j} ), v.at( {i, j + 1} ), v.at( {i + 1, j + 1} ) );
745  indices.emplace_back( v.at( {i, j} ), v.at( {i + 1, j + 1} ), v.at( {i + 1, j} ) );
746  }
747  }
748  result.setVertices( std::move( vertices ) );
749  result.setNormals( std::move( normals ) );
750  result.setIndices( std::move( indices ) );
751  if ( bool( color ) ) result.colorize( *color );
752  result.checkConsistency();
753 
754  return result;
755 }
