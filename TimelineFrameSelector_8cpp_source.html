479  else

492  return Ra::Core::Math::areApproxEqual( t, m_cursor );

494  // if already on keyFrame, move current KeyFrame
495  // ------------------- CURSOR ON KEYFRAME -----------------------
496  if ( it != m_keyFrames.end() )

505  return Ra::Core::Math::areApproxEqual( t, nearest );

508  { onMoveKeyFrame( m_cursor, nearest ); }

515  --itLeft;

517  if ( nearest > left ) { onMoveKeyFrames( m_cursor, nearest - m_cursor ); }
518  }
519  }
520  // ---------------- CURSOR NOT ON KEYFRAME --------------------------
521  else
522  {
523  // if shiftdown, slide first right KeyFrame to the left

528  // if KeyFrames on the right, remove or insert time

531  }

533  // ---------------- MOVE LEFT KEYFRAME TO THE RIGHT -----------
534  else

538  // if KeyFrames on the left, remove or insert time

541  }
542  }
543  }
544  // no catch mouse event

612 Scalar TimelineFrameSelector::nearestStep( Scalar time ) const {
613  Scalar deltaT =
614  TIMELINE_AUTO_SUGGEST_CURSOR_RADIUS / m_timelineUI->m_scrollArea->getPixPerSec();
615 
616  Scalar minDist = Scalar( m_timelineUI->m_durationSpin->maximum() );
617  Scalar dist;
618 
619  Scalar newCursor = time;
620 
621  for ( Scalar keyFrame : m_keyFrames )
622  {
623  dist = qAbs( keyFrame - time );
624  if ( dist < deltaT && dist < minDist )
625  {
626  minDist = dist;
627  newCursor = keyFrame;
628  }
629  if ( time < keyFrame ) break;
630  }
631 
632  Scalar step = m_timelineUI->m_scrollArea->getStep();
633  for ( int i = 0; i < m_timelineUI->m_scrollArea->getNbInterval() - 1; ++i )
634  {
635  Scalar pos = i * step;
636  dist = std::abs( pos - time );
637  if ( dist < deltaT && dist < minDist )
638  {
